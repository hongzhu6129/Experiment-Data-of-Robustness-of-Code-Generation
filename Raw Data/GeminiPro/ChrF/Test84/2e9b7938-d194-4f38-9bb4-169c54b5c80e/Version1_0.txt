```java
public class StringArray {

    public static void main(String[] args) {
        int n = 10; // Example: Create an array of length 10

        if (n <= 0) {
            System.out.println("Array length must be greater than 0.");
            return;
        }

        String[] stringArray = new String[n];

        for (int i = 0; i < n; i++) {
            stringArray[i] = String.valueOf(i); // Convert integer to string
        }


        // Print the array (optional) to verify output
        System.out.print("Created array: [");
        for (int i = 0; i < n; i++) {
            System.out.print(stringArray[i]);
            if (i < n - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }
}
```
