public class FirstUniqueChar {

    public static void main(String[] args) {
        String str = "wresource";
        int index = first_Uniq_Char(str);
        System.out.println("Original String: " + str);
        System.out.println("First unique character of the above: " + index);


        str = "abcabcde";
        index = first_Uniq_Char(str);
        System.out.println("Original String: " + str);
        System.out.println("First unique character of the above: " + index);


        str = "aabbccddeeffg";
        index = first_Uniq_Char(str);
        System.out.println("Original String: " + str);
        System.out.println("First unique character of the above: " + index);

    }

    public static int first_Uniq_Char(String str) {
        int[] charCounts = new int[26]; // Assuming lowercase English alphabet

        // Count the occurrences of each character
        for (char c : str.toCharArray()) {
            charCounts[c - 'a']++;
        }

        // Find the first character with a count of 1
        for (int i = 0; i < str.length(); i++) {
            if (charCounts[str.charAt(i) - 'a'] == 1) {
                return i;
            }
        }

        return -1; // Should not reach here if there's at least one unique character (as per problem statement) 
    }
}
