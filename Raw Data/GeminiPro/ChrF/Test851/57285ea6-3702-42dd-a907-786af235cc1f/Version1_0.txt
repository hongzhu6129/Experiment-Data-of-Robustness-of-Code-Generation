```java
public class SwapVariables {

    public static void main(String[] args) {
        int a = 10;
        int b = 20;

        System.out.println("Before swap:");
        System.out.println("a = " + a);
        System.out.println("b = " + b);

        // Method 1: Using a temporary variable
        int temp = a;
        a = b;
        b = temp;


        System.out.println("\nAfter swap (using temp variable):");
        System.out.println("a = " + a);
        System.out.println("b = " + b);


        // Reset values for demonstration of other methods
        a = 10;
        b = 20;

        // Method 2: Using addition and subtraction  (Only for numbers.  Susceptible to overflow/underflow.)
        a = a + b;
        b = a - b;
        a = a - b;

        System.out.println("\nAfter swap (using addition/subtraction):");
        System.out.println("a = " + a);
        System.out.println("b = " + b);


        // Reset values for demonstration of other methods
        a = 10;
        b = 20;


        // Method 3: Using bitwise XOR (Only for numbers)
        a = a ^ b;
        b = a ^ b;
        a = a ^ b;


        System.out.println("\nAfter swap (using XOR):");
        System.out.println("a = " + a);
        System.out.println("b = " + b);

    }
}
```
