```java
import java.util.Scanner;

public class CountDivisible {

    public static int function(int x, int y, int n) {
        int count = 0;
        for (int i = x; i <= y; i++) {
            if (i % n == 0) {
                count++;
            }
        }
        return count;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter the start of the range (x): ");
        int x = scanner.nextInt();

        System.out.print("Enter the end of the range (y): ");
        int y = scanner.nextInt();

        System.out.print("Enter the divisor (n): ");
        int n = scanner.nextInt();

        int result = function(x, y, n);
        System.out.println("The number of values divisible by " + n + " in the range [" + x + ", " + y + "] is: " + result);

        scanner.close();
    }
}
```
