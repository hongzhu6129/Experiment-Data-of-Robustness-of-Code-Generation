public class PentagonalNumbers {

    public static void main(String[] args) {
        // Display the first 50 pentagonal numbers
        final int NUMBER_OF_PENTAGONAL_NUMBERS = 50;
        int count = 0;  // Counter to track how many numbers we've printed

        for (int n = 1; count < NUMBER_OF_PENTAGONAL_NUMBERS; n++) {
            System.out.printf("%7d ", getPentagonalNumber(n));  // Format for consistent spacing
            count++;
            if (count % 10 == 0) { // Print 10 numbers per line
                System.out.println();
            }
        }
    }


    public static int getPentagonalNumber(int n) {
        return n * (3 * n - 1) / 2;
    }
}
