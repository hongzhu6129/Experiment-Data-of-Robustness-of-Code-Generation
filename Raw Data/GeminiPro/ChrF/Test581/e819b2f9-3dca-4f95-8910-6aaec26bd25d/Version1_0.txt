```java
import java.util.LinkedList;
import java.util.List;

public class InsertFirstLast {

    public static void main(String[] args) {
        // Create a LinkedList
        List<String> myList = new LinkedList<>();

        // Insert elements at the beginning
        myList.add(0, "Apple"); // Insert "Apple" at index 0 (beginning)
        myList.add(0, "Banana"); // Insert "Banana" at index 0 (beginning), shifting "Apple" to index 1

        // Insert elements at the end
        myList.add("Orange"); // Adds "Orange" to the end of the list
        myList.add("Grape");  // Adds "Grape" to the end of the list

        // Print the list
        System.out.println(myList); // Output: [Banana, Apple, Orange, Grape]


        // Example using an Integer list:
        List<Integer> numbers = new LinkedList<>();

        numbers.add(0, 10);
        numbers.add(0, 20);
        numbers.add(30);
        numbers.add(40);

        System.out.println(numbers); // Output: [20, 10, 30, 40]
    }
}
```
