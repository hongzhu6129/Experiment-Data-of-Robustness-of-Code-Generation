```java
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.time.LocalTime;

public class LocalDateTimeMixer {

    public static void main(String[] args) {
        // Example usage:

        // Create a LocalDate object
        LocalDate date = LocalDate.of(2024, 3, 15); 

        // Create a LocalTime object
        LocalTime time = LocalTime.of(10, 30, 0); // 10:30 AM

        // Combine date and time into a LocalDateTime object
        LocalDateTime dateTime = LocalDateTime.of(date, time);

        // Print the LocalDateTime object
        System.out.println("Combined LocalDateTime: " + dateTime);


        // Another example, directly creating the LocalDateTime
        LocalDateTime dateTime2 = LocalDateTime.of(2024, 3, 15, 14, 45, 15); // March 15, 2024, 2:45:15 PM
        System.out.println("Directly created LocalDateTime: " + dateTime2);
    }
}
```
