```java
public class StringFromCharArray {

    public static void main(String[] args) {

        // Method 1: Directly using the String constructor
        char[] charArray1 = {'H', 'e', 'l', 'l', 'o', ' ', 'W', 'o', 'r', 'l', 'd'};
        String str1 = new String(charArray1);
        System.out.println("String 1: " + str1);


        // Method 2: Specifying a range of characters from the array
        char[] charArray2 = {'J', 'a', 'v', 'a', ' ', 'P', 'r', 'o', 'g', 'r', 'a', 'm'};
        String str2 = new String(charArray2, 5, 7); // Start at index 5, take 7 characters
        System.out.println("String 2: " + str2);

        //Method 3: Using valueOf()
        char[] charArray3 = {'E','x','a','m','p','l','e'};
        String str3 = String.valueOf(charArray3);
        System.out.println("String 3: " + str3);


    }
}
```
