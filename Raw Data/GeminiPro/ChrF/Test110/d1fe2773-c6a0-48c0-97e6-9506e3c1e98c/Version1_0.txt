class Solution {
    /**
     * Searches for a value (target) in an m x n matrix.
     *
     * @param matrix The m x n matrix to search in.
     * @param target The value to search for.
     * @return True if the target is found in the matrix, false otherwise.
     */
    public boolean searchMatrix(int[][] matrix, int target) {
        if (matrix == null || matrix.length == 0 || matrix[0].length == 0) {
            return false;
        }

        int rows = matrix.length;
        int cols = matrix[0].length;

        // Treat the matrix as a sorted 1D array
        int left = 0;
        int right = rows * cols - 1;

        while (left <= right) {
            int mid = left + (right - left) / 2; // Avoid potential overflow
            int row = mid / cols;
            int col = mid % cols;

            if (matrix[row][col] == target) {
                return true;
            } else if (matrix[row][col] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        return false;
    }


    public static void main(String[] args) {
        Solution solution = new Solution();

        int[][] matrix1 = {{1, 3, 5, 7}, {10, 11, 16, 20}, {23, 30, 34, 60}};
        int target1 = 3;
        System.out.println(solution.searchMatrix(matrix1, target1)); // Output: true

        int[][] matrix2 = {{1, 3, 5, 7}, {10, 11, 16, 20}, {23, 30, 34, 60}};
        int target2 = 13;
        System.out.println(solution.searchMatrix(matrix2, target2)); // Output: false


        // Edge cases:
        int[][] emptyMatrix = {};
        int target3 = 5;
        System.out.println(solution.searchMatrix(emptyMatrix, target3)); // Output: false


        int[][] nullMatrix = null;
        int target4 = 5;
        System.out.println(solution.searchMatrix(nullMatrix, target4)); // Output: false

         int[][] matrix3 = {{1}}; // Single element matrix
        int target5 = 1;
        System.out.println(solution.searchMatrix(matrix3, target5)); // Output: true

        int[][] matrix4 = {{1}}; // Single element matrix, target not found
        int target6 = 2;
        System.out.println(solution.searchMatrix(matrix4, target6)); // Output: false
    }
}
