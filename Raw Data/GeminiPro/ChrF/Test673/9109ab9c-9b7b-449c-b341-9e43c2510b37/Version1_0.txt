```java
import java.util.TreeMap;

public class RemoveGreatestKey {

    public static void main(String[] args) {
        // Example usage:
        TreeMap<Integer, String> map = new TreeMap<>();
        map.put(1, "One");
        map.put(5, "Five");
        map.put(3, "Three");
        map.put(2, "Two");
        map.put(4, "Four");

        System.out.println("Original map: " + map);

        removeGreatestKey(map);

        System.out.println("Map after removing greatest key: " + map);


        // Another example with an empty map
        TreeMap<Integer, String> emptyMap = new TreeMap<>();
        removeGreatestKey(emptyMap);
        System.out.println("Empty map after operation: " + emptyMap); //Should still be empty


        // Another example with a single element map
         TreeMap<Integer, String> singleElementMap = new TreeMap<>();
         singleElementMap.put(7, "Seven");
         removeGreatestKey(singleElementMap);
         System.out.println("Single element map after operation: " + singleElementMap); //Should be empty

    }

    public static void removeGreatestKey(TreeMap<Integer, String> map) {
        if (!map.isEmpty()) {
            map.remove(map.lastKey());
        }
    }
}
```
