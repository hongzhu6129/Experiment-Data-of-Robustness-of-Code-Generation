import java.util.TreeMap;

public class RemoveGreatestKeyEntry {

    public static void main(String[] args) {
        // Create a TreeMap to store key-value pairs. TreeMap automatically sorts entries by key.
        TreeMap<Integer, String> map = new TreeMap<>();

        // Add some sample entries
        map.put(1, "One");
        map.put(5, "Five");
        map.put(2, "Two");
        map.put(10, "Ten");
        map.put(7, "Seven");


        if (!map.isEmpty()) {
            // Get the greatest key (lastKey() because TreeMap is sorted)
            Integer greatestKey = map.lastKey();

            // Get the value associated with the greatest key before removing it
            String value = map.get(greatestKey);

            // Remove the entry with the greatest key
            map.remove(greatestKey);

            // Print the removed key-value pair
            System.out.println("Removed Entry: Key = " + greatestKey + ", Value = " + value);


            // Print the updated map (optional)
            System.out.println("Updated Map: " + map);
        } else {
            System.out.println("The map is empty.");
        }
    }
}
