public class LucasNumbers {

    public static void main(String[] args) {
        int n = 10;
        int a = 2;
        int b = 1;

        System.out.print("First " + n + " Lucas numbers: ");

        if (n > 0) {
            System.out.print(a + " "); 
        }
        if (n > 1) {
             System.out.print(b + " ");
        }

        for (int i = 2; i < n; i++) {
            int c = a + b;
            System.out.print(c + " ");
            a = b;
            b = c;
        }
        System.out.println(); // Add a newline for better formatting.

    }
}
