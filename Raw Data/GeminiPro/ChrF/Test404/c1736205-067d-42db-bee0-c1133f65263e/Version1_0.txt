class Solution {
    /**
     * Checks if a specified character is happy in a string.
     * A character is happy if the same character appears to its left or right in the string.
     *
     * @param str The string to check.
     * @return True if the character is happy, false otherwise.
     */
    public boolean aCharacterIsHappy(String str) {
        if (str == null || str.length() < 2) {
            return false; // Single character or empty string cannot be happy
        }

        for (int i = 0; i < str.length(); i++) {
            char currentChar = str.charAt(i);

            // Check left neighbor (if exists)
            if (i > 0 && str.charAt(i - 1) == currentChar) {
                return true;
            }

            // Check right neighbor (if exists)
            if (i < str.length() - 1 && str.charAt(i + 1) == currentChar) {
                return true;
            }
        }

        return false; // No happy character found
    }


    public static void main(String[] args) {
        Solution solution = new Solution();

        String str1 = "aabbccddee";
        System.out.println("Is a character happy in '" + str1 + "'? " + solution.aCharacterIsHappy(str1)); // True

        String str2 = "abcde";
        System.out.println("Is a character happy in '" + str2 + "'? " + solution.aCharacterIsHappy(str2)); // False

        String str3 = "a";
        System.out.println("Is a character happy in '" + str3 + "'? " + solution.aCharacterIsHappy(str3)); // False

        String str4 = "";
        System.out.println("Is a character happy in '" + str4 + "'? " + solution.aCharacterIsHappy(str4)); // False


        String str5 = "aabbc";
        System.out.println("Is a character happy in '" + str5 + "'? " + solution.aCharacterIsHappy(str5)); // True

        String str6 = "abacc";
        System.out.println("Is a character happy in '" + str6 + "'? " + solution.aCharacterIsHappy(str6)); // True


    }
}
