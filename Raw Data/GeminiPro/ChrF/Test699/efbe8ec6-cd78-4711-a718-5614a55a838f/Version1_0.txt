import java.util.Arrays;

public class SieveOfEratosthenes {

    private static int[] generatePrimes(int n) {
        boolean[] isPrime = new boolean[n + 1];
        Arrays.fill(isPrime, true); // Assume all numbers are prime initially

        isPrime[0] = isPrime[1] = false; // 0 and 1 are not prime

        for (int p = 2; p * p <= n; p++) {
            if (isPrime[p]) {
                // Mark multiples of p as not prime
                for (int i = p * p; i <= n; i += p) {
                    isPrime[i] = false;
                }
            }
        }

        // Count the number of primes
        int primeCount = 0;
        for (int i = 2; i <= n; i++) {
            if (isPrime[i]) {
                primeCount++;
            }
        }

        // Create an array to store primes
        int[] primes = new int[primeCount];
        int index = 0;
        for (int i = 2; i <= n; i++) {
            if (isPrime[i]) {
                primes[index++] = i;
            }
        }

        return primes;
    }

    public static void main(String[] args) {
        int n = 30; // Example: Find primes up to 30
        int[] primes = generatePrimes(n);

        System.out.println("Prime numbers up to " + n + " are:");
        for (int prime : primes) {
            System.out.print(prime + " ");
        }
        System.out.println();



        // Print all numbers divisible by any of the found primes
        System.out.println("Numbers up to " + n + " divisible by at least one prime:");

        for (int i = 1; i <= n; i++) {
            boolean divisible = false;
            for (int prime : primes) {
                if (i % prime == 0) {
                    divisible = true;
                    break; // Optimization: No need to check further if divisible by one prime
                }
            }
            if (divisible) {  // If the number was divisible by at least one prime
                System.out.print(i + " ");
            }

        }
        System.out.println();




    }
}
