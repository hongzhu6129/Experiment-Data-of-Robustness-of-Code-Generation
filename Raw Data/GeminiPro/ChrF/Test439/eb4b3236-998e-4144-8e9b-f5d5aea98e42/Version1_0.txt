```java
import java.util.Calendar;
import java.util.GregorianCalendar;

public class DefaultCalendarInfo {

    public static void main(String[] args) {
        // Get a default calendar instance (GregorianCalendar is a subclass of Calendar)
        Calendar calendar = new GregorianCalendar();

        // Get year, month, day of week (to check for weekend), hour, and minute
        int year = calendar.get(Calendar.YEAR);
        int month = calendar.get(Calendar.MONTH); // Note: Month is 0-indexed (0=January, 1=February, etc.)
        int dayOfWeek = calendar.get(Calendar.DAY_OF_WEEK); // 1=Sunday, 2=Monday, ..., 7=Saturday
        int hour = calendar.get(Calendar.HOUR_OF_DAY); // 24-hour format
        int minute = calendar.get(Calendar.MINUTE);

        // Display the information
        System.out.println("Year: " + year);
        System.out.println("Month: " + (month + 1)); // Add 1 to display month in the usual 1-12 format.
        System.out.println("Weekend: " + (dayOfWeek == Calendar.SATURDAY || dayOfWeek == Calendar.SUNDAY)); // Check if it's Saturday or Sunday
        System.out.println("Hour: " + hour);
        System.out.println("Minute: " + minute); 
    }
}
```
