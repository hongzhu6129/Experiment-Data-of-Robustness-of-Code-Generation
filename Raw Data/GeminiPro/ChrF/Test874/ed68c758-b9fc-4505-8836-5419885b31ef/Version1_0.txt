public class CapitalizeFirstLetter {

    public static void main(String[] args) {
        String sentence = "the quick brown fox jumps past the lazy dog. the quick brown fox jumps over the lazy dog.";
        String capitalizedSentence = "";

        // Split the sentence into words based on spaces and periods.
        String[] words = sentence.split("(?<=\\b\\.)|\\s+");  // Split on spaces or periods with word boundary

        for (String word : words) {
            if (!word.isEmpty()) {  // Handle potential empty strings from multiple spaces
                // Capitalize the first letter and append the rest of the word.
                capitalizedSentence += word.substring(0, 1).toUpperCase() + word.substring(1) + " ";
            }
        }

        // Remove the trailing space and print the capitalized sentence.
        System.out.println(capitalizedSentence.trim()); 
    }
}
