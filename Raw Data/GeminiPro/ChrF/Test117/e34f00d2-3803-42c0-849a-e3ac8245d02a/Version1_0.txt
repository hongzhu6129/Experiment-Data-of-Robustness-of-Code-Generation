public class IntegerToString {

    public static String transform_int_to_string(int n) {
        return String.valueOf(n);
    }

    public static void main(String[] args) {
        int inputInteger = 35; // You can change this value for different inputs
        System.out.println("input an integer: " + inputInteger);
        String stringRepresentation = transform_int_to_string(inputInteger);
        System.out.println("string format of the said integer: " + stringRepresentation); 
    }
}
