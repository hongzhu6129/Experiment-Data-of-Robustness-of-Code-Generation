```java
public class CheckForG {

    public static void main(String[] args) {
        String word1 = "programming";
        String word2 = "hello";
        String word3 = "language";

        System.out.println("'" + word1 + "' contains 'g': " + validate(word1));
        System.out.println("'" + word2 + "' contains 'g': " + validate(word2));
        System.out.println("'" + word3 + "' contains 'g': " + validate(word3));

    }

    public static String validate(String word) {
        if (word.contains("g")) {
            return "true";
        } else {
            return "false";
        }
    }
}
```
