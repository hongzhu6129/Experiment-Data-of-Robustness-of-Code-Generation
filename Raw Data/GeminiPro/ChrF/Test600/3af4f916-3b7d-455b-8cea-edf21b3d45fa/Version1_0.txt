```java
import java.util.LinkedList;

public class CheckLinkedListEmpty {

    public static void main(String[] args) {
        // Create a LinkedList
        LinkedList<Integer> list1 = new LinkedList<>();

        // Check if the list is empty
        if (list1.isEmpty()) {
            System.out.println("Linked List is empty");
        } else {
            System.out.println("Linked List is not empty"); 
        }

        // Add an element to the list
        list1.add(10);

        // Check again if the list is empty
        if (list1.isEmpty()) {
            System.out.println("Linked List is empty");
        } else {
            System.out.println("Linked List is not empty");
        }


         //Another example with a String LinkedList
        LinkedList<String> list2 = new LinkedList<>();
        list2.add("Hello");
        list2.add("World");

        if (list2.isEmpty()) {
            System.out.println("String Linked List is empty");
        } else {
            System.out.println("String Linked List is not empty");
        }


    }
}
```
