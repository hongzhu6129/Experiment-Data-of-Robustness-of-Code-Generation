```java
public class StringSeparator {

    public static void main(String[] args) {
        String inputString = "HelloWorld"; // Example input
        String result = validate(inputString);
        System.out.println(result);

        inputString = "programming";
        result = validate(inputString);
        System.out.println(result);

        inputString = "aeiou"; // All vowels
        result = validate(inputString);
        System.out.println(result);


        inputString = "rhythm"; // No vowels (traditional sense)
        result = validate(inputString);
        System.out.println(result);


        inputString = ""; // Empty String
        result = validate(inputString);
        System.out.println(result);




    }

    public static String validate(String inputString) {
        if (inputString == null) {
            return "Invalid Input (null)"; // Handle null input
        }

        StringBuilder consonants = new StringBuilder();
        StringBuilder vowels = new StringBuilder();

        inputString = inputString.toLowerCase(); // Case-insensitive comparison

        for (int i = 0; i < inputString.length(); i++) {
            char ch = inputString.charAt(i);
            if (Character.isLetter(ch)) { // Consider only letters
                if ("aeiou".indexOf(ch) != -1) {
                    vowels.append(ch);
                } else {
                    consonants.append(ch);
                }
            }
        }

        return "Consonants: " + consonants.toString() + ", Vowels: " + vowels.toString();
    }
}
```
