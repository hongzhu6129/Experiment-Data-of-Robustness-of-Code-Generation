```java
import java.util.TreeSet;

public class LowerValueInTreeSet {

    public static void main(String[] args) {
        // Example usage:
        TreeSet<Integer> treeSet = new TreeSet<>();
        treeSet.add(10);
        treeSet.add(20);
        treeSet.add(30);
        treeSet.add(40);
        treeSet.add(50);

        Integer givenElement = 35;

        Integer lowerElement = treeSet.lower(givenElement);

        if (lowerElement != null) {
            System.out.println("Element lower than " + givenElement + ": " + lowerElement);
        } else {
            System.out.println("No element lower than " + givenElement + " found.");
        }



        givenElement = 10; // Test with the smallest element

        lowerElement = treeSet.lower(givenElement);

        if (lowerElement != null) {
            System.out.println("Element lower than " + givenElement + ": " + lowerElement);
        } else {
            System.out.println("No element lower than " + givenElement + " found.");
        }



        givenElement = 60;  // Test with an element greater than all in the set

        lowerElement = treeSet.lower(givenElement);

        if (lowerElement != null) {
            System.out.println("Element lower than " + givenElement + ": " + lowerElement);
        } else {
            System.out.println("No element lower than " + givenElement + " found.");
        }
    }
}
```
