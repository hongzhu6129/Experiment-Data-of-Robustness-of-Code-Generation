```java
import java.util.TreeSet;

public class TreeSetLower {

    public static void main(String[] args) {
        // Create a TreeSet and populate it with some data
        TreeSet<Integer> treeSet = new TreeSet<>();
        treeSet.add(10);
        treeSet.add(20);
        treeSet.add(30);
        treeSet.add(40);
        treeSet.add(50);

        // Element to find the lower value of
        Integer element = 40;

        // Get the element with a lower value
        Integer lower = treeSet.lower(element);

        // Print the result
        if (lower != null) {
            System.out.println("Lower element of " + element + " is: " + lower);
        } else {
            System.out.println("No lower element found for " + element);
        }


        // Example with a String TreeSet
        TreeSet<String> stringTreeSet = new TreeSet<>();
        stringTreeSet.add("apple");
        stringTreeSet.add("banana");
        stringTreeSet.add("grape");
        stringTreeSet.add("orange");

        String strElement = "grape";
        String strLower = stringTreeSet.lower(strElement);

        if (strLower != null) {
            System.out.println("Lower element of " + strElement + " is: " + strLower);
        } else {
            System.out.println("No lower element found for " + strElement);
        }


          // Example demonstrating when no lower element exists
        Integer nonExistentElement = 10; // Smallest element in the set
        Integer lowerNonExistent = treeSet.lower(nonExistentElement);

        if (lowerNonExistent != null) {
            System.out.println("Lower element of " + nonExistentElement + " is: " + lowerNonExistent);
        } else {
            System.out.println("No lower element found for " + nonExistentElement);
        }

    }
}
```
