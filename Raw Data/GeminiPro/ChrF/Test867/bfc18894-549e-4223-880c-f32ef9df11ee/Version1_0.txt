```java
public class Main {
    public static void main(String[] args) {
        System.out.println("a\ta^2\ta^3\ta^4"); // Print header row with tabs for alignment

        for (int a = 1; a <= 4; a++) {
            System.out.print(a + "\t"); // Print 'a' value and a tab
            System.out.print((int) Math.pow(a, 2) + "\t");  // Print a^2 and a tab
            System.out.print((int) Math.pow(a, 3) + "\t");  // Print a^3 and a tab
            System.out.println((int) Math.pow(a, 4));      // Print a^4 and a newline
        }
    }
}
```
