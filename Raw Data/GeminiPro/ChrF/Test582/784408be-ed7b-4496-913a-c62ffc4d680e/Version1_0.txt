```java
import java.util.LinkedList;

public class InsertAtFrontLinkedList {

    public static void main(String[] args) {
        // Create a LinkedList
        LinkedList<String> books = new LinkedList<>();

        // Add some initial elements (optional)
        books.add("Book 2");
        books.add("Book 3");

        // Element to insert at the front
        String newBook = "Book 1";

        // Insert the element at the front using addFirst()
        books.addFirst(newBook);

        // Print the updated LinkedList
        System.out.println("Updated LinkedList: " + books);


        // Example with integers
         LinkedList<Integer> numbers = new LinkedList<>();
        numbers.add(2);
        numbers.add(3);

        int newNumber = 1;
        numbers.addFirst(newNumber);

        System.out.println("Updated LinkedList of numbers: " + numbers);

    }
}
```
