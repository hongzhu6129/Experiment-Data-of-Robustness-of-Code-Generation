public class CheckHashMapEmpty {

        // Check if the HashMap is empty using isEmpty()
        boolean isEmpty = hashMap.isEmpty();

        if (isEmpty) {
            System.out.println("The HashMap is empty.");
        } else {
            System.out.println("The HashMap is not empty.");
        }

        // Add some elements to the HashMap
        hashMap.put("Apple", 1);
        hashMap.put("Banana", 2);
        hashMap.put("Orange", 3);

        // Check again if the HashMap is empty
        isEmpty = hashMap.isEmpty();

        if (isEmpty) {
            System.out.println("The HashMap is empty.");
        } else {
            System.out.println("The HashMap is not empty.");
        }
    }
}
