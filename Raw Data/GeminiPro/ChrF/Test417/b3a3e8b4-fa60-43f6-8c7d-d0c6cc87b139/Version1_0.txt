public class StringContains {

    public static void main(String[] args) {
        String mainString = "This is a test string";
        String subString1 = "test";
        String subString2 = "not present";

        System.out.println("\"" + subString1 + "\" is present in \"" + mainString + "\": " + is_present(mainString, subString1));
        System.out.println("\"" + subString2 + "\" is present in \"" + mainString + "\": " + is_present(mainString, subString2));
    }

    public static boolean is_present(String mainString, String subString) {
        return mainString.contains(subString);
    }
}
