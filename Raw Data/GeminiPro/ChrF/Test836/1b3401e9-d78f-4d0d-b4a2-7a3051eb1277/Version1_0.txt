```java
public class Main {
    public static void main(String[] args) {
        // Swapping two variables
        int a = 10;
        int b = 20;

        System.out.println("Before swapping:");
        System.out.println("a = " + a);
        System.out.println("b = " + b);

        // Swapping using a temporary variable
        int temp = a;
        a = b;
        b = temp;

        System.out.println("\nAfter swapping:");
        System.out.println("a = " + a);
        System.out.println("b = " + b);


        // Printing a face
        System.out.println("\nPrinting a face:");
        System.out.println(" +\"\"\"\"\"+ ");
        System.out.println("[| o o |]");
        System.out.println(" |  ^  | ");
        System.out.println(" | '-' | ");
        System.out.println(" +-----+ ");


        //Swapping without temp variable (using addition and subtraction)
        int x = 5;
        int y = 15;
        System.out.println("\nBefore swapping without temp:");
        System.out.println("x = " + x);
        System.out.println("y = " + y);


        x = x + y;  // x now becomes 20
        y = x - y;  // y becomes 5
        x = x - y;  // x becomes 15

        System.out.println("\nAfter swapping without temp:");
        System.out.println("x = " + x);
        System.out.println("y = " + y);

          //Swapping without temp variable (using multiplication and division - ONLY if neither is 0)
        int p = 7;
        int q = 21;
        System.out.println("\nBefore swapping without temp (multiplication):");
        System.out.println("p = " + p);
        System.out.println("q = " + q);


        p = p * q;  
        q = p / q;
        p = p / q;

        System.out.println("\nAfter swapping without temp (multiplication):");
        System.out.println("p = " + p);
        System.out.println("q = " + q);



    }
}
```
