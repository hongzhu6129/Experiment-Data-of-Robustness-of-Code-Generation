public class AmericanFlag {

    public static void main(String[] args) {
        // Print the top part of the flag (stars and stripes)
        for (int i = 0; i < 5; i++) {
            // Print the stars section
            for (int j = 0; j < 6; j++) {
                System.out.print("* ");
            }
            if (i == 2) {
                System.out.print("pct ");
            } else {
                System.out.print("    "); // Add spaces for alignment.
            }

            System.out.print("==");
            for (int j = 0; j < 22; j++) {
                System.out.print("=");
            }
            System.out.println();
        }

        // Print the bottom part of the flag (stripes only)
        for (int i = 0; i < 6; i++) {
            System.out.print("==");
            for (int j = 0; j < 32; j++) {
                System.out.print("=");
            }
            System.out.println();
        }
    }
}
