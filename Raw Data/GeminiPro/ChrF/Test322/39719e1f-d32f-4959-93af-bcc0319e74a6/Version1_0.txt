public class StringComparison {

    public static void main(String[] args) {
        String str1 = "hello";
        String str2 = "world";

        int result = str1.compareTo(str2);

        if (result < 0) {
            System.out.println(str1 + " is lexicographically less than " + str2);
        } else if (result > 0) {
            System.out.println(str1 + " is lexicographically greater than " + str2);
        } else {
            System.out.println(str1 + " is lexicographically equal to " + str2);
        }


        // Example demonstrating lexicographical equality check:
        String str3 = "apple";
        String str4 = "apple";
        String str5 = "Apple"; // Note the capital 'A'

        if (str3.compareTo(str4) == 0) {
            System.out.println(str3 + " is lexicographically equal to " + str4);
        } else {
            System.out.println(str3 + " is not lexicographically equal to " + str4);
        }


        if (str3.compareTo(str5) == 0) {
            System.out.println(str3 + " is lexicographically equal to " + str5);
        } else {
            System.out.println(str3 + " is not lexicographically equal to " + str5); // This will be printed due to case sensitivity
        }


        //  Ignoring case:
         if (str3.compareToIgnoreCase(str5) == 0) {
            System.out.println(str3 + " (ignoring case) is lexicographically equal to " + str5);
        } else {
            System.out.println(str3 + " (ignoring case) is not lexicographically equal to " + str5); 
        }


    }
}
