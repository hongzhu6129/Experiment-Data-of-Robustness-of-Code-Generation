```java
import java.util.HashMap;

public class HashTableSize {

    public static void main(String[] args) {
        // Create a HashMap
        HashMap<String, Integer> hashMap = new HashMap<>();

        // Add some elements
        hashMap.put("apple", 1);
        hashMap.put("banana", 2);
        hashMap.put("orange", 3);
        hashMap.put("grape", 4);


        // Get the number of elements (size)
        int size = hashMap.size();

        // Print the size
        System.out.println("The number of elements in the HashMap is: " + size);


        // Example with an empty HashMap
        HashMap<String, Integer> emptyHashMap = new HashMap<>();
        System.out.println("The number of elements in the empty HashMap is: " + emptyHashMap.size());


    }
}
```
