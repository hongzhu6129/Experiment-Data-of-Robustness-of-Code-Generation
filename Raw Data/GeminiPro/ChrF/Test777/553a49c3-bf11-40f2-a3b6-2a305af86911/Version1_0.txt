```java
public class Rectangle {

    public static void main(String[] args) {
        double width = 5.6;
        double height = 8.5;

        double area = width * height;
        double perimeter = 2 * (width + height);

        System.out.println("area is 5.6 * 8.5 = " + String.format("%.2f", area));
        System.out.println("perimeter is 2 * (5.6 + 8.5) = " + String.format("%.2f", perimeter)); 
    }
}
```
