public class ColorPicker {

    public String pickColor(String input) {
        if (input != null && input.startsWith("red")) {
            return "red";
        } else if (input != null && input.startsWith("black")) {
            return "black";
        } else {
            return "";
        }
    }

    public static void main(String[] args) {
        ColorPicker cp = new ColorPicker();

        System.out.println(cp.pickColor("reddish")); // Output: red
        System.out.println(cp.pickColor("blackish")); // Output: black
        System.out.println(cp.pickColor("blue")); // Output: 
        System.out.println(cp.pickColor(null)); // Output: 
        System.out.println(cp.pickColor("")); // Output:

    }
}
