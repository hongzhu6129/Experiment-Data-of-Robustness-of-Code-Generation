```java
import java.util.ArrayList;
import java.util.List;

public class CheckArrayListEmpty {

    public static void main(String[] args) {

        // Example 1: Empty ArrayList
        List<String> list1 = new ArrayList<>();
        System.out.println("Is list1 empty? " + list1.isEmpty()); // Output: true

        // Example 2: ArrayList with elements
        List<Integer> list2 = new ArrayList<>();
        list2.add(1);
        list2.add(2);
        list2.add(3);
        System.out.println("Is list2 empty? " + list2.isEmpty()); // Output: false

        // Example 3: Another way to check for emptiness (using size())
        List<Double> list3 = new ArrayList<>();
        System.out.println("Is list3 empty? " + (list3.size() == 0)); // Output: true


    }
}
```
