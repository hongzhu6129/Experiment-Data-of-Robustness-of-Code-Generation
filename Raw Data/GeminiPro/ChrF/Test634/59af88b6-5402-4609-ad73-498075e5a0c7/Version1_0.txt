```java
import java.util.PriorityQueue;

public class ClearPriorityQueue {

    public static void main(String[] args) {
        // Create a PriorityQueue
        PriorityQueue<Integer> pq = new PriorityQueue<>();

        // Add some elements to the PriorityQueue
        pq.add(10);
        pq.add(5);
        pq.add(20);
        pq.add(15);

        System.out.println("Original PriorityQueue: " + pq);

        // Discard all elements from the PriorityQueue
        pq.clear();

        System.out.println("PriorityQueue after clearing: " + pq); 
    }
}
```
