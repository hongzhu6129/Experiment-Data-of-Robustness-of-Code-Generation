public class CubeChecker {

    public static void main(String[] args) {
        // Input data (hardcoded for demonstration, ideally should be taken from user input)
        int number = 8;

        // Check if the number is a cube
        boolean isCube = checkCube(number);

        // Print the result
        if (isCube) {
            System.out.println("Number is a cube.");
        } else {
            System.out.println("Number is not a cube.");
        }


         number = 27;  // Test another value

        // Check if the number is a cube
        isCube = checkCube(number);

        // Print the result
        if (isCube) {
            System.out.println("Number is a cube.");
        } else {
            System.out.println("Number is not a cube.");
        }


        number = 10;   //And a non-cube

        // Check if the number is a cube
        isCube = checkCube(number);

        // Print the result
        if (isCube) {
            System.out.println("Number is a cube.");
        } else {
            System.out.println("Number is not a cube.");
        }
    }


    // Helper function to check if a number is a perfect cube
    private static boolean checkCube(int n) {
        // Handle negative numbers (their cubes are negative)
        if (n < 0) {
            n = -n; // Make n positive for calculation purposes
        }

        // Efficiently check for cubes using Math.cbrt (cube root) and comparing to the nearest integer
        int cubeRoot = (int) Math.round(Math.cbrt(n)); // Calculate integer cube root
        return cubeRoot * cubeRoot * cubeRoot == n;    // Check if cube of integer root equals original n

    }

}
