        num = -8; // Cube of -2

        } else {
            System.out.println("Number is not a cube.");
        }

    }


    public static boolean isCube(int num) {
        // Efficiently check if a number is a perfect cube:

        // Handle 0 explicitly
        if (num == 0) return true;


        // For larger numbers, using cbrt (cube root) and checking if it's an integer is more efficient than looping.
        double cubeRoot = Math.cbrt(num);
        return Math.round(cubeRoot) * Math.round(cubeRoot) * Math.round(cubeRoot) == num;


    }
}
