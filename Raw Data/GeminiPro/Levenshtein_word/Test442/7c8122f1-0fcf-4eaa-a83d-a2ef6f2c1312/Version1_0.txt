```java
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;

public class CurrentTimeNewYork {

    public static void main(String[] args) {
        // Get the current time in New York
        ZoneId newYorkZoneId = ZoneId.of("America/New_York");
        ZonedDateTime nowNewYork = ZonedDateTime.now(newYorkZoneId);

        // Format the time (optional, but recommended for readability)
        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss z"); // Example format
        String formattedDateTime = nowNewYork.format(formatter);

        // Print the current time
        System.out.println("Current time in New York: " + formattedDateTime);


    }
}
```
