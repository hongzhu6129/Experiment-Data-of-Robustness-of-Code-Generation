class Solution {
    /**
     * Given an array of integers, every element appears twice except for one. Find that single one.
     *
     * @param nums The input array of integers.
     * @return The integer that appears only once.
     */
    public int getSingleNumber(int[] nums) {
        int result = 0;
        for (int num : nums) {
            result ^= num;
        }
        return result;
    }

    public static void main(String[] args) {
        Solution solution = new Solution();
        int[] nums1 = {2, 2, 1};
        int singleNumber1 = solution.getSingleNumber(nums1);
        System.out.println("Single number in nums1: " + singleNumber1); // Output: 1

        int[] nums2 = {4, 1, 2, 1, 2};
        int singleNumber2 = solution.getSingleNumber(nums2);
        System.out.println("Single number in nums2: " + singleNumber2); // Output: 4

        int[] nums3 = {1};
        int singleNumber3 = solution.getSingleNumber(nums3);
        System.out.println("Single number in nums3: " + singleNumber3); // Output: 1

        int[] nums4 = {}; // Empty array - this isn't specified in the problem but handling it for completeness
        int singleNumber4 = solution.getSingleNumber(nums4);
        System.out.println("Single number in nums4: " + singleNumber4); // Output: 0
    }
}
