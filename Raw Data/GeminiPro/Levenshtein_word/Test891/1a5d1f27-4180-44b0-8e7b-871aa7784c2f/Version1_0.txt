```java
import java.util.ArrayList;
import java.util.Arrays;

public class LongestWords {

    static ArrayList<String> longestWords(String[] dictionary) {
        ArrayList<String> longestWordsList = new ArrayList<>();
        int maxLength = 0;

        // Find the maximum length
        for (String word : dictionary) {
            if (word.length() > maxLength) {
                maxLength = word.length();
            }
        }

        // Add words with maximum length to the list
        for (String word : dictionary) {
            if (word.length() == maxLength) {
                longestWordsList.add(word);
            }
        }

        return longestWordsList;
    }

    public static void main(String[] args) {
        String[] dictionary1 = {"cat", "flag", "green", "country", "w3resource"};
        ArrayList<String> longestWords1 = longestWords(dictionary1);
        System.out.println(longestWords1); // Output: [w3resource]


        String[] dictionary2 = {"cat", "dog", "red", "is", "am"};
        ArrayList<String> longestWords2 = longestWords(dictionary2);
        System.out.println(longestWords2); // Output: [cat, dog, red]



         String[] dictionary3 = {}; // Empty dictionary
        ArrayList<String> longestWords3 = longestWords(dictionary3);
        System.out.println(longestWords3); // Output: []

         String[] dictionary4 = {null, "a", null, "bb"}; //Handles nulls gracefully
        ArrayList<String> longestWords4 = longestWords(dictionary4);
        System.out.println(longestWords4); // Output: [bb]



    }
}
```
