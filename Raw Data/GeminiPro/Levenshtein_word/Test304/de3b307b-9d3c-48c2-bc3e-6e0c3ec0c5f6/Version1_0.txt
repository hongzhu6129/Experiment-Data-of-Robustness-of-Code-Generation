public class MoveZeros {

    public static int[] move_zero(int[] nums) {
        if (nums == null || nums.length == 0) {
            return nums;
        }

        int nonZeroIndex = 0; // Keeps track of the next position for a non-zero element

        // Iterate through the array
        for (int i = 0; i < nums.length; i++) {
            // If the current element is not zero, move it to the next available non-zero position
            if (nums[i] != 0) {
                swap(nums, i, nonZeroIndex);
                nonZeroIndex++;
            }
        }

        return nums;
    }


    private static void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    public static void main(String[] args) {
        int[] arr1 = {0, 1, 0, 3, 12};
        int[] result1 = move_zero(arr1);
        System.out.print("Result 1: ");
        for (int num : result1) {
            System.out.print(num + " "); // Output: 1 3 12 0 0
        }
        System.out.println();


        int[] arr2 = {0};
        int[] result2 = move_zero(arr2);
        System.out.print("Result 2: ");
        for (int num : result2) {
            System.out.print(num + " "); // Output: 0
        }
        System.out.println();

        int[] arr3 = {}; // Empty array
        int[] result3 = move_zero(arr3);
        System.out.print("Result 3: ");
        for (int num : result3) {
            System.out.print(num + " ");  // Output: (nothing)
        }
        System.out.println();


        int[] arr4 = {1, 2, 3, 4}; // No zeros
        int[] result4 = move_zero(arr4);
        System.out.print("Result 4: ");
        for (int num : result4) {
            System.out.print(num + " ");  // Output: 1 2 3 4
        }
        System.out.println();

        int[] arr5 = {0, 0, 0, 0}; // All zeros
        int[] result5 = move_zero(arr5);
        System.out.print("Result 5: ");
        for (int num : result5) {
            System.out.print(num + " ");  // Output: 0 0 0 0
        }
        System.out.println();

    }
}
