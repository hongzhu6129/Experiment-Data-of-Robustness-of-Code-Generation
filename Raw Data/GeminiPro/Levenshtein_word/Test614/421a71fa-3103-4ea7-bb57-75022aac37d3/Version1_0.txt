```java
import java.util.TreeSet;

public class TreeSetColors {

    public static void main(String[] args) {
        // Create a TreeSet to store colors (Strings).
        // TreeSets automatically store elements in sorted order.
        TreeSet<String> colors = new TreeSet<>();

        // Add some colors to the TreeSet.
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Orange");


        // Print the TreeSet. The output will be sorted alphabetically.
        System.out.println(colors); 
    }
}
```
