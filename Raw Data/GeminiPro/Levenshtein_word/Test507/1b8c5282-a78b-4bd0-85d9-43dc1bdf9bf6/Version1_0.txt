    public static boolean isUgly(int num) {
        if (num <= 0) {
            return false;
        }

        while (num % 2 == 0) {
            num /= 2;
        }
        while (num % 3 == 0) {
            num /= 3;
        }
        while (num % 5 == 0) {
            num /= 5;
        }

        return num == 1;
    }


    // Optional helper function for testing multiple cases:
    public static void testUglyNumber(int num) {
        if (isUgly(num)) {
            System.out.println(num + ": it is an ugly number.");
        } else {
            System.out.println(num + ": it is not an ugly number.");
        }
    }
}
