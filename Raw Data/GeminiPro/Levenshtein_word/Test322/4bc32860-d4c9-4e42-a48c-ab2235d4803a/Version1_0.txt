public class StringComparison {

    public static void main(String[] args) {
        String str1 = "apple";
        String str2 = "banana";

        // Using compareTo() method for lexicographical comparison
        int result = str1.compareTo(str2);

        if (result < 0) {
            System.out.println(str1 + " comes before " + str2 + " lexicographically.");
        } else if (result > 0) {
            System.out.println(str1 + " comes after " + str2 + " lexicographically.");
        } else {
            System.out.println(str1 + " and " + str2 + " are lexicographically equal.");
        }



        // Example demonstrating lexicographical equality check:
        String str3 = "hello";
        String str4 = "hello";
        String str5 = "Hello"; // Note: Case-sensitive comparison

        if (str3.compareTo(str4) == 0) {
            System.out.println(str3 + " and " + str4 + " are lexicographically equal.");
        } else {
            System.out.println(str3 + " and " + str4 + " are not lexicographically equal.");
        }

        if (str3.compareTo(str5) == 0) {
            System.out.println(str3 + " and " + str5 + " are lexicographically equal.");
        } else {
            System.out.println(str3 + " and " + str5 + " are not lexicographically equal.");
        }


        // For case-insensitive comparison, use compareToIgnoreCase()
          if (str3.compareToIgnoreCase(str5) == 0) {
            System.out.println(str3 + " and " + str5 + " are lexicographically equal (ignoring case).");
        } else {
            System.out.println(str3 + " and " + str5 + " are not lexicographically equal (ignoring case).");
        }



    }
}
