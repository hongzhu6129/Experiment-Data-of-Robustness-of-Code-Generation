```java
import java.util.TreeSet;

public class NumbersLessThanSevenInTreeSet {

    public static void main(String[] args) {
        // Create a TreeSet to store integers.
        TreeSet<Integer> numbers = new TreeSet<>();

        // Add some numbers to the TreeSet (including some greater than and less than 7).
        numbers.add(9);
        numbers.add(2);
        numbers.add(5);
        numbers.add(12);
        numbers.add(1);
        numbers.add(7); // 7 itself is NOT less than 7
        numbers.add(6);


        // Find and print numbers less than 7.  headSet() is the most efficient way
        System.out.println("Numbers less than 7:");
        System.out.println(numbers.headSet(7));


        // Alternatively, you can iterate and check manually (less efficient)
        System.out.println("\nNumbers less than 7 (using iteration):");
        for (int number : numbers) {
            if (number < 7) {
                System.out.print(number + " ");
            }
        }
        System.out.println(); // Add a newline for better formatting


    }
}
```
