```java
import java.util.HashSet;

public class AppendToHashSet {

    public static void main(String[] args) {
        // Create a HashSet
        HashSet<String> hashSet = new HashSet<>();

        // Add some initial elements
        hashSet.add("Apple");
        hashSet.add("Banana");
        hashSet.add("Orange");

        // Element to append
        String elementToAppend = "Grape";

        // Append the element (HashSets don't have a specific "append" operation, 
        // so we just add it ï¿½ if it already exists, it won't be duplicated).
        hashSet.add(elementToAppend); 

        // Print the updated HashSet
        System.out.println("Updated HashSet: " + hashSet); 
    }
}
```
