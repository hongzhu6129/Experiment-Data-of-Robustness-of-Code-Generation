```java
public class MiddleCharacter {

    public static void main(String[] args) {
        System.out.println(middle("hello")); // l
        System.out.println(middle("java")); // av
        System.out.println(middle("coding")); // di
        System.out.println(middle("a")); // a
        System.out.println(middle("")); // 
        System.out.println(middle(null)); // 
    }


    public static String middle(String str) {
        if (str == null || str.isEmpty()) {
            return ""; // Handle null or empty strings
        }

        int length = str.length();
        int middleIndex = length / 2;

        if (length % 2 == 0) { // Even length
            if (middleIndex > 0) {
                return str.substring(middleIndex - 1, middleIndex + 1);
            } else {
                return str; // Handle the case where length is 0 or 1. 
            }

        } else { // Odd length
            return String.valueOf(str.charAt(middleIndex));
        }
    }
}
```
