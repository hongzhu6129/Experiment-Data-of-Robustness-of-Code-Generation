class Solution {
    /**
     * Displays the middle character(s) of a string.
     * If the string has an odd length, two middle characters are displayed.
     * If the string has an even length, one middle character is displayed.
     *
     * @param str The input string.
     * @return The middle character(s).
     */
    public static String middle(String str) {
        if (str == null || str.isEmpty()) {
            return ""; // Or handle empty/null strings as needed
        }

        int len = str.length();
        int middleIndex = len / 2;

        if (len % 2 == 0) { // Even length
            return String.valueOf(str.charAt(middleIndex -1 )) + String.valueOf(str.charAt(middleIndex));
        } else { // Odd length
            return String.valueOf(str.charAt(middleIndex));
        }
    }


    public static void main(String[] args) {
        String str1 = "hello";
        String str2 = "world";
        String str3 = "a";
        String str4 = ""; // Empty string
        String str5 = null; // Null string

        System.out.println("Middle of \"" + str1 + "\": " + middle(str1)); // Output: l
        System.out.println("Middle of \"" + str2 + "\": " + middle(str2)); // Output: rl
        System.out.println("Middle of \"" + str3 + "\": " + middle(str3)); // Output: a
        System.out.println("Middle of \"" + str4 + "\": " + middle(str4)); // Output: (empty string)
        System.out.println("Middle of \"" + str5 + "\": " + middle(str5)); // Output: (empty string - handles null) 
    }
}
