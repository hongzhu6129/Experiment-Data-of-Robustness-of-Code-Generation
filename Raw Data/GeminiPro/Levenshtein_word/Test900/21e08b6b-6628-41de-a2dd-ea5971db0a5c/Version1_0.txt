class TreeUtils {

    /**
     * Checks if tree t is a subtree of tree s.
     *
     * @param s The main tree.
     * @param t The potential subtree.
     * @return True if t is a subtree of s, false otherwise.
     */
    public boolean isSubtree(TreeNode s, TreeNode t) {
        if (t == null) {
            return true; // An empty tree is a subtree of any tree
        }
        if (s == null) {
            return false; // If s is null but t is not, t cannot be a subtree
        }

        if (areIdentical(s, t)) {
            return true; // s and t are identical, so t is a subtree
        }

        // Recursively check if t is a subtree of the left or right subtrees of s
        return isSubtree(s.left, t) || isSubtree(s.right, t);
    }


    /**
     * Checks if two trees are identical.
     *
     * @param s The first tree.
     * @param t The second tree.
     * @return True if the trees are identical, false otherwise.
     */
    private boolean areIdentical(TreeNode s, TreeNode t) {
        if (s == null && t == null) {
            return true; // Both trees are empty
        }
        if (s == null || t == null) {
            return false; // One tree is empty and the other is not
        }

        // Check if the current nodes and their subtrees are identical
        return (s.val == t.val) && areIdentical(s.left, t.left) && areIdentical(s.right, t.right);
    }



     public static void main(String[] args) {
        TreeUtils tu = new TreeUtils();


        // Example usage:
        // Tree s:     3
        //           / \
        //          4   5
        //         / \
        //        1   2
        TreeNode s = new TreeNode(3);
        s.left = new TreeNode(4);
        s.right = new TreeNode(5);
        s.left.left = new TreeNode(1);
        s.left.right = new TreeNode(2);

        // Tree t:     4
        //           / \
        //          1   2
        TreeNode t = new TreeNode(4);
        t.left = new TreeNode(1);
        t.right = new TreeNode(2);


        // Tree t2 (not a subtree): 4
        //                        / \
        //                       1   3
        TreeNode t2 = new TreeNode(4);
        t2.left = new TreeNode(1);
        t2.right = new TreeNode(3);


        System.out.println("Original strings: xxyz yxzx " + tu.isSubtree(s, t)); // Output: true
        System.out.println("Original strings: xxyz yxzx " +tu.isSubtree(s, t2)); // Output: false

    }
}

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode(int x) {
        val = x;
    }
}
