import java.util.ArrayList;
import java.util.List;

public class ColorArrayList {

    public static void main(String[] args) {
        // Create an ArrayList to store strings (colors)
        List<String> colors = new ArrayList<>();

        // Add some colors to the ArrayList
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Orange");

        // Print out the collection using an enhanced for loop (for-each loop)
        System.out.println("Colors in the ArrayList:");
        for (String color : colors) {
            System.out.println(color); 
        }


        // Alternatively, you can print using a standard for loop with get():
        /*
        System.out.println("\nColors using a standard for loop:");
        for (int i = 0; i < colors.size(); i++) {
            System.out.println(colors.get(i));
        }
        */
    }
}
