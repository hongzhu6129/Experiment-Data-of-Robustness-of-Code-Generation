import java.util.PriorityQueue;

public class LeastElementCountPriorityQueue {

    public static void main(String[] args) {
        // Example usage:
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        pq.add(5);
        pq.add(2);
        pq.add(8);
        pq.add(2); // Adding duplicate least element
        pq.add(1); // Least element
        pq.add(1); // Another duplicate least element
        pq.add(9);

        int leastElement = pq.peek(); // Get the least element (head of the queue)
        int count = 0;

        // Iterate through the PriorityQueue and count occurrences of the least element
        for (Integer element : pq) {
            if (element == leastElement) {
                count++;
            }
        }

        System.out.println("Least element: " + leastElement);
        System.out.println("Count of least element: " + count);
