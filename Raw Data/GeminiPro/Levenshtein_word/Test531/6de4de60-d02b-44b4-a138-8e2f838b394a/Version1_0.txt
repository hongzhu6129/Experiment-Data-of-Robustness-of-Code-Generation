public class PellSeries {

    public static void main(String[] args) {
        int n = 15; // Number of Pell numbers to print
        int a = 0;
        int b = 1;

        System.out.println("The first " + n + " Pell numbers are:");

        for (int i = 0; i < n; i++) {
            System.out.print(a + " ");

            int next = 2 * b + a;  // Calculate the next Pell number
            a = b;                 // Update 'a' for the next iteration
            b = next;              // Update 'b' for the next iteration
        }
    }
}
