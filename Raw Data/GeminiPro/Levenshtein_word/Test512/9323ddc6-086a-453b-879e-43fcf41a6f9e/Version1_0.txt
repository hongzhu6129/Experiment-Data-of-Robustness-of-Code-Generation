    public static boolean is_narc_dec_num(long num) {
        String numStr = String.valueOf(num);
        int n = numStr.length();
        long sum = 0;

        for (int i = 0; i < n; i++) {
            int digit = Character.getNumericValue(numStr.charAt(i));
            sum += Math.pow(digit, n);  // Correctly calculate the sum of powers
        }

        return sum == num; 
    }
}
