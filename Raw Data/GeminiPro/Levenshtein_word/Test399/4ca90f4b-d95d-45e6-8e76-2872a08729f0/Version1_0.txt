class Solution {
    /**
     * Creates a string containing every character twice of a given string.
     *
     * @param str The input string.
     * @return A new string with each character repeated twice.
     */
    public String repeatAllTwice(String str) {
        if (str == null) {
            return null; 
        }

        StringBuilder sb = new StringBuilder();
        for (char c : str.toCharArray()) {
            sb.append(c).append(c);
        }
        return sb.toString();
    }

    public static void main(String[] args) {
        Solution solution = new Solution();

        String test1 = "hello";
        String result1 = solution.repeatAllTwice(test1);
        System.out.println(result1); // Output: hheelllloo

        String test2 = "abc";
        String result2 = solution.repeatAllTwice(test2);
        System.out.println(result2); // Output: aabbcc

        String test3 = null;
        String result3 = solution.repeatAllTwice(test3);
        System.out.println(result3); // Output: null

         String test4 = "";
        String result4 = solution.repeatAllTwice(test4);
        System.out.println(result4); // Output: "" (empty string)
    }
}
