import java.util.Calendar;
import java.util.GregorianCalendar;

public class MinCalendarValues {

    public static void main(String[] args) {

        // Get a default calendar instance (typically represents the current date and time)
        Calendar calendar = new GregorianCalendar();

        // Get minimum values for different fields
        int minYear = calendar.getActualMinimum(Calendar.YEAR);
        int minMonth = calendar.getActualMinimum(Calendar.MONTH);
        int minWeekOfYear = calendar.getActualMinimum(Calendar.WEEK_OF_YEAR);
        int minDayOfMonth = calendar.getActualMinimum(Calendar.DAY_OF_MONTH); // "previous" likely means day of month


        System.out.println("Minimum Year: " + minYear);
        System.out.println("Minimum Month: " + (minMonth +1 ) ); // Month is 0-indexed (0-11), so add 1 for usual representation
        System.out.println("Minimum Week of Year: " + minWeekOfYear);
        System.out.println("Minimum Day of Month: " + minDayOfMonth);


    }
}
