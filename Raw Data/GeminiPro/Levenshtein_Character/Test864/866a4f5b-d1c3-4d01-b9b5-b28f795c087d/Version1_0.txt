public class ApproximatePi {

    public static void main(String[] args) {
        // Calculate the first approximation (up to 1/11)
        double sum1 = 1.0 - (1.0 / 3.0) + (1.0 / 5.0) - (1.0 / 7.0) + (1.0 / 9.0) - (1.0 / 11.0);
        double pi1 = 4.0 * sum1;

        // Calculate the second approximation (up to 1/13)
        double sum2 = 1.0 - (1.0 / 3.0) + (1.0 / 5.0) - (1.0 / 7.0) + (1.0 / 9.0) - (1.0 / 11.0) + (1.0 / 13.0);
        double pi2 = 4.0 * sum2;

        System.out.println("Approximation 1 (up to 1/11): " + pi1);
        System.out.println("Approximation 2 (up to 1/13): " + pi2);
    }
}
