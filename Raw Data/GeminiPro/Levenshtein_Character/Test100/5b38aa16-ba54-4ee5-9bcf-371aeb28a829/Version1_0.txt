```java
public class PowerOfFour {

    public static void main(String[] args) {
        int num1 = 64;
        int num2 = 6;
        int num3 = 1;
        int num4 = 0;
        int num5 = -4;

        System.out.println(num1 + ": " + isPowerOfFour(num1)); // Output: 64: true
        System.out.println(num2 + ": " + isPowerOfFour(num2)); // Output: 6: bogus
        System.out.println(num3 + ": " + isPowerOfFour(num3)); // Output: 1: true
        System.out.println(num4 + ": " + isPowerOfFour(num4)); // Output: 0: bogus
        System.out.println(num5 + ": " + isPowerOfFour(num5)); // Output: -4: bogus


    }

    public static String isPowerOfFour(int num) {
        if (num <= 0) {
            return "bogus";
        }

        while (num % 4 == 0) {
            num /= 4;
        }

        return (num == 1) ? "true" : "bogus";
    }
}
```
