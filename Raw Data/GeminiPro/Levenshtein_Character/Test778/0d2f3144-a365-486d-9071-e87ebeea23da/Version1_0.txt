```java
public class AmericanFlag {

    public static void main(String[] args) {
        for (int i = 0; i < 6; i++) {
            // Stars and Stripes rows
            for (int j = 0; j < 6; j++) {
                System.out.print("* ");
            }
            System.out.print(" "); // Space between stars and stripes
            for (int j = 0; j < 20; j++) {
                System.out.print("=");
            }
            System.out.println(); // New line
        }

        // Stripes only rows
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j < 30; j++) {
                System.out.print("=");
            }
            System.out.println(); // New line
        }
    }
}
```
