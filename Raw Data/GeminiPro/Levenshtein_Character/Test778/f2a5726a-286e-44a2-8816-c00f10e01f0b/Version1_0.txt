public class AmericanFlag {

    public static void main(String[] args) {
        // Star Section
        for (int i = 0; i < 6; i++) {
            // Stars
            for (int j = 0; j < 6; j++) {
                System.out.print("* ");
            }
            System.out.print("  ");

            // Stripes
            System.out.print("==================================");
            System.out.println();
        }


        // Stripe Section
        for (int i = 0; i < 6; i++) {
            // Stripes
            System.out.print("==================================");
            System.out.println();
        }
    }
}
