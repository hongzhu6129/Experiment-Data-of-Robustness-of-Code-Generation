    public static void main(String[] args) {
        // Print the top part of the flag (stars and stripes)
        for (int i = 0; i < 5; i++) {
            // Stars section
            for (int j = 0; j < 6; j++) {
                System.out.print("* ");
            }
            System.out.print(" "); // Space between stars and stripes

            // Stripes section
            for (int j = 0; j < 30; j++) {
                System.out.print("=");
            }
            System.out.println(); // Newline

            if (i==1) {
                System.out.println("* * pct * * ==================================");
            }
        }

             // Stripes section
