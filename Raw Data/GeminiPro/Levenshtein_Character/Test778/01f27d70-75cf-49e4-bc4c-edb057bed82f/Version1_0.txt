public class AmericanFlag {

    public static void main(String[] args) {
        // Stripes and stars section
        for (int i = 0; i < 9; i++) {
            if (i % 2 == 0) { // Star row
                System.out.print("* * * ");
                if (i < 6) {
                     System.out.print("* ");
                }
            } else { // Plain stripe row
                 System.out.print("      ");
                if(i < 6){
                     System.out.print("  ");
                }
            }

            System.out.println("=================================="); 
        }

        // Lower Stripes section
        for (int i = 0; i < 6; i++) {
            System.out.println("==================================");
        }
    }
}
