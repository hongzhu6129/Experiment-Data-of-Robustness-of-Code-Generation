public class AmericanFlag {

    public static void main(String[] args) {
        // Print the star section
        for (int i = 0; i < 5; i++) {
            for (int j = 0; j < 6; j++) {
                if (i % 2 == 0) { // Even rows have 6 stars
                  System.out.print("* ");
                } else if (j < 5) {  // Odd rows have 5 stars
                  System.out.print("* ");
                } 
            }
            if (i == 2) { // Add "pct" on the 3rd row of stars
                System.out.print("pct ");
            }
            System.out.print("=="); // Separator between stars and stripes
            for(int k = 0; k < 26; k++){
                System.out.print("=");
            }
            System.out.println(); 
        }

        // Print the stripe section
        for (int i = 0; i < 6; i++) {

    }
}
