        decimalNumber = 15; // Resetting for demonstration
        octalNumber = "";
        int remainder;

        while (decimalNumber > 0) {
            remainder = decimalNumber % 8;
            octalNumber = remainder + octalNumber; // Prepend the remainder
            decimalNumber /= 8;
        }
