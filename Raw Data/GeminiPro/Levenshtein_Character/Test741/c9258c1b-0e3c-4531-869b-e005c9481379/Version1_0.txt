import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class MatchG {

    public static void main(String[] args) {
        String[] words = {"green", "programming", "dog", "egg", "globe", "google"};

        for (String word : words) {
            String result = validate(word);
            System.out.println(word + ": " + result);
        }
    }

    public static String validate(String word) {
        // Regular expression to match 'g' not at the beginning or end
        Pattern pattern = Pattern.compile("^.+(g).+$");  // At least one character before 'g' and one after
        Matcher matcher = pattern.matcher(word);

        if (matcher.matches()) {
            return "Match";
        } else {
            return "No Match";
        }
    }
}
