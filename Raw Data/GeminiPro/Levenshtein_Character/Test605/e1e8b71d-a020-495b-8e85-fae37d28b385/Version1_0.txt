```java
import java.util.HashSet;

public class EmptyHashSet {

    public static void main(String[] args) {
        // Create a HashSet
        HashSet<String> hashSet = new HashSet<>();

        // Add some elements to the HashSet (optional, just for demonstration)
        hashSet.add("Apple");
        hashSet.add("Banana");
        hashSet.add("Orange");

        System.out.println("Original HashSet: " + hashSet);

        // Empty the HashSet using the clear() method
        hashSet.clear();

        System.out.println("HashSet after clearing: " + hashSet);


        //Alternative way to empty a HashSet is to create a new empty HashSet and assign to the variable:

        HashSet<String> anotherHashSet = new HashSet<>();
        anotherHashSet.add("one");
        anotherHashSet.add("two");

         System.out.println("anotherHashSet before clearing: " + anotherHashSet);


        anotherHashSet = new HashSet<>(); //creates a brand new empty hashset

        System.out.println("anotherHashSet after clearing: " + anotherHashSet);


    }
}
```
