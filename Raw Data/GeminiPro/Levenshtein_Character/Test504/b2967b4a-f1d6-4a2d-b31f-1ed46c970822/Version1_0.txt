```java
class Solution {
    /**
     * Given an integer, display its factors.
     */
    public static void main(String[] args) {
        test(12); // Example usage: Factors of 12
        test(30); // Example usage: Factors of 30
        test(7);  // Example usage: Factors of 7 (a prime number)
    }


    public static void test(int n) {
        System.out.print("Factors of " + n + ": ");
        for (int i = 1; i <= n; i++) {
            if (n % i == 0) {
                System.out.print(i + " ");
            }
        }
        System.out.println(); // Newline for better output formatting
    }
}
```
