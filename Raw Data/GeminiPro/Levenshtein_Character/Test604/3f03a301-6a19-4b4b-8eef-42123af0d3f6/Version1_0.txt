```java
import java.util.HashSet;

public class HashSetSize {

    public static void main(String[] args) {
        // Create a HashSet
        HashSet<String> hashSet = new HashSet<>();

        // Add some elements
        hashSet.add("Apple");
        hashSet.add("Banana");
        hashSet.add("Orange");
        hashSet.add("Grape");
        hashSet.add("Apple"); // Adding a duplicate - won't be added twice

        // Get the number of elements (size)
        int size = hashSet.size();

        // Print the size
        System.out.println("The number of elements in the HashSet is: " + size);
    }
}
```
