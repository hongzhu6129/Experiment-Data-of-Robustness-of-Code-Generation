public class PentagonalNumbers {

    public static void main(String[] args) {
        // Display the first 50 pentagonal numbers
        displayPentagonalNumbers(50);
    }

    public static int getPentagonalNumber(int n) {
        // Calculate the nth pentagonal number using the formula: n(3n-1)/2
        return n * (3 * n - 1) / 2;
    }

    public static void displayPentagonalNumbers(int count) {
        // Loop to calculate and display each pentagonal number
        for (int i = 1; i <= count; i++) {
            System.out.printf("%7d ", getPentagonalNumber(i));  // %7d formats output for alignment

            // Add a newline every 10 numbers for better formatting
            if (i % 10 == 0) {
                System.out.println();
            }
        }
    }
}
