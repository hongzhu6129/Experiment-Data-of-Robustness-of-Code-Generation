    public static void test(int[] arr1, int[] arr2, boolean expected) {
        boolean result = isFirstAndLastEqual(arr1, arr2);
        System.out.println("Arrays: " + printArray(arr1) + ", " + printArray(arr2));
        System.out.println("Expected: " + expected + ", Result: " + result);
        System.out.println(expected == result ? "PASS" : "FAIL");
        System.out.println();


    }

    public static boolean isFirstAndLastEqual(int[] arr1, int[] arr2) {

        if (arr1 == null || arr2 == null) {
           return arr1 == arr2; // true if both null, false otherwise.
        }

        if (arr1.length == 0 || arr2.length == 0) {
            return arr1.length == arr2.length; // true if both empty, false otherwise
        }


        return arr1[0] == arr2[0] && arr1[arr1.length - 1] == arr2[arr2.length - 1];
    }

    // Helper method to print arrays for clearer output
    public static String printArray(int[] arr) {
        if (arr == null) return "null";
        if(arr.length==0) return "[]";
        StringBuilder sb = new StringBuilder("[");
        for (int i = 0; i < arr.length; i++) {
            sb.append(arr[i]);
            if (i < arr.length - 1) {
                sb.append(", ");
            }
        }
        sb.append("]");
        return sb.toString();
    }
