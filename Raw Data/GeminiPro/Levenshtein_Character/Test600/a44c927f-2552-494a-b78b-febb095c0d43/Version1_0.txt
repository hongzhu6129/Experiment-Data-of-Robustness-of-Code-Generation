```java
import java.util.LinkedList;

public class LinkedListIsEmpty {

    public static void main(String[] args) {
        // Create a linked list
        LinkedList<String> list1 = new LinkedList<>();

        // Check if the list is empty
        if (list1.isEmpty()) {
            System.out.println("List 1 is empty.");
        } else {
            System.out.println("List 1 is not empty.");
        }

        // Add elements to the list
        list1.add("Apple");
        list1.add("Banana");
        list1.add("Orange");

        // Check if the list is empty again
        if (list1.isEmpty()) {
            System.out.println("List 1 is empty.");
        } else {
            System.out.println("List 1 is not empty.");
        }


        // Another Example with an initially populated list
        LinkedList<Integer> list2 = new LinkedList<>();
        list2.add(1);
        list2.add(2);

        if (list2.isEmpty()) {
            System.out.println("List 2 is empty.");
        } else {
            System.out.println("List 2 is not empty.");
        }

        // Clear the list and check again
        list2.clear(); 
        if (list2.isEmpty()) {
            System.out.println("List 2 is now empty.");
        } else {
            System.out.println("List 2 is not empty.");
        }


    }
}
```
