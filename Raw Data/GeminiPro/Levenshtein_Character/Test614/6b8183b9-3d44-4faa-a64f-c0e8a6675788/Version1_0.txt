```java
import java.util.TreeSet;

public class TreeSetExample {

    public static void main(String[] args) {
        // Create a TreeSet of Strings. TreeSet automatically sorts elements in natural order (alphabetical for Strings).
        TreeSet<String> colors = new TreeSet<>();

        // Add some colors
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Orange");

        // Print out the TreeSet
        System.out.println(colors); 
    }
}
```
