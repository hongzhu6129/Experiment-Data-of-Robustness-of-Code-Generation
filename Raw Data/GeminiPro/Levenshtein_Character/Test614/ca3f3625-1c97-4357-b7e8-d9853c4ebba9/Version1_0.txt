```java
import java.util.TreeSet;

public class TreeSetExample {

    public static void main(String[] args) {
        // Create a TreeSet to store colors (Strings). TreeSets automatically maintain elements in sorted order.
        TreeSet<String> colors = new TreeSet<>();

        // Add some colors to the TreeSet
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Orange");


        // Print the entire TreeSet (demonstrating sorted order)
        System.out.println("Colors in the TreeSet:");
        System.out.println(colors);


        // Check if "Green" is present and print a message accordingly
        if (colors.contains("Green")) {
            System.out.println("\nGreen is present in the TreeSet.");
        } else {
            System.out.println("\nGreen is not present in the TreeSet.");
        }

         //There is no concept of getting a 'green set' from a TreeSet.  Sets don't have subsets readily available.
         //If you wanted elements less than green (lexicographically)
        System.out.println("\nColors before Green:");
        System.out.println(colors.headSet("Green")); // Returns elements strictly less than "Green"

         //If you wanted elements greater than or equal to green
        System.out.println("\nColors from Green onwards:");
        System.out.println(colors.tailSet("Green")); // Returns elements greater than or equal to "Green"



    }
}
```
