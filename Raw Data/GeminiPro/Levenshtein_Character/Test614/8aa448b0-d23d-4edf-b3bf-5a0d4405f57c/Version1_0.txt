```java
import java.util.TreeSet;

public class GreenSet {

    public static void main(String[] args) {
        // Create a TreeSet. TreeSets are sorted naturally.
        TreeSet<String> greenSet = new TreeSet<>();

        // Add some colors (strings)
        greenSet.add("Green"); // Of course!
        greenSet.add("Lime");
        greenSet.add("Emerald");
        greenSet.add("Olive");
        greenSet.add("Forest Green");
        greenSet.add("Seafoam");
        greenSet.add("Chartreuse");


        // Print out the TreeSet.  The elements will be printed in their natural order (alphabetical).
        System.out.println(greenSet); 
    }
}
```
