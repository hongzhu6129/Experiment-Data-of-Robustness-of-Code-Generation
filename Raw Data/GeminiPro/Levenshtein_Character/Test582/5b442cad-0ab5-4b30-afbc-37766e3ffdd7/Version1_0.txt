```java
import java.util.LinkedList;

public class InsertAtFrontLinkedList {

    public static void main(String[] args) {
        // Create a LinkedList
        LinkedList<String> linkedList = new LinkedList<>();

        // Add some initial elements (optional)
        linkedList.add("Apple");
        linkedList.add("Banana");
        linkedList.add("Cherry");

        // Element to insert at the front
        String elementToInsert = "Orange";

        // Insert the element at the front
        linkedList.addFirst(elementToInsert); // Or offerFirst(elementToInsert);
        // Both addFirst and offerFirst do the same thing for LinkedLists.


        // Print the modified LinkedList
        System.out.println("Modified LinkedList: " + linkedList); 
    }
}
```
