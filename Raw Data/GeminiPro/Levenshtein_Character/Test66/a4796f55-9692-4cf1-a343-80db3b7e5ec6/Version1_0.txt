```java
public class ArrayCombination {

    public static void main(String[] args) {
        int[] array1 = {1, 2, 3};
        int[] array2 = {4, 5, 6};

        int[] newArray = new int[2];

        newArray[0] = array1[0]; // First element of array1
        newArray[1] = array2[0]; // First element of array2


        System.out.print("New Array: [");
        for (int i = 0; i < newArray.length; i++) {
            System.out.print(newArray[i]);
            if (i < newArray.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");


        //Alternative (if you meant last elements, not "ago"):
        newArray[0] = array1[0]; // First element of array1
        newArray[1] = array2[array2.length -1]; // Last element of array2

        System.out.print("New Array (with last element of second array): [");
        for (int i = 0; i < newArray.length; i++) {
            System.out.print(newArray[i]);
            if (i < newArray.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");

    }
}
```
