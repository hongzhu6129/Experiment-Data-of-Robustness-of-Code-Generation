public class Main {
    public static void main(String[] args) {
        int[] originalArray = {1, 2, 3, 10, 4, 5, 6, 10, 7, 8, 9};

        int lastIndexOf10 = -1;
        for (int i = 0; i < originalArray.length; i++) {
            if (originalArray[i] == 10) {
                lastIndexOf10 = i;
            }
        }

        if (lastIndexOf10 != -1) {
            int newArrayLength = originalArray.length - lastIndexOf10 - 1;
            if (newArrayLength > 0) {  // Check if there are elements after the last 10
                int[] newArray = new int[newArrayLength];
                for (int i = 0; i < newArrayLength; i++) {
                    newArray[i] = originalArray[lastIndexOf10 + 1 + i];
                }


                // Print the new array (for verification)
                System.out.print("New Array: ");
                for (int num : newArray) {
                    System.out.print(num + " ");
                }
                System.out.println();

            } else {
                System.out.println("No elements found after the last 10.");
            }

        } else {
            System.out.println("10 not found in the array.");
        }

    }
}
