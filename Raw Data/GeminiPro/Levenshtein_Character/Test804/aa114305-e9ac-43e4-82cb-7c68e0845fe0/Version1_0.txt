public class AmericanFlag {

    public static void main(String[] args) {
        for (int i = 0; i < 6; i++) {
            // Print stars section
            for (int j = 0; j < 6; j++) {
                if (i % 2 == 0) {
                    System.out.print("* ");
                } else {
                    System.out.print(" *"); 
                }
            }
            //Print Stripes section
            System.out.print(" ");
            for (int j = 0; j < 20; j++) {
                System.out.print("=");
            }
            System.out.println(); // New line
        }
        //Print the bottom red stripes
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j < 30; j++) {
                System.out.print("=");
            }
            System.out.println();
        }

    }
}
