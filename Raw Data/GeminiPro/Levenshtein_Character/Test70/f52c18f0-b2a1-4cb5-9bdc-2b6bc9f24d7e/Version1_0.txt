public class LargestElement {

    public static void main(String[] args) {
        int[] arr = {10, 20, 30, 40, 50, 60}; // Example even length array

        if (arr.length == 0 || arr.length % 2 != 0) {
            System.out.println("Array must be even length and non-empty.");
            return;
        }

        int first = arr[0];
        int middle = arr[arr.length / 2 - 1]; // Middle element for even length
        int last = arr[arr.length - 1];

        int largest = first;

        if (middle > largest) {
            largest = middle;
        }

        if (last > largest) {
            largest = last;
        }

        System.out.println("Largest element among first, middle, and last: " + largest);


        // Example with different array
         int[] arr2 = {5, 12, 25, 8, 15, 3}; 

         if (arr2.length == 0 || arr2.length % 2 != 0) {
            System.out.println("Array must be even length and non-empty.");
            return;
        }
        first = arr2[0];
        middle = arr2[arr2.length / 2 - 1]; 
        last = arr2[arr2.length - 1];

        largest = first;

        if (middle > largest) {
            largest = middle;
        }

        if (last > largest) {
            largest = last;
        }
         System.out.println("Largest element among first, middle, and last (second example): " + largest);
    }
}
