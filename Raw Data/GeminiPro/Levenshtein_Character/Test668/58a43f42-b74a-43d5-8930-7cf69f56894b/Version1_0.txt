import java.util.TreeMap;

public class LeastKeyGreaterThan {

    public static void main(String[] args) {
        // Example usage:
        TreeMap<Integer, String> treeMap = new TreeMap<>();
        treeMap.put(1, "One");
        treeMap.put(3, "Three");
        treeMap.put(5, "Five");
        treeMap.put(7, "Seven");
        treeMap.put(9, "Nine");

        Integer key1 = 4;
        Integer key2 = 10;
        Integer key3 = 1;
        Integer key4 = 5;

        System.out.println("Least key greater than " + key1 + ": " + getLeastKeyGreaterThan(treeMap, key1)); // Output: 5
        System.out.println("Least key greater than " + key2 + ": " + getLeastKeyGreaterThan(treeMap, key2)); // Output: null
        System.out.println("Least key greater than " + key3 + ": " + getLeastKeyGreaterThan(treeMap, key3)); // Output: 3
        System.out.println("Least key greater than " + key4 + ": " + getLeastKeyGreaterThan(treeMap, key4)); // Output: 7



    }


    public static <K, V> K getLeastKeyGreaterThan(TreeMap<K, V> treeMap, K key) {
        return treeMap.higherKey(key);
    }
}
