import java.util.TreeSet;

public class TreeSetGreaterOrEqual {

    public static void main(String[] args) {
        // Create a TreeSet
        TreeSet<Integer> treeSet = new TreeSet<>();

        // Add elements to the TreeSet (order is maintained automatically)
        treeSet.add(10);
        treeSet.add(20);
        treeSet.add(30);
        treeSet.add(40);
        treeSet.add(50);

        // Element to search for
        int searchElement = 25;

        // Get the element greater than or equal to the search element
        Integer greaterOrEqualElement = treeSet.ceiling(searchElement);

        // Print the result
        if (greaterOrEqualElement != null) {
            System.out.println("Element greater than or equal to " + searchElement + ": " + greaterOrEqualElement);
        } else {
            System.out.println("No element found greater than or equal to " + searchElement);
        }


        //Example 2: Search element exists in the set.
        searchElement = 30;
        greaterOrEqualElement = treeSet.ceiling(searchElement);
        if (greaterOrEqualElement != null) {
            System.out.println("Element greater than or equal to " + searchElement + ": " + greaterOrEqualElement);
        } else {
            System.out.println("No element found greater than or equal to " + searchElement);
        }


        //Example 3: Search element is smaller than the smallest element in the set.
        searchElement = 5;
        greaterOrEqualElement = treeSet.ceiling(searchElement);
        if (greaterOrEqualElement != null) {
            System.out.println("Element greater than or equal to " + searchElement + ": " + greaterOrEqualElement);
        } else {
            System.out.println("No element found greater than or equal to " + searchElement);
        }

        //Example 4: Search element is larger than the largest element in the set.
        searchElement = 60;
        greaterOrEqualElement = treeSet.ceiling(searchElement);
        if (greaterOrEqualElement != null) {
            System.out.println("Element greater than or equal to " + searchElement + ": " + greaterOrEqualElement);
        } else {
            System.out.println("No element found greater than or equal to " + searchElement);
        }
    }
}
