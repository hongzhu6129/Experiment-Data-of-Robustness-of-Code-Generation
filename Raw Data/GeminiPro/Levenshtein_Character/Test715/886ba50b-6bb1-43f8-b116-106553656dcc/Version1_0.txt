class Solution {
    /**
     * Sorts the given integer array using the Insertion Sort algorithm.
     *
     * @param arr The integer array to be sorted.
     */
    void InsertionSort(int[] arr) {
        int n = arr.length;
        for (int i = 1; i < n; ++i) {
            int key = arr[i];
            int j = i - 1;

            /* Move elements of arr[0..i-1], that are
               greater than key, to one position ahead
               of their current position */
            while (j >= 0 && arr[j] > key) {
                arr[j + 1] = arr[j];
                j = j - 1;
            }
            arr[j + 1] = key;
        }
    }


    public static void main(String[] args) {
        Solution solution = new Solution();
        int[] arr1 = {12, 11, 13, 5, 6};
        solution.InsertionSort(arr1);
        System.out.print("Sorted array: ");
        for (int num : arr1) {
            System.out.print(num + " ");
        }
        System.out.println();


        int[] arr2 = {5, 2, 4, 6, 1, 3};  // Example usage
        solution.InsertionSort(arr2);
        System.out.print("Sorted array: ");
        for (int num : arr2) {
            System.out.print(num + " ");
        }
        System.out.println();

         int[] arr3 = {};  // Empty array test
        solution.InsertionSort(arr3);
        System.out.print("Sorted array: ");
        for (int num : arr3) {
            System.out.print(num + " ");
        }
        System.out.println();


         int[] arr4 = {5};  // Single element array test
        solution.InsertionSort(arr4);
        System.out.print("Sorted array: ");
        for (int num : arr4) {
            System.out.print(num + " ");
        }
        System.out.println();
    }
}
