```java
import java.util.HashSet;
import java.util.Set;

public class CompareGreenSets {

    public static void main(String[] args) {
        // Create two sets of "green" strings (or any other type)
        Set<String> set1 = new HashSet<>();
        set1.add("Green Apple");
        set1.add("Green Grass");
        set1.add("Green Light");

        Set<String> set2 = new HashSet<>();
        set2.add("Green Light");
        set2.add("Green Tea");
        set2.add("Green Apple");


        // 1. Check if sets are equal (contain the same elements, regardless of order)
        boolean areEqual = set1.equals(set2);
        System.out.println("Sets are equal: " + areEqual);

        // 2. Check if one set is a subset of another
        boolean set1IsSubsetOfSet2 = set1.containsAll(set2);
        boolean set2IsSubsetOfSet1 = set2.containsAll(set1);

        System.out.println("Set 1 is a subset of Set 2: " + set1IsSubsetOfSet2);
        System.out.println("Set 2 is a subset of Set 1: " + set2IsSubsetOfSet1);


        // 3. Find the common elements (intersection)
        Set<String> intersection = new HashSet<>(set1); // Create a copy to avoid modifying the original
        intersection.retainAll(set2); 

        System.out.println("Common elements (intersection): " + intersection);


        // 4. Find the elements that are in either set but not both (symmetric difference)
        Set<String> symmetricDifference = new HashSet<>(set1);
        symmetricDifference.addAll(set2); // Union
        symmetricDifference.removeAll(intersection); // Remove the common elements

        System.out.println("Elements in either set but not both (symmetric difference): " + symmetricDifference);



        // 5. Find elements unique to set1
        Set<String> uniqueToSet1 = new HashSet<>(set1);
        uniqueToSet1.removeAll(set2);

        System.out.println("Elements unique to Set 1: " + uniqueToSet1);

        // 6. Find elements unique to set2
        Set<String> uniqueToSet2 = new HashSet<>(set2);
        uniqueToSet2.removeAll(set1);

        System.out.println("Elements unique to Set 2: " + uniqueToSet2);

    }
}
```
