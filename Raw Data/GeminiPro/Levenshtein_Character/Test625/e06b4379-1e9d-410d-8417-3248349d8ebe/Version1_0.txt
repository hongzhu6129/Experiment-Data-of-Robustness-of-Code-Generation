import java.util.TreeSet;

public class GreaterOrEqualInTreeSet {

    public static void main(String[] args) {
        // Example usage:
        TreeSet<Integer> greenSet = new TreeSet<>();
        greenSet.add(10);
        greenSet.add(20);
        greenSet.add(30);
        greenSet.add(40);
        greenSet.add(50);

        int givenElement1 = 25;
        Integer result1 = greenSet.ceiling(givenElement1);  // Get element >= givenElement1
        System.out.println("Element greater than or equal to " + givenElement1 + ": " + result1);

        int givenElement2 = 30;
        Integer result2 = greenSet.ceiling(givenElement2); // Get element >= givenElement2
        System.out.println("Element greater than or equal to " + givenElement2 + ": " + result2);

        int givenElement3 = 60;
        Integer result3 = greenSet.ceiling(givenElement3);  // Get element >= givenElement3 (returns null if none)
        System.out.println("Element greater than or equal to " + givenElement3 + ": " + result3);


        // Example with other data types (e.g., String):
        TreeSet<String> stringSet = new TreeSet<>();
        stringSet.add("apple");
        stringSet.add("banana");
        stringSet.add("orange");

        String givenString = "apricot";
        String stringResult = stringSet.ceiling(givenString);
        System.out.println("String greater than or equal to " + givenString + ": " + stringResult);


    }
}
