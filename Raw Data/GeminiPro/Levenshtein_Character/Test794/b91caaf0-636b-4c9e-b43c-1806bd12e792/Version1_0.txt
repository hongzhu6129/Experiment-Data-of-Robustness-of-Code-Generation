```java
public class Operations {

    public static void main(String[] args) {
        // a. -7 + 8 * 6
        int resultA = -7 + 8 * 6;
        System.out.println(resultA); // Output: 41

        // b. (55+9) % 9
        int resultB = (55 + 9) % 9;
        System.out.println(resultB); // Output: 1

        // c. 20 + -3*5 / 8
        int resultC = 20 + -3 * 5 / 8;
        System.out.println(resultC); // Output: 19  (integer division)

        // d. 5 + 15 / 3 * 2 - 8 % 3
        int resultD = 5 + 15 / 3 * 2 - 8 % 3;
        System.out.println(resultD); // Output: 13
    }
}
```
