    private static int[] Saddleback(int[][] arr, int target, int m, int n) {
        int row = 0;
        int col = n - 1;

        while (row < m && col >= 0) {
            if (arr[row][col] == target) {
                return new int[]{row, col};  // Found the target
            } else if (arr[row][col] > target) {
                col--;  // Move left
            } else {
                row++;  // Move down
            }
        }

        return new int[]{-1, -1}; // Target not found
    }
}
