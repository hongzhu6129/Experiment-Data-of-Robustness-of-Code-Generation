```java
public class StringFromCharArray {

    public static void main(String[] args) {

        // Method 1: Using String constructor directly
        char[] charArray = {'H', 'e', 'l', 'l', 'o', ' ', 'W', 'o', 'r', 'l', 'd'};
        String str1 = new String(charArray);
        System.out.println("String created using constructor: " + str1);


        // Method 2: Using String.valueOf()
        char[] charArray2 = {'J', 'a', 'v', 'a'};
        String str2 = String.valueOf(charArray2);
        System.out.println("String created using valueOf(): " + str2);



        // Example with a subset of the character array
        char[] charArray3 = {'A', 'B', 'C', 'D', 'E', 'F'};
        String str3 = new String(charArray3, 1, 3); // Start at index 1, take 3 characters
        System.out.println("String created using subset of array: " + str3); // Output: BCD
    }
}
```
