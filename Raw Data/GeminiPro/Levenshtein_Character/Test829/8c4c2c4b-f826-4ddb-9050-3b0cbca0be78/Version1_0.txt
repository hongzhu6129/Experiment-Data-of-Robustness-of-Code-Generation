```java
public class Pattern {

    public static void main(String[] args) {
        System.out.println("j        a        v     v        a");
        System.out.println("j       a a        v   v        a a");
        System.out.println("j      a   a        v v        a   a");
        System.out.println("j     a     a        v        a     a");
        System.out.println("jjjjjj aaaaa        v        aaaaa");
        System.out.println("j     a     a        v        a     a");
        System.out.println("j      a   a        v v        a   a");
        System.out.println("j       a a        v   v        a a");
        System.out.println("j        a        v     v        a");


        // Alternative using a loop (more efficient for complex patterns):

        // Define the pattern rows as strings in an array
        String[] pattern = {
                "j        a        v     v        a",
                "j       a a        v   v        a a",
                "j      a   a        v v        a   a",
                "j     a     a        v        a     a",
                "jjjjjj aaaaa        v        aaaaa",
                "j     a     a        v        a     a",
                "j      a   a        v v        a   a",
                "j       a a        v   v        a a",
                "j        a        v     v        a"
        };


        // Loop through the array and print each row
        for (String row : pattern) {
            System.out.println(row);
        }

    }
}
```
