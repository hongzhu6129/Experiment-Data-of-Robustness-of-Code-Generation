import java.util.Arrays;

class Solution {
    /**
     * Removes duplicates from a sorted integer array in-place.
     *
     * @param nums The sorted integer array.
     * @return The new length of the array after removing duplicates.
     */
    static int removeDuplicates(int[] nums) {
        if (nums == null || nums.length == 0) {
            return 0;
        }

        int uniqueIndex = 0; 
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[uniqueIndex]) {
                uniqueIndex++;
                nums[uniqueIndex] = nums[i];
            }
        }

        return uniqueIndex + 1;
    }


    public static void main(String[] args) {
        int[] nums1 = {1, 1, 2};
        int len1 = removeDuplicates(nums1);
        System.out.println("Array after removing duplicates: " + Arrays.toString(Arrays.copyOfRange(nums1, 0, len1)) + ", Length: " + len1); // Expected: [1, 2], 2


        int[] nums2 = {0, 0, 1, 1, 1, 2, 2, 3, 3, 4};
        int len2 = removeDuplicates(nums2);
        System.out.println("Array after removing duplicates: " + Arrays.toString(Arrays.copyOfRange(nums2, 0, len2)) + ", Length: " + len2); // Expected: [0, 1, 2, 3, 4], 5

        int[] nums3 = {}; // Empty array
        int len3 = removeDuplicates(nums3);
        System.out.println("Array after removing duplicates: " + Arrays.toString(Arrays.copyOfRange(nums3, 0, len3)) + ", Length: " + len3); // Expected: [], 0

        int[] nums4 = null; // Null array
        int len4 = removeDuplicates(nums4);
        System.out.println("Array after removing duplicates: " + (nums4 == null ? "null" : Arrays.toString(Arrays.copyOfRange(nums4, 0,len4))) + ", Length: " + len4); // Expected: [], 0



    }
}
