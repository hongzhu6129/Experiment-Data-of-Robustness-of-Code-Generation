public class RemoveVowels {

    public static void main(String[] args) {
        String str = "Hello World";
        String updatedString = validate(str);
        System.out.println(updatedString); // Output: Hll Wrld

        str = "AEIOUaeiou";
        updatedString = validate(str);
        System.out.println(updatedString); // Output:

        updatedString = validate(str);
        System.out.println(updatedString); // Output: Ths s  tst strng


        str = ""; // Empty string
        updatedString = validate(str);
        System.out.println(updatedString); // Output: 


        str = null; // Null string - handle this case to prevent NullPointerException
        updatedString = validate(str);
        System.out.println(updatedString); // Output: null (or handle as needed)


    }

    public static String validate(String str) {
        if (str == null) {
            return null; // Or return an empty string or throw an exception as needed
        }
        return str.replaceAll("[aeiouAEIOU]", "");
    }
}
