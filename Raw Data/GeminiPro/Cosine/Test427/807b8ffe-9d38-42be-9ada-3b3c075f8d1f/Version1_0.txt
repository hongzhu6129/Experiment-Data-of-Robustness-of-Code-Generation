```java
import java.util.HashMap;
import java.util.Map;

public class CountCharactersMoreThanTwice {

    public static void main(String[] args) {
        String str1 = "abcabaabbccd";
        System.out.println(test(str1)); // Output: 3 (a, b, c)

        String str2 = "1234567890";
        System.out.println(test(str2)); // Output: 0

        String str3 = "aabbccddeeffgghhiijjkkllmmnnooppqqrrssttuuvvwwxxyyzz";
        System.out.println(test(str3)); // Output: 26 (all characters)
        
        String str4 = "aabbccddee";
        System.out.println(test(str4)); // Output: 5

        String str5 = "";
        System.out.println(test(str5)); // Output: 0
    }

    public static int test(String str) {
        if (str == null || str.isEmpty()) {
            return 0;
        }

        Map<Character, Integer> charCounts = new HashMap<>();
        int count = 0;

        for (char c : str.toCharArray()) {
            if (Character.isLetterOrDigit(c)) { // Check if alphanumeric
                charCounts.put(c, charCounts.getOrDefault(c, 0) + 1);
            }
        }

        for (int charCount : charCounts.values()) {
            if (charCount > 2) {
                count++;
            }
        }

        return count;
    }
}
```
