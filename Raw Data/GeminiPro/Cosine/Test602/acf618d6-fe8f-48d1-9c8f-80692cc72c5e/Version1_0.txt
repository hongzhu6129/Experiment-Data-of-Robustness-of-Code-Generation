```java
import java.util.HashMap;

public class AppendToHashMap {

    public static void main(String[] args) {
        // Create a sample HashMap
        HashMap<String, Integer> hashMap = new HashMap<>();
        hashMap.put("apple", 1);
        hashMap.put("banana", 2);
        hashMap.put("orange", 3);

        // Element to append
        String newKey = "grape";
        int newValue = 4;

        // Append the element
        hashMap.put(newKey, newValue); // HashMaps don't have a specific "append" operation; put() effectively appends if the key doesn't exist.


        // Print the updated HashMap
        System.out.println("Updated HashMap: " + hashMap);


        // Another example demonstrating handling potential duplicate keys:
        HashMap<String, String> hashMap2 = new HashMap<>();
        hashMap2.put("color", "red");


        String newKey2 = "color"; // A duplicate key
        String newValue2 = "blue";

        // Appending (or rather, updating since the key already exists)
        hashMap2.put(newKey2, newValue2); // The value associated with "color" will be updated to "blue".
        System.out.println("Updated HashMap2: " + hashMap2); // Output will show "color" mapped to "blue"


       // A more explicit "append" logic if you only want to add if the key *doesn't* yet exist:
        HashMap<String, String> hashMap3 = new HashMap<>();
        hashMap3.put("city", "London");
        String newKey3 = "country";
        String newValue3 = "UK";


        if (!hashMap3.containsKey(newKey3)) { // Check if the key is already present
            hashMap3.put(newKey3, newValue3); // Only add if it's a new key
            System.out.println("Updated HashMap3: " + hashMap3);
        } else {
            System.out.println("Key 'country' already exists. Not appending.");
        }
    }
}
```
