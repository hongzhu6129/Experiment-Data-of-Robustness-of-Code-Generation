```java
import java.util.ArrayList;
import java.util.LinkedList;

public class LinkedListToArray {

    public static void main(String[] args) {

        // Create a LinkedList
        LinkedList<String> linkedList = new LinkedList<>();
        linkedList.add("Apple");
        linkedList.add("Banana");
        linkedList.add("Orange");
        linkedList.add("Grape");

        // Convert LinkedList to ArrayList
        ArrayList<String> arrayList = new ArrayList<>(linkedList);

        // Print the ArrayList
        System.out.println("ArrayList: " + arrayList);


        //Example with Integer types
        LinkedList<Integer> linkedListNumbers = new LinkedList<>();
        linkedListNumbers.add(1);
        linkedListNumbers.add(2);
        linkedListNumbers.add(3);

        ArrayList<Integer> arrayListNumbers = new ArrayList<>(linkedListNumbers);
        System.out.println("ArrayList of numbers: " + arrayListNumbers);

    }
}
```
