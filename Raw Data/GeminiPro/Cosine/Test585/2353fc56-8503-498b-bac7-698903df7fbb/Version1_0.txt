import java.util.LinkedList;
import java.util.List;

public class LinkedListOccurrences {

    public static void main(String[] args) {
        // Create a sample LinkedList
        List<Integer> numbers = new LinkedList<>();
        numbers.add(1);
        numbers.add(2);
        numbers.add(3);
        numbers.add(2);
        numbers.add(4);
        numbers.add(2);
        numbers.add(5);

        // Element to search for
        int target = 2;

        // Find the first and last occurrences
        int firstOccurrence = -1;
        int lastOccurrence = -1;


        // Method 1: Using indexOf and lastIndexOf (most efficient for LinkedLists)

        firstOccurrence = numbers.indexOf(target);
        lastOccurrence = numbers.lastIndexOf(target);




        // Method 2: Manual iteration (demonstrative, less efficient)
        // Useful if you need to perform other actions during the traversal or if you
        // are working with a custom linked list implementation that does not have 
        // indexOf/lastIndexOf methods

        if (firstOccurrence == -1 && lastOccurrence == -1) { //Only execute if indexOf/lastIndexOf wasn't used.
            for (int i = 0; i < numbers.size(); i++) {
                if (numbers.get(i) == target) {
                    if (firstOccurrence == -1) {
                        firstOccurrence = i;
                    }
                    lastOccurrence = i;
                }
            }
        }




        // Output the results
        if (firstOccurrence != -1) {
            System.out.println("First occurrence of " + target + " is at index: " + firstOccurrence);
            System.out.println("Last occurrence of " + target + " is at index: " + lastOccurrence);
        } else {
            System.out.println(target + " is not found in the LinkedList.");
        }



        // Example demonstrating handling null values in the list
         List<Integer> numbersWithNull = new LinkedList<>();
        numbersWithNull.add(1);
        numbersWithNull.add(null); // Add a null element
        numbersWithNull.add(2);
        numbersWithNull.add(null); // Add another null element
        numbersWithNull.add(3);

        int target2 = null; // search for null 
        firstOccurrence = numbersWithNull.indexOf(target2); //indexOf handles nulls correctly
        lastOccurrence = numbersWithNull.lastIndexOf(target2);

        if (firstOccurrence != -1) {
            System.out.println("First occurrence of null is at index: " + firstOccurrence);
            System.out.println("Last occurrence of null is at index: " + lastOccurrence);
        } else {
            System.out.println("null is not found in the LinkedList.");
        }


    }
}
