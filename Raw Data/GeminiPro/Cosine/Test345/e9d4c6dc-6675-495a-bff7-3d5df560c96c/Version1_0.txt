public class StringToArray {

    public static void main(String[] args) {

        String str = "sample"; // The string you want to convert

        char[] charArray = str.toCharArray(); // Convert the string to a character array

        // Print the character array (optional, for verification)
        System.out.print("Character array: [");
        for (int i = 0; i < charArray.length; i++) {
            System.out.print(charArray[i]);
            if (i < charArray.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");


        // Alternatively, create a character array directly with a string literal:

        char[] charArray2 = {'a', 'n', 'o', 't', 'h', 'e', 'r', ' ', 's', 't', 'r', 'i', 'n', 'g'};

         // Print the second character array (optional)

            System.out.print(charArray2[i]);
            if (i < charArray2.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }
}
