```java
import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class StringSort {

    public static void main(String[] args) {
        List<String> strings = Arrays.asList("apple", "banana", "orange", "grape", "kiwi");

        // Ascending sort
        List<String> ascendingSorted = strings.stream()
                .sorted() // Natural ordering (alphabetical for Strings)
                .collect(Collectors.toList());

        System.out.println("Ascending order: " + ascendingSorted);


        // Descending sort
        List<String> descendingSorted = strings.stream()
                .sorted((s1, s2) -> s2.compareTo(s1)) // Comparator for reverse order 
                .collect(Collectors.toList());

        System.out.println("Descending order: " + descendingSorted);


        // Descending sort using Comparator.reverseOrder() - More concise
         List<String> descendingSorted2 = strings.stream()
                .sorted(java.util.Comparator.reverseOrder()) //  Built-in reverse comparator
                .collect(Collectors.toList());

        System.out.println("Descending order (using Comparator.reverseOrder()): " + descendingSorted2);

    }
}
```
