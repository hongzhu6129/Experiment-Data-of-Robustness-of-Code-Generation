```java
public class ColorArray {

    public static void main(String[] args) {
        String[] names = new String[5]; // Create an array of strings with a size of 5

        names[0] = "Red";
        names[1] = "Green";
        names[2] = "Blue";
        names[3] = "Yellow";
        names[4] = "Purple";

        // Print out the colors
        for (int i = 0; i < names.length; i++) {
            System.out.println(names[i]);
        }


        // Alternatively, using an enhanced for loop (for-each loop):
        System.out.println("\nUsing enhanced for loop:");
        for (String color : names) {
            System.out.println(color);
        }

         // Another way to initialize and print in a single loop:
        System.out.println("\nInitializing and printing in a single loop:");
        String[] colors = {"Red", "Green", "Blue", "Yellow", "Orange"}; // Initialize directly

        for(String color : colors) {
            System.out.println(color);
        }


    }
}
```
