        int[] array3 = {1, 2, 3, 4, 6};

        // Test cases
        System.out.println("Array 1 and Array 2 are equal at start and end: " + compareFirstLast(array1, array2)); // True
        System.out.println("Array 1 and Array 3 are equal at start and end: " + compareFirstLast(array1, array3)); // False
        System.out.println("Array 1 and Array 4 are equal at start and end: " + compareFirstLast(array1, array4)); // False
        System.out.println("Array 4 and Array 5 are equal at start and end: " + compareFirstLast(array4, array5)); // False
        System.out.println("Array 5 and Array 5 are equal at start and end: " + compareFirstLast(array5, array5)); // True


    }

    // Helper function to check if first and last elements are the same in two arrays.
    // Handles empty arrays gracefully.
    public static boolean compareFirstLast(int[] arr1, int[] arr2) {
        if (arr1.length == 0 || arr2.length == 0) { // Check for empty arrays first
            return false;  // If either array is empty, they can't match
        }

        return arr1[0] == arr2[0] && arr1[arr1.length - 1] == arr2[arr2.length - 1];
    }
}
