public class Collatz {

    public static void main(String[] args) {
        int n = 10; // Example starting value. Change as needed.

        System.out.print(n);

        while (n != 1) {
            if (n % 2 == 0) {
                n /= 2;
            } else {
                n = (n * 3) + 1;
            }
            System.out.print(" -> " + n);
        }
    }
}
