```java
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class BesideArrayList {

    public static void main(String[] args) {
        // Example 1:  Beside two array lists of Integers
        List<Integer> list1 = Arrays.asList(1, 2, 3);
        List<Integer> list2 = Arrays.asList(4, 5, 6);

        List<Integer> besideList1 = beside(list1, list2);
        System.out.println("Beside of two Integer lists: " + besideList1); // Output: [1, 4, 2, 5, 3, 6]


        // Example 2: Beside two array lists of Strings
        List<String> stringList1 = Arrays.asList("a", "b", "c");
        List<String> stringList2 = Arrays.asList("d", "e", "f");
        List<String> besideList2 = beside(stringList1, stringList2);
        System.out.println("Beside of two String lists: " + besideList2); // Output: [a, d, b, e, c, f]

        // Example 3: Handling lists of different lengths
        List<Integer> list3 = Arrays.asList(1, 2, 3, 4);  // Longer list
        List<Integer> list4 = Arrays.asList(5, 6);         // Shorter list
        List<Integer> besideList3 = beside(list3, list4);
        System.out.println("Beside with unequal lengths : " + besideList3); // Output: [1, 5, 2, 6, 3, 4]


        // Example 4: One list is empty
        List<Double> list5 = Arrays.asList(1.1, 2.2);
        List<Double> emptyList = new ArrayList<>(); // Empty list
        List<Double> besideList4 = beside(list5, emptyList);
        System.out.println("Beside with an empty list: " + besideList4);  // Output: [1.1, 2.2]

    }


    public static <T> List<T> beside(List<T> list1, List<T> list2) {
        List<T> result = new ArrayList<>();
        int i = 0, j = 0;

        while (i < list1.size() || j < list2.size()) {
            if (i < list1.size()) {
                result.add(list1.get(i++));
            }
            if (j < list2.size()) {
                result.add(list2.get(j++));
            }
        }
        return result;

    }
}
```
