```java
import java.util.PriorityQueue;

public class ComparePriorityQueues {

    public static void main(String[] args) {
        // Create two priority queues
        PriorityQueue<Integer> pq1 = new PriorityQueue<>();
        PriorityQueue<Integer> pq2 = new PriorityQueue<>();

        // Add elements to the first priority queue
        pq1.add(3);
        pq1.add(1);
        pq1.add(4);
        pq1.add(2);

        // Add elements to the second priority queue
        pq2.add(1);
        pq2.add(3);
        pq2.add(2);
        pq2.add(4);


        // Compare the two priority queues
        if (pq1.equals(pq2)) {
            System.out.println("The two priority queues are equal.");
        } else {
            System.out.println("The two priority queues are not equal.");
        }



        // Example demonstrating that order doesn't matter for equality:
        PriorityQueue<Integer> pq3 = new PriorityQueue<>();
        pq3.add(1);
        pq3.add(4);
        pq3.add(2);
        pq3.add(3);


         if (pq1.equals(pq3)) { // pq1 and pq3 have the same elements
            System.out.println("pq1 and pq3 are equal.");
        } else {
            System.out.println("pq1 and pq3 are not equal.");
        }


        //Example with different sizes
        PriorityQueue<Integer> pq4 = new PriorityQueue<>();
        pq4.add(1);
        pq4.add(2);
        pq4.add(3);


         if (pq1.equals(pq4)) { 
            System.out.println("pq1 and pq4 are equal.");
        } else {
            System.out.println("pq1 and pq4 are not equal."); // This will be printed because sizes are different.
        }




        // Example with different elements
        PriorityQueue<Integer> pq5 = new PriorityQueue<>();
        pq5.add(1);
        pq5.add(2);
        pq5.add(5); // Different element
        pq5.add(3);

        if (pq1.equals(pq5)) {
            System.out.println("pq1 and pq5 are equal.");
        } else {
            System.out.println("pq1 and pq5 are not equal."); //This will be printed
        }


    }
}
```
