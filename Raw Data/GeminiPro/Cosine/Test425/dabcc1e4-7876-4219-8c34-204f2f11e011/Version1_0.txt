```java
public class ConsecutiveLetters {

    public static void main(String[] args) {
        String str1 = "PHP";
        System.out.println("String: " + str1 + ", Result: " + test(str1)); // true

        String str2 = "Java";
        System.out.println("String: " + str2 + ", Result: " + test(str2)); // false

        String str3 = "aabbccddee";
        System.out.println("String: " + str3 + ", Result: " + test(str3)); // true

        String str4 = ""; // Empty String
        System.out.println("String: " + str4 + ", Result: " + test(str4)); // false

        String str5 = "a"; // Single character string
        System.out.println("String: " + str5 + ", Result: " + test(str5)); // false


    }

    public static boolean test(String str) {
        if (str.length() < 2) {
            return false;
        }

        for (int i = 0; i < str.length() - 1; i++) {
            if (str.charAt(i) == str.charAt(i + 1)) {
                return true;
            }
        }

        return false;
    }
}
```
