```java
import java.util.Calendar;
import java.util.GregorianCalendar;

public class MaxDateValues {

    public static void main(String[] args) {
        Calendar calendar = new GregorianCalendar(); // Creates a default calendar (current date/time)


        // Get the maximum values
        int maxYear = calendar.getActualMaximum(Calendar.YEAR); //  Effectively the largest year supported
        int maxMonth = calendar.getActualMaximum(Calendar.MONTH); //  11 (December, 0-indexed)
        int maxWeekOfYear = calendar.getActualMaximum(Calendar.WEEK_OF_YEAR); // 52 or 53 depending on the year
        int maxDayOfMonth = calendar.getActualMaximum(Calendar.DAY_OF_MONTH); // 28, 29, 30, or 31


        System.out.println("Maximum Year: " + maxYear);
        System.out.println("Maximum Month (0-indexed): " + maxMonth);
        System.out.println("Maximum Week of Year: " + maxWeekOfYear);
        System.out.println("Maximum Day of Month: " + maxDayOfMonth);


    }
}
```
