public class PentagonalNumbers {

    public static void main(String[] args) {
        // Display the first 50 pentagonal numbers
        final int NUMBER_OF_PENTAGONAL_NUMBERS = 50;
        int count = 0;  // Counter for the number of pentagonal numbers displayed

        for (int n = 1; count < NUMBER_OF_PENTAGONAL_NUMBERS; n++) {
            int pentagonalNumber = getPentagonalNumber(n);
            System.out.printf("%7d ", pentagonalNumber); // Format output for better readability
            count++;

            if (count % 10 == 0) { // Start a new line after every 10 numbers
                System.out.println();
            }

        }

    }



    public static int getPentagonalNumber(int n) {
        return n * (3 * n - 1) / 2;
    }
}
