```java
public class StringComparison {

    public static void main(String[] args) {
        String str1 = "hello";
        String str2 = "world";

        // Compare lexicographically using compareTo()
        int result = str1.compareTo(str2);

        if (result == 0) {
            System.out.println("The strings are lexicographically equal.");
        } else if (result < 0) {
            System.out.println("String 1 is lexicographically less than String 2.");
        } else {
            System.out.println("String 1 is lexicographically greater than String 2.");
        }


        //Example demonstrating lexicographically equal strings
        String str3 = "apple";
        String str4 = "apple";
        
        result = str3.compareTo(str4);

        if (result == 0) {
            System.out.println("The strings are lexicographically equal.");
        } else if (result < 0) {
            System.out.println("String 3 is lexicographically less than String 4.");
        } else {
            System.out.println("String 3 is lexicographically greater than String 4.");
        }

         // Another example
        String str5 = "Apple"; // Note the capital 'A'
        String str6 = "apple";

        result = str5.compareTo(str6); // Capital letters come before lowercase

         if (result == 0) {
            System.out.println("The strings are lexicographically equal.");
        } else if (result < 0) {
            System.out.println("String 5 is lexicographically less than String 6.");
        } else {
            System.out.println("String 5 is lexicographically greater than String 6.");
        }


    }
}
```
