import java.util.Arrays;
import java.util.PriorityQueue;

public class KLargestElements {

    public static void main(String[] args) {
        int[] arr = {1, 23, 12, 9, 30, 2, 50};
        int k = 3;
        findKLargest(arr, k);

        // Another Example
        int[] arr2 = {11, 5, 22, 1, 7, 15};
        int k2 = 2;
        findKLargest(arr2, k2);
    }

    public static void findKLargest(int[] arr, int k) {
        if (k <= 0 || k > arr.length) {
            System.out.println("Invalid value of k.");
            return;
        }


        // Method 1: Using a Min-Heap (Priority Queue) - More efficient for larger datasets
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();

        for (int num : arr) {
            minHeap.offer(num);
            if (minHeap.size() > k) {
                minHeap.poll(); // Remove the smallest element if the heap size exceeds k
            }
        }

        Integer[] kLargestArray = minHeap.toArray(new Integer[0]);  // Convert to array for easy printing

        System.out.println("The " + k + " largest elements (using Min-Heap): " + Arrays.toString(kLargestArray));



        // Method 2: Sorting and then taking the last k elements  - Simpler but less efficient for very large datasets where only k largest are needed.
        Arrays.sort(arr); // Sort the array in ascending order

        int[] kLargestSorted = Arrays.copyOfRange(arr, arr.length - k, arr.length); // Extract the last k elements

        System.out.println("The " + k + " largest elements (using Sorting): " + Arrays.toString(kLargestSorted));

    }
}
