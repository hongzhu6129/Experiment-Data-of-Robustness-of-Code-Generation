```java
public class Main {
    public static void main(String[] args) {
        // Deal two variables (demonstration with integers)
        int a = 10;
        int b = 20;

        System.out.println("Before swapping: a = " + a + ", b = " + b);

        // Swap the values using a temporary variable
        int temp = a;
        a = b;
        b = temp;

        System.out.println("After swapping: a = " + a + ", b = " + b);


        // Print a face
        System.out.println("  +\"\"\"\"+  ");
        System.out.println(" [| o o |]");
        System.out.println("  |  ^  |");
        System.out.println("  | '-' |");
        System.out.println("  +-----+"); 
    }
}
```
