public class AmericanFlag {

    public static void main(String[] args) {
        for (int i = 0; i < 5; i++) {
            // Stars and Stripes rows (alternating)
            if (i % 2 == 0) { // Even rows - more stars
                System.out.print("* * * * * * ");
            } else { // Odd rows - fewer stars
                System.out.print("* * * * * ");
            }
            System.out.println("==================================");

        }
        for (int i = 0; i < 4; i++) {
            // Stars and Stripes rows (alternating) - continued pattern
            if ((i + 5) % 2 == 0) { // Using (i+5) to maintain the alternation from above 
                System.out.print("* * * * * * ");
            } else { 
                System.out.print("* * * * * ");
            }
            System.out.println("==================================");
        }

        // Stripes only
        for (int i = 0; i < 6; i++) {
            System.out.println("==================================");
        }

    }
}
