import java.util.TreeMap;

public class GreatestKeyStrictlyLessThan {

    public static void main(String[] args) {
        // Example usage:
        TreeMap<Integer, String> treeMap = new TreeMap<>();
        treeMap.put(10, "Ten");
        treeMap.put(20, "Twenty");
        treeMap.put(30, "Thirty");
        treeMap.put(40, "Forty");
        treeMap.put(50, "Fifty");


        Integer key1 = 35;
        Integer result1 = getGreatestKeyStrictlyLessThan(treeMap, key1);
        System.out.println("Greatest key strictly less than " + key1 + ": " + result1); // Output: 30

        Integer key2 = 15;
        Integer result2 = getGreatestKeyStrictlyLessThan(treeMap, key2);
        System.out.println("Greatest key strictly less than " + key2 + ": " + result2); // Output: 10


        Integer key3 = 5;
        Integer result3 = getGreatestKeyStrictlyLessThan(treeMap, key3);
        System.out.println("Greatest key strictly less than " + key3 + ": " + result3); // Output: null


        Integer key4 = 60;
        Integer result4 = getGreatestKeyStrictlyLessThan(treeMap, key4);
        System.out.println("Greatest key strictly less than " + key4 + ": " + result4); // Output: 50
    }

    public static <K, V> K getGreatestKeyStrictlyLessThan(TreeMap<K, V> treeMap, K key) {
        return treeMap.lowerKey(key); 
    }
}
