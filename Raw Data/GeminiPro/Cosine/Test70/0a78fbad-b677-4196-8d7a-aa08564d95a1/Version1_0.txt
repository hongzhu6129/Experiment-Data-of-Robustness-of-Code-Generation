public class LargestOfThree {

    public static void main(String[] args) {
        // Example array (even length)
        int[] arr = {10, 5, 25, 15, 8, 20};

        if (arr == null || arr.length == 0 || arr.length % 2 != 0) {
            System.out.println("Invalid input array.");
            return;
        }


        int first = arr[0];
        int middleIndex = arr.length / 2 -1; // Adjust for 0-based indexing
        int middle = arr[middleIndex]; 
        int last = arr[arr.length - 1];

        int largest = first;

        if (middle > largest) {
            largest = middle;
        }

        if (last > largest) {
            largest = last;
        }

        System.out.println("Largest of first, middle, and last: " + largest);
