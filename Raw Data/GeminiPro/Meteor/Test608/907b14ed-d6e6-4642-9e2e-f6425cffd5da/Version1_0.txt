```java
import java.util.Arrays;
import java.util.HashSet;
import java.util.Set;

public class HashSetToArray {

    public static void main(String[] args) {

        // 1. Create a HashSet
        Set<String> hashSet = new HashSet<>();
        hashSet.add("apple");
        hashSet.add("banana");
        hashSet.add("orange");
        hashSet.add("grape");

        System.out.println("HashSet: " + hashSet);


        // 2. Convert HashSet to an array using toArray() with type parameter
        String[] stringArray = hashSet.toArray(new String[0]); 

        System.out.println("String Array (using toArray() with type parameter): " + Arrays.toString(stringArray));

        // Explanation:
        // This is the preferred and type-safe way.  `new String[0]` creates an empty array of the correct type.
        // toArray() will create a new array of the correct size if the provided array is too small.
        // If the provided array is larger, toArray() will use it and set the remaining elements to null. This can be useful for pre-allocating a large array to prevent reallocations if you know the maximum possible size of the HashSet.


        // 3. Convert HashSet to an array using toArray() without type parameter (less common now)
        Object[] objectArray = hashSet.toArray();

        System.out.println("Object Array (using toArray() without type parameter): " + Arrays.toString(objectArray));
        // Explanation: This returns an Object array. You will generally need to cast the elements back to their original type.
        // This is less type-safe than the first method and requires an extra step.



        // 4. Convert HashSet to an array using a loop (more involved, less efficient)
        String[] loopArray = new String[hashSet.size()];
        int i = 0;
        for (String fruit : hashSet) {
            loopArray[i++] = fruit;
        }

        System.out.println("String Array (using a loop): " + Arrays.toString(loopArray));
        // Explanation: This is more verbose than using toArray() and less efficient. It's not typically recommended unless you need very fine-grained control over the process.


        // Important Considerations:

        // Order:  Remember that HashSets are unordered. The order of elements in the array might not match the order they were added to the HashSet.
        // Nulls: HashSets can contain null values. These will be included in the resulting array.

    }
}
```
