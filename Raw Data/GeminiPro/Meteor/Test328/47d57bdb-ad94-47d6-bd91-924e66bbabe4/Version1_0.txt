```java
public class StringFromCharArray {

    public static void main(String[] args) {

        // Method 1: Using String constructor directly
        char[] charArray1 = {'H', 'e', 'l', 'l', 'o', ' ', 'W', 'o', 'r', 'l', 'd'};
        String str1 = new String(charArray1);
        System.out.println("String created using constructor: " + str1);

        // Method 2: Using String.valueOf()
        char[] charArray2 = {'J', 'a', 'v', 'a'};
        String str2 = String.valueOf(charArray2);
        System.out.println("String created using valueOf(): " + str2);

        // Method 3: Creating a string from a subset of the character array
        char[] charArray3 = {'P', 'r', 'o', 'g', 'r', 'a', 'm', 'm', 'i', 'n', 'g'};
        String str3 = new String(charArray3, 0, 8); // Creates string from index 0 to 7 (exclusive of 8)
        System.out.println("String created from subset of array: " + str3);


    }
}
```
