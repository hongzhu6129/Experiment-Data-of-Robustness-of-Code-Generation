import java.util.TreeSet;

public class AddTreeSetElements {

    public static void main(String[] args) {
        // Create a TreeSet called "greenSet" (assuming it contains some elements)
        TreeSet<String> greenSet = new TreeSet<>();
        greenSet.add("Lime");
        greenSet.add("Emerald");
        greenSet.add("Olive");
        greenSet.add("Jade");

        // Create another TreeSet called "treeSet" (this will be the destination set)
        TreeSet<String> treeSet = new TreeSet<>();
        treeSet.add("Red");
        treeSet.add("Blue");


        // Add all elements of greenSet to treeSet
        treeSet.addAll(greenSet); 

        // Print the updated treeSet
        System.out.println("TreeSet after adding elements from greenSet: " + treeSet);
    }
}
