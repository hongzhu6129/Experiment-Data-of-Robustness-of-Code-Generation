public class AmericanFlag {

    public static void main(String[] args) {
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j < 6; j++) {
                System.out.print("* ");
            }
            System.out.print(" "); // Space between stars and stripes
            for (int j = 0; j < 26; j++) {
                System.out.print("=");
            }
            System.out.println(); // Newline after each row
        }
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j < 32; j++) {
                System.out.print("=");
            }
            System.out.println(); // Newline after each stripe row
        }
    }
}
