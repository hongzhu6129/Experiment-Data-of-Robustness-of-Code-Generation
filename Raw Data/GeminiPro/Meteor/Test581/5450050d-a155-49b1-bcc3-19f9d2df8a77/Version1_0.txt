```java
import java.util.LinkedList;

public class LinkedListEdit {

    public static void main(String[] args) {
        // Create a LinkedList of Strings
        LinkedList<String> linkedList = new LinkedList<>();

        // Add some initial elements
        linkedList.add("Apple");
        linkedList.add("Banana");
        linkedList.add("Orange");

        System.out.println("Original LinkedList: " + linkedList);

        // Add an element at the beginning (first position)
        linkedList.addFirst("Grape");

        // Add an element at the end (last position)
        linkedList.addLast("Mango");


        System.out.println("Modified LinkedList: " + linkedList);


         //Example using other methods for first/last manipulation
         linkedList.offerFirst("Kiwi"); //same as addFirst
         linkedList.offerLast("Pineapple"); //same as addLast

         //push also adds to the beginning (like addFirst)
         linkedList.push("Watermelon");

        System.out.println("Further Modified LinkedList: " + linkedList);


        // You can also replace elements at the first and last positions
         linkedList.set(0, "Strawberry"); //Replace at index 0 (first)
         linkedList.set(linkedList.size()-1, "Pear"); //replace at last index



        System.out.println("Final LinkedList: " + linkedList);
    }
}
```
