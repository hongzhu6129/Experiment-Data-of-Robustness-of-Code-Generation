public class CheckTenTwenty {

    public static void main(String[] args) {
        int[] arr1 = {1, 10, 20, 30};
        int[] arr2 = {10, 1, 20, 30};
        int[] arr3 = {20, 10, 20, 30};
        int[] arr4 = {1, 20, 10, 30}; // 10 comes after 20
        int[] arr5 = {1, 10, 30, 20}; // 20 comes later
        int[] arr6 = {1, 10}; // No 20


        System.out.println("Array 1: " + checkTenThenTwenty(arr1)); // True
        System.out.println("Array 2: " + checkTenThenTwenty(arr2)); // True
        System.out.println("Array 3: " + checkTenThenTwenty(arr3)); // False (10 after 20)
        System.out.println("Array 4: " + checkTenThenTwenty(arr4)); // False (10 after 20)
        System.out.println("Array 5: " + checkTenThenTwenty(arr5)); // True
        System.out.println("Array 6: " + checkTenThenTwenty(arr6)); // False (No 20)

    }

    public static boolean checkTenThenTwenty(int[] arr) {
        boolean foundTen = false;

        for (int num : arr) {
            if (num == 10) {
                foundTen = true;
            } else if (foundTen && num == 20) {
                return true;
            } else if(num==20 && !foundTen){ // 20 before 10
                return false;
            }

        }

        return false; // Didn't find both 10 then 20
    }
}
