        System.out.println("Array 3: " + check10Then20(arr3)); // Expected: true

        System.out.println("Array 4: " + check10Then20(arr4)); // Expected: true


        int[] arr5 = {1, 20, 30, 40};
        System.out.println("Array 5: " + check10Then20(arr5)); // Expected: false

        int[] arr6 = {10, 1, 2, 3};
        System.out.println("Array 6: " + check10Then20(arr6)); // Expected: false

        int[] arr7 = {20, 20, 20, 20};
        System.out.println("Array 7: " + check10Then20(arr7)); // Expected: false

        int[] arr8 = {10, 10, 10, 10};
        System.out.println("Array 8: " + check10Then20(arr8)); // Expected: false

        int[] arr9 = {};
        System.out.println("Array 9: " + check10Then20(arr9)); // Expected: false
       
    }


    public static boolean check10Then20(int[] arr) {
        boolean found10 = false;

        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == 10) {
                found10 = true;
            }
             if (found10) { // Only start checking for 20 *after* a 10 has been found
                for (int j = i + 1; j < arr.length; j++) {  // Start checking from the next position after finding a 10.
                    if (arr[j] == 20) {
                        return true; 
                    }
                }
            }
        }
        return false;
    }
}

Key improvements in this version:

* **Clearer Variable Name:** `found10` is more descriptive than just `found`.
* **Efficient Inner Loop:**  The inner loop now starts from `i + 1` (the element *after* the current 10).  This avoids unnecessary checks and ensures the 20 comes *after* the 10.
* **Return True Immediately:** Once a 20 is found after a 10, the function immediately returns `true`. This is more efficient than continuing to check the rest of the array.
* **Test Cases:**  A thorough set of test cases demonstrates different scenarios and ensures the function behaves correctly.  This helps catch edge cases and improves confidence in the code's reliability.
