```java
public class PrintBooleanArray {

    public static void main(String[] args) {
        boolean[][] boolArray = {
                {true, false, true},
                {false, true, false},
                {true, true, false}
        };

        // Iterate through the 2D array and print 't' for true and 'f' for false
        for (int i = 0; i < boolArray.length; i++) {
            for (int j = 0; j < boolArray[i].length; j++) {
                System.out.print(boolArray[i][j] ? 't' : 'f');
                if (j < boolArray[i].length - 1) {
                    System.out.print(" "); // Add space between elements in a row
                }
            }
            System.out.println(); // Newline after each row
        }
    }
}
```
