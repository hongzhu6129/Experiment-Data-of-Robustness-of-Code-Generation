```java
import java.util.ArrayList;
import java.util.List;

public class InsertAtFirst {

    public static void main(String[] args) {
        // Create an ArrayList of strings (you can change the type if needed)
        List<String> days = new ArrayList<>();

        // Add some initial elements (optional)
        days.add("Monday");
        days.add("Tuesday");
        days.add("Wednesday");

        // Element to insert at the beginning
        String newDay = "Sunday";

        // Insert the new element at the first position (index 0)
        days.add(0, newDay);


        // Print the updated ArrayList
        System.out.println(days); 
    }
}
```
