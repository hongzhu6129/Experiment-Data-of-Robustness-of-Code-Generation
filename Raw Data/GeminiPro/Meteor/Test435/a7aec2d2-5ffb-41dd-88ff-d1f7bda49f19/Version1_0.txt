```java
import java.util.Arrays;
import java.util.Comparator;
import java.util.List;
import java.util.stream.Collectors;

public class StringSort {

    public static void main(String[] args) {
        List<String> strings = Arrays.asList("apple", "banana", "orange", "grape", "kiwi");

        // Ascending order
        List<String> ascendingSorted = strings.stream()
                .sorted() // Uses natural ordering (alphabetical for Strings)
                .collect(Collectors.toList());

        System.out.println("Ascending order: " + ascendingSorted);


        // Descending order
        List<String> descendingSorted = strings.stream()
                .sorted(Comparator.reverseOrder()) // Use reverseOrder for descending
                .collect(Collectors.toList());

        System.out.println("Descending order: " + descendingSorted);


         //More concise descending (using method reference)
        List<String> descendingSorted2 = strings.stream()
                .sorted(String::compareToIgnoreCase).reversed() //Sort ascending, then reverse
                .collect(Collectors.toList());

        System.out.println("Descending order (method reference + reversed): " + descendingSorted2);



        // Case-insensitive ascending sort (if needed)
        List<String> caseInsensitiveAscending = strings.stream()
                .sorted(String.CASE_INSENSITIVE_ORDER)
                .collect(Collectors.toList());

        System.out.println("Case-insensitive ascending order: " + caseInsensitiveAscending);


        // Case-insensitive descending sort
         List<String> caseInsensitiveDescending = strings.stream()
                .sorted(String.CASE_INSENSITIVE_ORDER.reversed())
                .collect(Collectors.toList());

        System.out.println("Case-insensitive descending order: " + caseInsensitiveDescending);
    }
}
```
