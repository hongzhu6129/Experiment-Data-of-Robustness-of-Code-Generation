```java
public class MiddleCharacters {

    public static void main(String[] args) {
        String str1 = "hello";
        String str2 = "worldd";

        System.out.println("Middle characters of \"" + str1 + "\": " + middle(str1));
        System.out.println("Middle characters of \"" + str2 + "\": " + middle(str2));

        // Additional test cases
        System.out.println("Middle characters of \"" + "a" + "\": " + middle("a"));
        System.out.println("Middle characters of \"" + "ab" + "\": " + middle("ab"));

    }

    public static String middle(String str) {
        int len = str.length();
        int middleIndex = len / 2;

        if (len % 2 == 0) {
            // Even length: return the single middle character
            return String.valueOf(str.charAt(middleIndex -1 )) + String.valueOf(str.charAt(middleIndex));  // corrected
        } else {
            // Odd length: return the two middle characters
            return str.substring(middleIndex -1, middleIndex + 1); // Return as substring for two chars
        }
    }
}
```
