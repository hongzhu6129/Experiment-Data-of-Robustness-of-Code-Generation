import java.util.Arrays;
import java.util.PriorityQueue;

public class KLargestElements {

    public static void main(String[] args) {
        int[] arr = {1, 23, 12, 9, 30, 2, 50};
        int k = 3;
        findKLargest(arr, k);

        arr = new int[]{11, 5, 20, 8, 15, 7};
        k = 2;
        findKLargest(arr, k);

        arr = new int[]{1, 23, 12, 9, 30, 2, 50};
        k = 7;
        findKLargest(arr,k);


        arr = new int[]{1, 23, 12, 9, 30, 2, 50};
        k = 1;
        findKLargest(arr, k);


        arr = new int[]{5,5,5,5,5};
        k=3;
        findKLargest(arr,k);


    }

    public static void findKLargest(int[] arr, int k) {
        if (k <= 0 || k > arr.length) {
            System.out.println("Invalid value of k.");
            return;
        }

        // Method 1: Using a Min-Heap (Priority Queue) - Efficient for large arrays
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        for (int num : arr) {
            minHeap.offer(num);
            if (minHeap.size() > k) {
                minHeap.poll(); // Remove the smallest element if heap size exceeds k
            }
        }

        // Print the k largest elements from the min-heap
        Integer[] kLargestArray = minHeap.toArray(new Integer[0]); //Convert to array for easier printing

        System.out.print("K Largest elements using Min-Heap: [");
        for(int i = 0; i < kLargestArray.length; i++){
            System.out.print(kLargestArray[i]);
            if (i < kLargestArray.length -1){
                System.out.print(", ");
            }
        }
        System.out.println("]");



        // Method 2: Sorting - Simpler for smaller arrays
        Arrays.sort(arr); // Sort in ascending order

        System.out.print("K Largest elements using Sorting: [");
        for (int i = arr.length - k; i < arr.length; i++) {
            System.out.print(arr[i]);
            if (i < arr.length - 1){
                System.out.print(", ");
            }
        }
        System.out.println("]");


        System.out.println(); // Add a newline for better output formatting


    }
}
