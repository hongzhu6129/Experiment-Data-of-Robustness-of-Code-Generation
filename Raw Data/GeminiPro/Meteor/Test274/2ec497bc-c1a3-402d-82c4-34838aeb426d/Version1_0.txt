import java.util.Arrays;

class Solution {

    /**
     * Finds the number of unique paths from top-left to bottom-right in a grid with obstacles.
     *
     * @param obstacleGrid The grid representing obstacles (1) and empty spaces (0).
     * @return The number of unique paths.
     */
    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length;
        int n = obstacleGrid[0].length;

        // If the starting cell has an obstacle, there are no paths.
        if (obstacleGrid[0][0] == 1) {
            return 0;
        }

        // Create a DP table to store the number of paths to each cell.
        int[][] dp = new int[m][n];

        // Initialize the starting cell to 1 (one path to reach the starting cell itself).
        dp[0][0] = 1;


        // Initialize the first row. Cells in the first row can only be reached from the left.
        for (int j = 1; j < n; j++) {
            if(obstacleGrid[0][j] == 0) {
                dp[0][j] = dp[0][j-1]; //Inherit path count from left if no obstacle
            } else {
                dp[0][j] = 0; //No path if obstacle present
            }
        }


        // Initialize the first column. Cells in the first column can only be reached from above.
        for (int i = 1; i < m; i++) {
            if(obstacleGrid[i][0] == 0) {
                 dp[i][0] = dp[i-1][0]; //Inherit path count from above if no obstacle
            }else{
                dp[i][0] = 0; //No path if obstacle present
            }
        }


        // Iterate through the rest of the grid, calculating paths based on the cells above and to the left.
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                if (obstacleGrid[i][j] == 0) { //If no obstacle
                    dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
                } else { //If obstacle is present
                    dp[i][j] = 0; //No path can reach this cell
                }
            }
        }

        // The bottom-right cell of the DP table contains the total number of unique paths.
        return dp[m - 1][n - 1];
    }



    public static void main(String[] args) {
        Solution sol = new Solution();
        int[][] obstacleGrid = {{0, 0, 0}, {0, 1, 0}, {0, 0, 0}};
        int paths = sol.uniquePathsWithObstacles(obstacleGrid);
        System.out.println("Number of unique paths: " + paths); // Output: 2

        obstacleGrid = new int[][] {{0,1},{0,0}};
        paths = sol.uniquePathsWithObstacles(obstacleGrid);
        System.out.println("Number of unique paths: " + paths); // Output: 1
    }
}
