import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class CircleCalculations {

    public static double calculatePerimeter(double radius) {
        return 2 * Math.PI * radius;
    }

    public static double calculateArea(double radius) {
        return Math.PI * radius * radius;
    }

    public static void main(String[] args) {
        double radius = 7.5;
        double perimeter = calculatePerimeter(radius);
        double area = calculateArea(radius);

        System.out.println("Perimeter is = " + perimeter);
        System.out.println("Area is = " + area);
    }
}


class CircleCalculationsTest {

    @Test
    void calculatePerimeter_positiveRadius() {
        assertEquals(47.12388980384689, CircleCalculations.calculatePerimeter(7.5), 0.000001);  // Delta for double comparison
    }

    @Test
    void calculateArea_positiveRadius() {
        assertEquals(176.71458676442586, CircleCalculations.calculateArea(7.5), 0.000001);
    }



    @Test
    void calculatePerimeter_zeroRadius() {

    }

* **Double Comparisons:**  When comparing floating-point numbers (like `double`), you should never use `assertEquals` directly.  Floating-point calculations have inherent imprecision.  The `assertEquals` method for doubles takes a third argument ï¿½ a `delta`. This `delta` specifies the tolerance within which the two doubles are considered equal.

* **Exception Handling (Recommended):** The best practice is to add exception handling directly in the `calculateArea` and `calculatePerimeter` methods. This prevents unexpected behavior in your main application if someone tries to use a negative radius:

    public static double calculatePerimeter(double radius) {
        if (radius < 0) {
            throw new IllegalArgumentException("Radius cannot be negative.");
        }
        return 2 * Math.PI * radius;
    }

    public static double calculateArea(double radius) {
