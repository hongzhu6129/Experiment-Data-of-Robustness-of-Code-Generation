```java
public class Main {
    public static void main(String[] args) {
        // Swapping two variables
        int a = 10;
        int b = 20;

        System.out.println("Before swapping:");
        System.out.println("a = " + a);
        System.out.println("b = " + b);

        // Swapping using a temporary variable
        int temp = a;
        a = b;
        b = temp;


        System.out.println("\nAfter swapping:");
        System.out.println("a = " + a);
        System.out.println("b = " + b);

        // Printing a face
        System.out.println("\nPrinting a face:");
        System.out.println(" +\"\"\"\"\"+ ");
        System.out.println("[| o o |]");
        System.out.println(" |  ^  | ");
        System.out.println(" | '-' | ");
        System.out.println(" +-----+ ");



       //Swapping without temporary variable (using addition and subtraction)

         int x = 50;
         int y = 100;


        System.out.println("\nBefore swapping (without temp):");
        System.out.println("x = " + x);
        System.out.println("y = " + y);

        x = x + y;  // x now becomes 150
        y = x - y;  // y becomes 50 (150-100)
        x = x - y;  // x becomes 100 (150 - 50)


        System.out.println("\nAfter swapping (without temp):");
        System.out.println("x = " + x);
        System.out.println("y = " + y);


        //Swapping using bitwise XOR (another method without temp)
        int p = 25;
        int q = 75;


        System.out.println("\nBefore swapping (using XOR):");
        System.out.println("p = " + p);
        System.out.println("q = " + q);

        p = p ^ q;
        q = p ^ q;
        p = p ^ q;

        System.out.println("\nAfter swapping (using XOR):");
        System.out.println("p = " + p);
        System.out.println("q = " + q);
    }
}
```
