class Solution {
    /**
     * Checks if three integers are consecutive and the same.
     *
     * @param a The first integer.
     * @param b The second integer.
     * @param c The third integer.
     * @return True if the integers are consecutive and the same, false otherwise.
     */
    public static boolean test(int a, int b, int c) {
        return a == b && b == c;
    }

    public static void main(String[] args) {
        System.out.println(test(1, 1, 1)); // Output: true
        System.out.println(test(1, 2, 3)); // Output: false
        System.out.println(test(2, 2, 1)); // Output: false


    }
}
