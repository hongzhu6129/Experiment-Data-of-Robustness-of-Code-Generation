class Solution {
    /**
     * Checks if two given strings are rotations of each other.
     *
     * @param s1 The first string.
     * @param s2 The second string.
     * @return True if the strings are rotations of each other, false otherwise.
     */
    static boolean checkForRotation(String s1, String s2) {
        if (s1 == null || s2 == null || s1.length() != s2.length()) {
            return false;
        }
        if (s1.length() == 0) { // Handle empty strings
            return true;
        }

        String concatenated = s1 + s1;
        return concatenated.contains(s2);
    }

    public static void main(String[] args) {
        String str1 = "waterbottle";
        String str2 = "erbottlewat";
        System.out.println(str1 + ", " + str2 + ": " + checkForRotation(str1, str2));  // Output: true

        str1 = "abcde";
        str2 = "eabcd";
        System.out.println(str1 + ", " + str2 + ": " + checkForRotation(str1, str2));  // Output: true

        str1 = "abcde";
        str2 = "abced";
        System.out.println(str1 + ", " + str2 + ": " + checkForRotation(str1, str2));  // Output: false


        str1 = "";
        str2 = "";
        System.out.println(str1 + ", " + str2 + ": " + checkForRotation(str1, str2));  // Output: true

        str1 = "abcd";
        str2 = null;
        System.out.println(str1 + ", " + str2 + ": " + checkForRotation(str1, str2));  // Output: false

        str1 = null;
        str2 = "abcd";
        System.out.println(str1 + ", " + str2 + ": " + checkForRotation(str1, str2));  // Output: false



    }
}
