public class AmericanFlag {

    public static void main(String[] args) {
        // Print the star section
        for (int i = 0; i < 5; ++i) {
            // Print the stars
            for (int j = 0; j < 6; ++j) {
                System.out.print("* ");
            }
            System.out.print(" "); // Space after the stars

            // Print the stripes
            System.out.println("==================================");


            // Print the stars for the alternating rows
            if (i % 2 == 1) {
                 for (int j = 0; j < 5; ++j) {
                     System.out.print("* ");
                 }
                  System.out.print(" "); // Space after the stars

                  // Print the stripe
                  System.out.println("==================================");

            }
             
        }

        // Print the remaining stripes
        for (int i = 0; i < 6; ++i) {
            System.out.println("==================================");
        }
    }
}
