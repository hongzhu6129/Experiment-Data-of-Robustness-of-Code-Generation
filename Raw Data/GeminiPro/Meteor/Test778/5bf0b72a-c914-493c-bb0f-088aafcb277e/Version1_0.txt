public class AmericanFlag {

    public static void main(String[] args) {
        // Print the striped section
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j < 6; j++) {
                System.out.print("* ");
            }
            System.out.print(" ==================================");
            System.out.println();

             if (i < 5){ // Don't print the extra line after the last stripe row
                for (int j = 0; j < 5; j++) {
                    System.out.print("* ");
                 }
                System.out.print(" ==================================");
                System.out.println();
             }
        }

        // Print the solid section
        for (int i = 0; i < 6; i++) {
            System.out.print("==================================");
            System.out.println();
        }
    }
}
