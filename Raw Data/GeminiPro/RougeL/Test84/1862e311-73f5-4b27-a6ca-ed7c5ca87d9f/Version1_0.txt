```java
public class StringArrayGenerator {

    public static void main(String[] args) {
        int n = 10; // Example size, you can change this

        if (n <= 0) {
            System.out.println("Array size must be greater than 0.");
            return;
        }

        String[] stringArray = new String[n];

        for (int i = 0; i < n; i++) {
            stringArray[i] = String.valueOf(i); // Convert int to String
        }

        // Print the generated array (optional)
        System.out.print("Generated String array: [");
        for (int i = 0; i < n; i++) {
            System.out.print(stringArray[i]);
            if (i < n - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }
}
```
