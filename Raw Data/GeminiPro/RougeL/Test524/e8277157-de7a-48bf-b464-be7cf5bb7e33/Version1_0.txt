        // Handle negative numbers
        if (num < 0) {
            return isPerfectCube(-num); // Check the absolute value for negative cubes
        }


        //Efficient approach using cubic root 
        int cubeRoot = (int) Math.round(Math.cbrt(num)); // Calculate the integer cubic root
        return cubeRoot * cubeRoot * cubeRoot == num;
