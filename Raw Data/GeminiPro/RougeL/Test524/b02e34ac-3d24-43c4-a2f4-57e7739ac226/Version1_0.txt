public class CubeChecker {

    public static void main(String[] args) {
        // Input data (hardcoded for this example as per problem description)
        int number = 8;

        if (isCube(number)) {
            System.out.println("Number is a cube.");
        } else {
            System.out.println("Number is not a cube.");
        }


        // Example with a different number (not a cube)
        number = 10;
        if (isCube(number)) {
            System.out.println("Number is a cube.");
        } else {
            System.out.println("Number is not a cube.");
        }

         // Example with a negative cube
        number = -27;
        if (isCube(number)) {
            System.out.println("Number is a cube.");
        } else {
            System.out.println("Number is not a cube.");
        }

         // Example with zero
        number = 0;
        if (isCube(number)) {
            System.out.println("Number is a cube.");
        } else {
            System.out.println("Number is not a cube.");
        }
    }



    // Helper function to check if a number is a perfect cube
    private static boolean isCube(int n) {
        // Handle negative numbers (the cube root of a negative number is negative)
        if (n < 0) {
            return isCube(-n); // Recursively check the absolute value
        }


        // Binary search for the cube root. This is more efficient than looping from 1 to n.
        long low = 0;
        long high = n;  // Use long to avoid overflow for larger cubes
        while (low <= high) {
            long mid = low + (high - low) / 2; // Safe calculation to prevent overflow
            long cube = mid * mid * mid;
            if (cube == n) {
                return true;
            } else if (cube < n) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return false;
    }
}
