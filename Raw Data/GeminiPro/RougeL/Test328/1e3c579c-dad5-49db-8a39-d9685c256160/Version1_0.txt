public class StringFromArray {

    public static void main(String[] args) {
        // Character array
        char[] charArray = {'J', 'a', 'v', 'a', ' ', 'S', 't', 'r', 'i', 'n', 'g'};

        // Create a String object from the character array
        String str = new String(charArray);

        System.out.println(str); 
    }
}
