import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class CountWordsEndingMOrRap {

    public int endWithmOrn(String text) {
        if (text == null || text.trim().isEmpty()) {
            return 0;
        }

        // Use a regular expression to find words ending in 'm' or 'rap' (case-insensitive)
        Pattern pattern = Pattern.compile("\\b\\w*(?i)(m|rap)\\b"); // \b for word boundaries
        Matcher matcher = pattern.matcher(text);

        int count = 0;
        while (matcher.find()) {
            count++;
        }
        return count;
    }

    public static void main(String[] args) {
        CountWordsEndingMOrRap counter = new CountWordsEndingMOrRap();

        String text1 = "ham program exam Ram program scrap trap wrap";
        int count1 = counter.endWithmOrn(text1);

        int count2 = counter.endWithmOrn(text2);

- **Regular Expression:** Uses `\b\w*(?i)(m|rap)\b` for accurate word matching:

    - `\w*`: Matches zero or more word characters (letters, numbers, underscore).
    - `(?i)`: Case-insensitive flag.
    - `(m|rap)`: Matches either "m" or "rap".
- **Handles Null/Empty:** Includes checks for `null` or empty input strings to prevent `NullPointerException`.
- **Trimmed Input:** Uses `text.trim()` to handle leading/trailing whitespace, avoiding incorrect counts for spaces.
