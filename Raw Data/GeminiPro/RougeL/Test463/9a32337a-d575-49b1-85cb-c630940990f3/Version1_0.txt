```java
import java.time.LocalDate;
import java.time.Month;
import java.time.YearMonth;
import java.time.format.TextStyle;
import java.util.Locale;

public class GetMonthInfo {

    public static void main(String[] args) {
        // Get current month information
        LocalDate currentDate = LocalDate.now();
        Month currentMonth = currentDate.getMonth();
        int currentYear = currentDate.getYear();

        displayMonthInfo(currentYear, currentMonth);


        // Example for a specific month (e.g., February 2024)
        YearMonth specificMonth = YearMonth.of(2024, Month.FEBRUARY); // or YearMonth.of(2024, 2)
        displayMonthInfo(specificMonth.getYear(), specificMonth.getMonth());

    }

    private static void displayMonthInfo(int year, Month month) {
        System.out.println("Month: " + month.getDisplayName(TextStyle.FULL, Locale.getDefault()));  //e.g.,  "January" or "Dezember" (Locale dependent)
        System.out.println("Month Number: " + month.getValue()); // 1 for January, 2 for Feb, ...
        System.out.println("Number of Days: " + YearMonth.of(year, month).lengthOfMonth()); // Days in the month
        System.out.println("--------------------");

    }
}
```
