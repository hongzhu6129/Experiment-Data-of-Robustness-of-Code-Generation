```java
public class Staircase {

    public static void main(String[] args) {
        int n = 10; // Example number of coins

        int row = 1;
        int coinsUsed = 0;

        while (coinsUsed + row <= n) {
            for (int i = 0; i < row; i++) {
                System.out.print("* ");
            }
            System.out.println();

            coinsUsed += row;
            row++;
        }

        // Handle remaining coins (if any) on the last row
        if (coinsUsed < n) {
            for (int i = 0; i < n - coinsUsed; i++) {
                System.out.print("* ");
            }
            System.out.println();
        }
    }
}
```
