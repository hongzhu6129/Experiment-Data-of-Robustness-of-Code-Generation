class Solution {
    /**
     * Checks if the string "good" appears at index 0 or 1.
     *
     * @param str The input string.
     * @return True if "good" appears at index 0 or 1, false otherwise.
     */
    public boolean hasGood(String str) {
        if (str == null || str.length() < 4) {
            return false;
        }

        if (str.startsWith("good")) {
            return true;
        }

        if (str.length() >= 5 && str.substring(1).startsWith("good")) {
            return true;
        }

        return false;
    }

    /**
     * Main method for testing.
     *
     * @param args Command-line arguments.
     */
    public static void main(String[] args) {
        Solution solution = new Solution();

        // Test cases
        System.out.println(solution.hasGood("goodpi")); // true
        System.out.println(solution.hasGood("xgood"));  // true
        System.out.println(solution.hasGood("bad"));   // false
        System.out.println(solution.hasGood("good"));  // true
        System.out.println(solution.hasGood("ggood")); // false 
        System.out.println(solution.hasGood(null));   // false
        System.out.println(solution.hasGood("goo"));   // false
        System.out.println(solution.hasGood("g"));     // false


    }
}
