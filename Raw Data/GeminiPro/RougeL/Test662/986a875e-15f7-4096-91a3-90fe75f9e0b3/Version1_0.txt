```java
import java.util.Map;
import java.util.TreeMap;

public class Main {
    public static void main(String[] args) {
        // Create a TreeMap to maintain sorted order of keys
        Map<Integer, String> map = new TreeMap<>();

        // Add some entries to the map
        map.put(3, "Three");
        map.put(1, "One");
        map.put(4, "Four");
        map.put(2, "Two");
        map.put(5, "Five");

        if (!map.isEmpty()) { // Check if the map is empty before getting keys
            // Get the first (lowest) key using firstKey()
            Integer firstKey = map.keySet().stream().findFirst().get(); // Or map.firstKey() for TreeMap

            // Get the last (highest) key using lastKey() for TreeMap, or using stream for general Maps.
            Integer lastKey = map.keySet().stream().reduce((first, second) -> second).get(); // Or map.lastKey() for TreeMap
            
             System.out.println("First (Lowest) key: " + firstKey);
            System.out.println("Last (Highest) key: " + lastKey);
        } else {
            System.out.println("The map is empty.");
        }
    }
}
```
