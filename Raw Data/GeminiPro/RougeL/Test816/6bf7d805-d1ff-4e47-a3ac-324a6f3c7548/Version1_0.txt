```java
import java.util.Scanner;

public class OctalToBinary {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Input any octal number: ");
        String octalString = scanner.next();

        try {
            int octal = Integer.parseInt(octalString, 8); // Parse octal string directly
            String binary = Integer.toBinaryString(octal);

            System.out.println("Equivalent binary number: " + binary);
        } catch (NumberFormatException e) {
            System.out.println("Invalid octal number input.");
        }
        scanner.close(); 
    }
}
```
