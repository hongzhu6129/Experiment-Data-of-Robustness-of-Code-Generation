public class ConsecutiveNumbers {

    public static void main(String[] args) {
        System.out.println(test(1, 2, 3)); // Output: true
        System.out.println(test(3, 2, 1)); // Output: true
        System.out.println(test(1, 3, 2)); // Output: true
        System.out.println(test(1, 1, 1)); // Output: false  (same, not consecutive)
        System.out.println(test(1, 2, 4)); // Output: false
        System.out.println(test(4, 2, 1)); // Output: true
    }

    public static boolean test(int a, int b, int c) {
        int[] nums = {a, b, c};
        // Sort the numbers
        java.util.Arrays.sort(nums);

        // Check if they are consecutive and not the same
        return (nums[1] == nums[0] + 1 && nums[2] == nums[1] + 1) && !(a == b && b == c);


        // Alternative (without sorting, slightly more complex logic):
        // return ( (a==b+1 && b==c+1) || (a==b-1 && b ==c-1) || (a==b-1 && b==c+1) || (a==b+1 && b==c-1) || (a==c-1 && b==c-1) || (a==c+1 && b==c+1) )   && !(a==b && b==c);
    }
}
