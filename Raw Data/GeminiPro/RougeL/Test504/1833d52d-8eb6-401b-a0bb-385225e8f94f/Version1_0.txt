class Solution {
    /**
     * Given an integer, display the factors of 3 within the integer.
     */
    public static void main(String[] args) {
        test(27); // Output: 3 9 27
        test(100); // Output: 3 9 27 81
        test(5); // Output: 3
        test(1); // Output: 
        test(0); // Output:
        test(-27); // Output: -3 -9 -27
    }

    public static void test(int n) {
        int factor = 3;
        if (n < 0) {
            while (n <= -factor) {
                System.out.print(-factor + " ");
                factor *= 3;
            }
        } else {
            while (n >= factor) {
                System.out.print(factor + " ");
                factor *= 3;
            }
        }
        System.out.println();

    }
}
