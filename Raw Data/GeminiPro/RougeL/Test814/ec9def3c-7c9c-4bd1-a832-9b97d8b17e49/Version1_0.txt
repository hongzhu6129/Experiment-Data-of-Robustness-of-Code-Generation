```java
import java.util.Scanner;

public class BinaryToOctal {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        System.out.print("Input a binary number: ");
        String binaryString = input.nextLine();

        try {
            int decimal = Integer.parseInt(binaryString, 2); // Convert binary string to decimal integer
            String octalString = Integer.toOctalString(decimal); // Convert decimal integer to octal string

            System.out.println("Octal number: " + octalString);

        } catch (NumberFormatException e) {
            System.out.println("Invalid binary input. Please enter a valid binary number.");
        }
    }
}
```
