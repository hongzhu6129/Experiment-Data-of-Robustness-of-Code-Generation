```java
public class Main {
    public static void main(String[] args) {
        // Swapping two variables

        int a = 10;
        int b = 20;

        System.out.println("Before swapping:");
        System.out.println("a = " + a);
        System.out.println("b = " + b);

        // Method 1: Using a temporary variable
        int temp = a;
        a = b;
        b = temp;


        System.out.println("\nAfter swapping (using temp variable):");
        System.out.println("a = " + a);
        System.out.println("b = " + b);


        // Reset values for the next method
        a = 10;
        b = 20;

         // Method 2: Using addition and subtraction (only for numbers, might cause overflow with very large numbers)
        a = a + b;  
        b = a - b;
        a = a - b;


        System.out.println("\nAfter swapping (using addition/subtraction):");
        System.out.println("a = " + a);
        System.out.println("b = " + b);



        // Reset values for the next method
        a = 10;
        b = 20;

        // Method 3: Using bitwise XOR (only for integers)
        a = a ^ b;
        b = a ^ b;
        a = a ^ b;

        System.out.println("\nAfter swapping (using XOR):");
        System.out.println("a = " + a);
        System.out.println("b = " + b);




        // Printing a face
        System.out.println("\nPrinting a face:");

        System.out.println(" +\"\"\"\"\"+ ");
        System.out.println("[| o o |]");
        System.out.println(" |  ^  | ");
        System.out.println(" | '-' | ");
        System.out.println(" +-----+ ");


    }
}
```
