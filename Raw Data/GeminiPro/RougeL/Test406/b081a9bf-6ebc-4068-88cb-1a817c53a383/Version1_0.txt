import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class SumNumbersInString {

    public int sumOfTheNumbers(String str) {
        int sum = 0;
        Pattern p = Pattern.compile("\\d+"); // Regular expression to find digits
        Matcher m = p.matcher(str);
        while (m.find()) {
            sum += Integer.parseInt(m.group());
        }
        return sum;
    }

    public static void main(String[] args) {
        SumNumbersInString obj = new SumNumbersInString();
        String str1 = "abc 123 def 45 ghi 6";
        System.out.println("Sum of numbers in \"" + str1 + "\" is: " + obj.sumOfTheNumbers(str1)); // Output: 174

        String str2 = "hello world";
        System.out.println("Sum of numbers in \"" + str2 + "\" is: " + obj.sumOfTheNumbers(str2)); // Output: 0

        String str3 = "1a2b3c";
        System.out.println("Sum of numbers in \"" + str3 + "\" is: " + obj.sumOfTheNumbers(str3)); // Output: 6


        String str4 = "123.45"; //Handles only integer parts.
        System.out.println("Sum of numbers in \"" + str4 + "\" is: " + obj.sumOfTheNumbers(str4)); // Output: 123


    }
}
