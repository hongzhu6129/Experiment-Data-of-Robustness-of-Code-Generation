```java
public class SumSeries {

    public static void main(String[] args) {
        int sum = 0;
        for (int i = 1; i <= 10; i++) {
            sum += i;
        }
        System.out.println("The sum of the series 1 to 10 is: " + sum);


        // If you specifically need to show the series as 1 + 2 + 3 ... + 10 = sum
        //  rather than just the final result,  use the following:
        System.out.print("1");
        for (int i = 2; i <= 10; i++) {
             System.out.print(" + " + i);
        }
        System.out.println(" = " + sum);

    }
}
```
