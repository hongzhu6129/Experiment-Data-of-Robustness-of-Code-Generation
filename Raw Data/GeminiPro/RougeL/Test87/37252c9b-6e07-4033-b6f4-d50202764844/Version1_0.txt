public class Check20Occurrence {

    public static void main(String[] args) {
        int[] arr1 = {1, 20, 3, 4, 5, 20, 7, 8, 20}; // Valid
        int[] arr2 = {1, 20, 20, 4, 5, 20, 7, 8, 9}; // Invalid (adjacent 20s)
        int[] arr3 = {1, 20, 3, 4, 5, 20, 7, 8, 9, 20, 10}; // Valid
        int[] arr4 = {1, 20, 3, 4, 5, 6, 7, 8, 9}; // Invalid (not enough 20s)


        System.out.println("arr1: " + check20(arr1));
        System.out.println("arr2: " + check20(arr2));
        System.out.println("arr3: " + check20(arr3));
        System.out.println("arr4: " + check20(arr4));
    }


    public static boolean check20(int[] arr) {
        int count = 0;
        boolean adjacent = false;

        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == 20) {
                count++;
                if (i > 0 && arr[i - 1] == 20) {  // Check for adjacent 20s
                    adjacent = true;
                    break; // No need to continue if adjacent 20s are found
                }
            }
        }

        return count == 3 && !adjacent;  // Return true only if exactly three 20s and no adjacent ones
    }
}
