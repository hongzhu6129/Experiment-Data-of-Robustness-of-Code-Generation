```java
import java.util.TreeSet;

public class TreeSetExample {

    public static void main(String[] args) {
        // Create a TreeSet of Strings (colors in this case)
        TreeSet<String> colors = new TreeSet<>();

        // Add some colors
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Orange");

        // Print out the TreeSet. Elements will be sorted naturally (alphabetically)
        System.out.println(colors); 
    }
}
```
