import java.util.TreeSet;

public class TreeSetExample {

    public static void main(String[] args) {

        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Orange");

        System.out.println(colors); 
    }
}
