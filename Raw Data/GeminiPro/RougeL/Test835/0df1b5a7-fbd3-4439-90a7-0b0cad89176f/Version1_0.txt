        for (int i = 0; i < 9; i++) {
            // Stars section
            if (i % 2 == 0) {
                for (int j = 0; j < 6; j++) {
                    System.out.print("* ");
                }
            } else {
                for (int j = 0; j < 5; j++) {
                    System.out.print("* ");
                }
            }

            // Stripes Section

                System.out.print("=");
            }
