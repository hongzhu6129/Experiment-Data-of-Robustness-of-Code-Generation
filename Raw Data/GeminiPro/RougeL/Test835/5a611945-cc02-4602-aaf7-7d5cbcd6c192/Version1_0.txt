    public static void main(String[] args) {
        // Print the striped section
        for (int i = 0; i < 6; ++i) {
            for (int j = 0; j < 6; ++j) {
                System.out.print("* ");
            }
            System.out.print(" =================================\n");
            if (i % 2 != 0) {
                for (int j = 0; j < 5; j++) {
                    System.out.print("* ");

                }
                System.out.print(" =================================\n");
            }


        }

        // Print the solid red section
        for (int i = 0; i < 6; ++i) {
            System.out.print("==================================\n");
        }
    }
}
