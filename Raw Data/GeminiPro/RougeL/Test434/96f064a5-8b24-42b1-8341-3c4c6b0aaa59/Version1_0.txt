```java
import java.util.List;
import java.util.Arrays;

public class StringCounter {

    public static void main(String[] args) {
        List<String> strings = Arrays.asList("apple", "apricot", "banana", "Avocado", "blueberry", "orange", "Apple");

        char specificLetter = 'a'; // Case-sensitive

        long count = strings.stream()
                .filter(s -> !s.isEmpty() && Character.toLowerCase(s.charAt(0)) == Character.toLowerCase(specificLetter)) // Handle empty strings & case
                .count();

        System.out.println("Number of strings starting with '" + specificLetter + "' (case-insensitive): " + count);


        specificLetter = 'A'; // Case-sensitive example
        count = strings.stream()
                .filter(s -> !s.isEmpty() && s.charAt(0) == specificLetter)
                .count();
         System.out.println("Number of strings starting with '" + specificLetter + "' (case-sensitive): " + count);



    }
}
```
