```java
import java.util.Arrays;

class Solution {
    /**
     * Given a sorted array nums, remove the duplicates in-place such that each element appears only once and returns the new length.
     *
     * @param nums sorted array of integers
     * @return the new length of the array after removing duplicates
     */
    static int removeDuplicates(int[] nums) {
        if (nums == null || nums.length == 0) {
            return 0;
        }

        int uniqueIndex = 0; // Index for storing unique elements

        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[uniqueIndex]) {
                uniqueIndex++;
                nums[uniqueIndex] = nums[i];
            }
        }

        return uniqueIndex + 1;
    }

    public static void main(String[] args) {
        int[] nums1 = {1, 1, 2, 3, 3, 3, 4, 5, 6, 7, 7};
        int newLength1 = removeDuplicates(nums1);
        System.out.println("Original array: " + Arrays.toString(nums1));
        System.out.println("The length of the original array is: " + nums1.length);
        System.out.println("After removing duplicates, the new length of the array is: " + newLength1);

        // Add more test cases as needed.
         int[] nums2 = {};
         int newLength2 = removeDuplicates(nums2);
         System.out.println("Original array: " + Arrays.toString(nums2));
        System.out.println("The length of the original array is: " + nums2.length);
        System.out.println("After removing duplicates, the new length of the array is: " + newLength2);
        
        int[] nums3 = null;
         int newLength3 = removeDuplicates(nums3);
         System.out.println("Original array: null"); // Can't use Arrays.toString() with null
        System.out.println("The length of the original array is: 0 (null array)");
        System.out.println("After removing duplicates, the new length of the array is: " + newLength3);


    }
}
```
