```java
import java.util.Arrays;

public class AboveAverage {

    public static void main(String[] args) {
        // Example usage:
        int[] numbers = {1, 5, 2, 8, 3, 9, 4, 7, 6};
        findAboveAverage(numbers);

        int[] numbers2 = {10, 20, 30, 40, 50};
        findAboveAverage(numbers2);


        int [] emptyArray = {};
        findAboveAverage(emptyArray); //Handle empty array case

        int[] singleElementArray = {5};
        findAboveAverage(singleElementArray); // Handle single element case

    }

    public static void findAboveAverage(int[] arr) {
        if (arr == null || arr.length == 0) {
            System.out.println("Array is empty or null. No average can be calculated.");
            return;
        }

        double sum = 0;
        for (int num : arr) {
            sum += num;
        }

        double average = sum / arr.length;

        System.out.print("Numbers above average " + average + ": ");
        for (int num : arr) {
            if (num > average) {
                System.out.print(num + " ");
            }
        }
        System.out.println(); // New line for better output formatting

    }
}
```
