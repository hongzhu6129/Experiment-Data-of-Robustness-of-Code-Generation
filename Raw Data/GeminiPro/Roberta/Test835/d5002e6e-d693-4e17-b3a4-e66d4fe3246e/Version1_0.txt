public class AmericanFlag {

    public static void main(String[] args) {
        // Stripes and Stars Section
        for (int i = 0; i < 9; i++) {
            if (i % 2 == 0) { // Even rows: stars and stripes
                for (int j = 0; j < 6; j++) {
                    System.out.print("* ");
                }
                System.out.print(" "); // Space between stars and stripes
                for (int j = 0; j < 22; j++) {
                    System.out.print("=");
                }

            } else { // Odd rows: Stripes only
                for (int j = 0; j < 4; j++) {
                    System.out.print("* ");
                }
                 System.out.print("  "); // Space between stars and stripes
                 for (int j = 0; j < 22; j++) {
                    System.out.print("=");
                }
            }
            System.out.println(); // New line after each row
        }


        // Bottom Stripes Section
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j < 40; j++) {
                System.out.print("=");
            }
            System.out.println();
        }
    }
}
