```java
import java.util.LinkedList;

public class LinkedListInsertion {

    public static void main(String[] args) {
        // Create a new LinkedList
        LinkedList<String> linkedList = new LinkedList<>();

        // Insert elements at the beginning (first position)
        linkedList.addFirst("Apple");  // LinkedList: [Apple]
        linkedList.addFirst("Banana"); // LinkedList: [Banana, Apple]

        // Insert elements at the end (last position)
        linkedList.addLast("Orange"); // LinkedList: [Banana, Apple, Orange]
        linkedList.addLast("Grape");  // LinkedList: [Banana, Apple, Orange, Grape]

        // Print the LinkedList
        System.out.println(linkedList); // Output: [Banana, Apple, Orange, Grape]

        //Alternative way to add at the end (addLast is preferred for clarity):
        linkedList.add("Mango"); // Adds to the end like addLast
        System.out.println(linkedList); // Output: [Banana, Apple, Orange, Grape, Mango]

    }
}
```
