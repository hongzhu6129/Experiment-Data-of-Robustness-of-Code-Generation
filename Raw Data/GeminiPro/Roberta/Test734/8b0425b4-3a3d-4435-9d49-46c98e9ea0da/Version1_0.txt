public class SaddlebackSearch {

    public static void main(String[] args) {
        int[][] array = {
                {1, 4, 7, 11, 15},
                {2, 5, 8, 12, 19},
                {3, 6, 9, 16, 22},
                {10, 13, 14, 17, 24},
                {18, 21, 23, 26, 30}
        };
        int m = array.length;
        int n = array[0].length;

        int target = 16;
        int[] result = Saddleback(array, m, n, target);

        if (result[0] != -1) {
            System.out.println("Target " + target + " found at row: " + result[0] + ", column: " + result[1]);
        } else {
            System.out.println("Target " + target + " not found.");
        }


        target = 20; // Example with a target not present
        result = Saddleback(array, m, n, target);

        if (result[0] != -1) {
            System.out.println("Target " + target + " found at row: " + result[0] + ", column: " + result[1]);
        } else {
            System.out.println("Target " + target + " not found.");
        }
    }


    private static int[] Saddleback(int[][] array, int m, int n, int target) {
        int row = 0;
        int col = n - 1;
        int[] result = {-1, -1}; // Default: not found

        while (row < m && col >= 0) {
            if (array[row][col] == target) {
                result[0] = row;
                result[1] = col;
                return result;
            } else if (array[row][col] < target) {
                row++; // Move down to the next row (larger values)
            } else {
                col--; // Move left to the previous column (smaller values)
            }
        }

        return result; // Target not found
    }
}

5. **Example with Target Not Found:** Added a second example in `main` to demonstrate the behavior when the target is not present in the array.
6. **Efficient Saddleback Search:**  The `Saddleback` function correctly implements the saddleback search algorithm, efficiently using the sorted nature of the rows and columns. It starts at the top right corner and moves down or left based on the comparison with the target.
