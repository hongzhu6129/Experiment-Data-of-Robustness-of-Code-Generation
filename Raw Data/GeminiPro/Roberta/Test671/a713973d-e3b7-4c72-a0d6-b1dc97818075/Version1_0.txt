```java
import java.util.Map;
import java.util.NavigableMap;
import java.util.NavigableSet;
import java.util.TreeMap;

public class NavigableMapKeysView {

    public static void main(String[] args) {

        // Create a TreeMap, which implements NavigableMap
        NavigableMap<String, Integer> studentScores = new TreeMap<>();

        // Populate the TreeMap with some sample data
        studentScores.put("Alice", 90);
        studentScores.put("Bob", 85);
        studentScores.put("Charlie", 95);
        studentScores.put("David", 78);
        studentScores.put("Eve", 88);


        // 1. Get a NavigableSet view of the keys
        NavigableSet<String> keys = studentScores.navigableKeySet();

        // Print the original TreeMap
        System.out.println("Original TreeMap: " + studentScores);

        // Print the NavigableSet of keys
        System.out.println("NavigableSet of keys: " + keys);


        // Demonstrate some NavigableSet operations:

        // 2. Get the first (lowest) key
        String firstKey = keys.first();
        System.out.println("First key: " + firstKey);


        // 3. Get the last (highest) key
        String lastKey = keys.last();
        System.out.println("Last key: " + lastKey);


        // 4. Get the key that is less than or equal to "Charlie"
        String floorKey = keys.floor("Charlie");
        System.out.println("Floor key of 'Charlie': " + floorKey);


        // 5. Get the key that is greater than or equal to "David"
        String ceilingKey = keys.ceiling("David");
        System.out.println("Ceiling key of 'David': " + ceilingKey);


        // 6. Get the key that is strictly less than "Eve"
        String lowerKey = keys.lower("Eve");
        System.out.println("Lower key of 'Eve': " + lowerKey);


        // 7. Get the key that is strictly greater than "Bob"
        String higherKey = keys.higher("Bob");
        System.out.println("Higher key of 'Bob': " + higherKey);


        // 8. Get a descending set of keys
        NavigableSet<String> descendingKeys = keys.descendingSet();  // Doesn't modify original 'keys'
        System.out.println("Descending set of keys: " + descendingKeys);



        // Important Note:  Changes to the NavigableSet view (keys) will be reflected in the
        // original TreeMap (studentScores), and vice-versa because it is a *view*, not a copy.

        keys.remove("Bob"); // Remove "Bob" from the keyset
        System.out.println("TreeMap after removing 'Bob' from keyset: " + studentScores); // Bob is also removed from TreeMap




    }
}
```
