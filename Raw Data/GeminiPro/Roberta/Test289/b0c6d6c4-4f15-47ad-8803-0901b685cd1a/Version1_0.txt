import java.util.Scanner;

public class CompareFourIntegers {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        System.out.print("Input first number: ");
        int num1 = input.nextInt();

        System.out.print("Input second number: ");
        int num2 = input.nextInt();

        System.out.print("Input third number: ");
        int num3 = input.nextInt();

        System.out.print("Input fourth number: ");
        int num4 = input.nextInt();

        if (num1 == num2 && num2 == num3 && num3 == num4) {
            System.out.println("Numbers are equal"); // Corrected output message
        } else {
            System.out.println("Numbers are not equal"); // Corrected output message
        }

        input.close(); // Close the scanner to release resources
    }
}
