```java
public class DivideTwoNumbers {

    public static void main(String[] args) {

        int numerator = 50;
        int denominator = 3;

        // Perform integer division
        int result = numerator / denominator;

        // Print the result to the console
        System.out.println(numerator + "/" + denominator + ": " + result);


        // For a more precise result (with decimals), use floating-point division:
        double doubleNumerator = 50.0;  // Or (double) numerator;
        double doubleDenominator = 3.0; // Or (double) denominator;
        double doubleResult = doubleNumerator / doubleDenominator;

        System.out.println(doubleNumerator + "/" + doubleDenominator + ": " + doubleResult);



        // Handling potential ArithmeticException (division by zero)
        try {
            int zeroDenominator = 0;
            int zeroResult = numerator / zeroDenominator;  // This will cause an exception
            System.out.println(numerator + "/" + zeroDenominator + ": " + zeroResult); // This won't be reached

        } catch (ArithmeticException e) {
            System.out.println("Error: Division by zero is not allowed.");
        }
    }
}
```
