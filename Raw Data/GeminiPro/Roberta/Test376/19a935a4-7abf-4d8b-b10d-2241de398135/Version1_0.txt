public class ColorPicker {

    public String pickColor(String input) {
        if (input != null && (input.startsWith("red") || input.startsWith("black"))) {
            return input.startsWith("red") ? "red" : "black";
        } else {
            return "";
        }
    }

    public static void main(String[] args) {
        ColorPicker cp = new ColorPicker();

        System.out.println(cp.pickColor("reddish")); // Output: red
        System.out.println(cp.pickColor("blackish")); // Output: black
        System.out.println(cp.pickColor("green")); // Output: 
        System.out.println(cp.pickColor(null)); // Output: 
        System.out.println(cp.pickColor("red")); // Output: red
        System.out.println(cp.pickColor("black")); // Output: black
    }
}
