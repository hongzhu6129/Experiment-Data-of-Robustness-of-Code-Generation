import java.util.TreeMap;

public class LeastKeyStrictlyGreaterThan {

    public static void main(String[] args) {
        // Example usage:
        TreeMap<Integer, String> treeMap = new TreeMap<>();
        treeMap.put(1, "One");
        treeMap.put(3, "Three");
        treeMap.put(5, "Five");
        treeMap.put(7, "Seven");
        treeMap.put(9, "Nine");

        Integer key1 = 4;
        Integer result1 = getLeastKeyStrictlyGreaterThan(treeMap, key1);
        System.out.println("Least key strictly greater than " + key1 + ": " + result1); // Output: 5

        Integer key2 = 1;
        Integer result2 = getLeastKeyStrictlyGreaterThan(treeMap, key2);
        System.out.println("Least key strictly greater than " + key2 + ": " + result2); // Output: 3

        Integer key3 = 9;
        Integer result3 = getLeastKeyStrictlyGreaterThan(treeMap, key3);
        System.out.println("Least key strictly greater than " + key3 + ": " + result3); // Output: null


        TreeMap<String, Integer> stringTreeMap = new TreeMap<>();
        stringTreeMap.put("apple", 1);
        stringTreeMap.put("banana", 2);
        stringTreeMap.put("orange", 3);

        String key4 = "apricot";  // lexicographically between "apple" and "banana"
        String result4 = getLeastKeyStrictlyGreaterThan(stringTreeMap, key4);
        System.out.println("Least key strictly greater than " + key4 + ": " + result4); // Output: banana

         String key5 = "zoo";
        String result5 = getLeastKeyStrictlyGreaterThan(stringTreeMap, key5);
        System.out.println("Least key strictly greater than " + key5 + ": " + result5); // Output: null
    }

    public static <K, V> K getLeastKeyStrictlyGreaterThan(TreeMap<K, V> treeMap, K key) {
        return treeMap.higherKey(key);
    }
}
