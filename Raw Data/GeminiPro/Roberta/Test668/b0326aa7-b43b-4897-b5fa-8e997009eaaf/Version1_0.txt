import java.util.TreeMap;

public class LeastKeyGreaterThan {

    public static void main(String[] args) {
        // Example usage:
        TreeMap<Integer, String> treeMap = new TreeMap<>();
        treeMap.put(10, "Ten");
        treeMap.put(20, "Twenty");
        treeMap.put(30, "Thirty");
        treeMap.put(40, "Forty");
        treeMap.put(50, "Fifty");

        Integer key1 = 35;
        Integer result1 = getLeastKeyGreaterThan(treeMap, key1);
        System.out.println("Least key greater than " + key1 + ": " + result1); // Output: 40

        Integer key2 = 15;
        Integer result2 = getLeastKeyGreaterThan(treeMap, key2);
        System.out.println("Least key greater than " + key2 + ": " + result2); // Output: 20

        Integer key3 = 55;
        Integer result3 = getLeastKeyGreaterThan(treeMap, key3);
        System.out.println("Least key greater than " + key3 + ": " + result3); // Output: null


        // Additional Test cases
        TreeMap<Integer, String> emptyTreeMap = new TreeMap<>();
        Integer key4 = 5;
        Integer result4 = getLeastKeyGreaterThan(emptyTreeMap, key4);
        System.out.println("Least key greater than " + key4 + ": " + result4); // Output: null



        TreeMap<Integer, String> singleTreeMap = new TreeMap<>();
         singleTreeMap.put(100, "Hundred");

        Integer key5 = 90;
        Integer result5 = getLeastKeyGreaterThan(singleTreeMap, key5);
        System.out.println("Least key greater than " + key5 + ": " + result5); // Output: 100


        Integer key6 = 110;
        Integer result6 = getLeastKeyGreaterThan(singleTreeMap, key6);
        System.out.println("Least key greater than " + key6 + ": " + result6); // Output: null

    }



    public static <K,V> K getLeastKeyGreaterThan(TreeMap<K,V> treeMap, K key) {
        if (treeMap == null || treeMap.isEmpty()) {
            return null;
        }

        return treeMap.higherKey(key);
    }
}
