import java.util.Arrays;
import java.util.PriorityQueue;

public class KLargestElements {

    public static void main(String[] args) {
        int[] arr = {1, 23, 12, 9, 30, 2, 50};
        int k = 3;
        findKLargest(arr, k);

        arr = new int[]{11, 5, 2, 7, 1, 15, 8, 3};
        k = 4;
        findKLargest(arr, k);
    }

    public static void findKLargest(int[] arr, int k) {
        if (k <= 0 || k > arr.length) {
            System.out.println("Invalid value of k");
            return;
        }

        // Method 1: Using a min-heap (Priority Queue) - More efficient for larger datasets
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        for (int num : arr) {
            minHeap.offer(num);
            if (minHeap.size() > k) {
                minHeap.poll(); // Remove the smallest element if heap size exceeds k
            }
        }

        Integer[] result = minHeap.toArray(new Integer[0]); // Convert to array
        Arrays.sort(result); // Sort to get the k largest in ascending order (optional)
        System.out.print("K Largest elements (using Min-Heap): ");
        for(int num : result) { // Enhanced for loop
            System.out.print(num + " "); 
        }
        System.out.println();



        // Method 2: Sorting and then printing the last k elements - Simpler, but less efficient for large k
        Arrays.sort(arr); // Sorts the array in ascending order. Modifies the original array.
        System.out.print("K Largest elements (using Sorting): ");
        for (int i = arr.length - k; i < arr.length; i++) {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
    }
}
