```java
public class SwapVariables {

    public static void main(String[] args) {

        // Method 1: Using a third variable
        int a = 10;
        int b = 20;

        System.out.println("Before swapping (using third variable): a = " + a + ", b = " + b);

        int temp = a;
        a = b;
        b = temp;

        System.out.println("After swapping (using third variable): a = " + a + ", b = " + b);


        // Method 2: Using addition and subtraction (for numeric types)
        int x = 5;
        int y = 15;

        System.out.println("Before swapping (using addition/subtraction): x = " + x + ", y = " + y);

        x = x + y;
        y = x - y;
        x = x - y;

        System.out.println("After swapping (using addition/subtraction): x = " + x + ", y = " + y);



        // Method 3: Using bitwise XOR (for numeric types)
        int p = 7;
        int q = 25;

        System.out.println("Before swapping (using XOR): p = " + p + ", q = " + q);

        p = p ^ q;
        q = p ^ q;
        p = p ^ q;


        System.out.println("After swapping (using XOR): p = " + p + ", q = " + q);

    }
}
```
