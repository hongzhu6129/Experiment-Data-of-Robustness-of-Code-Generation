public class DealTwoVariables {

    public static void main(String[] args) {
        // Declare two variables of the same type (e.g., int, double, String)
        int a = 10;
        int b = 20;

        System.out.println("Before dealing:");
        System.out.println("a = " + a);
        System.out.println("b = " + b);

        // Deal (swap) the values of the variables.  There are a few ways:

        // 1. Using a temporary variable
        int temp = a;
        a = b;
        b = temp;

        // 2. Using addition and subtraction (only for numeric types)
        // a = a + b;
        // b = a - b;
        // a = a - b;

        // 3. Using bitwise XOR (only for integer types)
        // a = a ^ b;
        // b = a ^ b;
        // a = a ^ b;


        System.out.println("\nAfter dealing:");
        System.out.println("a = " + a);
        System.out.println("b = " + b);
    }
}
