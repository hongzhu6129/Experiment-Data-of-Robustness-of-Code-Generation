```java
import java.util.Map;
import java.util.TreeMap;
import java.util.SortedMap;

public class TailMapExample {

    public static void main(String[] args) {
        // Create a TreeMap (naturally sorted by keys)
        TreeMap<Integer, String> map = new TreeMap<>();
        map.put(1, "One");
        map.put(2, "Two");
        map.put(3, "Three");
        map.put(4, "Four");
        map.put(5, "Five");
        map.put(6, "Six");


        // Get the portion of the map whose keys are greater than 3 (exclusive)
        SortedMap<Integer, String> tailMap = map.tailMap(4);  // Key 4 and greater

        // Print the tailMap
        System.out.println("Tail Map (exclusive): " + tailMap);

        // Get the portion of the map whose keys are greater than or equal to 3 (inclusive)
        SortedMap<Integer, String> tailMapInclusive = map.tailMap(3, true); // Key 3 and greater

        // Print the tailMapInclusive
        System.out.println("Tail Map (inclusive): " + tailMapInclusive);


        // Example with Strings as keys (ensure they are comparable)
        TreeMap<String, Integer> stringMap = new TreeMap<>();
        stringMap.put("apple", 1);
        stringMap.put("banana", 2);
        stringMap.put("orange", 3);
        stringMap.put("pear", 4);

        SortedMap<String, Integer> stringTailMap = stringMap.tailMap("banana"); // "banana" and after, alphabetically

        System.out.println("String Tail Map: " + stringTailMap);
    }
}
```
