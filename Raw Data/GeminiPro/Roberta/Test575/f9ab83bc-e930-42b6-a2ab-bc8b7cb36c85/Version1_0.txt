```java
import java.util.ArrayList;
import java.util.List;

public class ArrayListPrintWithIndex {

    public static void main(String[] args) {

        // Create an ArrayList of strings
        List<String> fruits = new ArrayList<>();
        fruits.add("Apple");
        fruits.add("Banana");
        fruits.add("Orange");
        fruits.add("Grape");
        fruits.add("Mango");


        // Method 1: Using a traditional for loop (index-based)
        System.out.println("Method 1: Traditional for loop:");
        for (int i = 0; i < fruits.size(); i++) {
            System.out.println("Element at index " + i + ": " + fruits.get(i));
        }

        System.out.println("\n"); // Add a newline for better readability

        // Method 2: Using an enhanced for loop (for-each loop) with a separate index variable.
        //  While the enhanced for loop doesn't directly give you the index, 
        // you can maintain one yourself if needed.
        System.out.println("Method 2: Enhanced for loop with index:");
        int index = 0;
        for (String fruit : fruits) {
            System.out.println("Element at index " + index + ": " + fruit);
            index++;
        }


        System.out.println("\n"); // Add a newline


        // Method 3: Using Java 8 streams and forEach with an IntStream to generate the index
        System.out.println("Method 3: Java 8 Streams with IntStream for index:");

       java.util.stream.IntStream.range(0, fruits.size())
                .forEach(i -> System.out.println("Element at index " + i +  ": " + fruits.get(i)));

    }
}
```
