```java
import java.util.TreeSet;

public class RemoveFromTreeSet {

    public static void main(String[] args) {
        // Create a TreeSet
        TreeSet<Integer> numbers = new TreeSet<>();

        // Add elements to the TreeSet
        numbers.add(10);
        numbers.add(20);
        numbers.add(30);
        numbers.add(40);
        numbers.add(50);

        System.out.println("Original TreeSet: " + numbers);

        // Element to discard
        int elementToRemove = 30;

        // Discard the element
        boolean removed = numbers.remove(elementToRemove); // returns true if removed, false otherwise

        if (removed) {
            System.out.println("TreeSet after discarding " + elementToRemove + ": " + numbers);
        } else {
            System.out.println(elementToRemove + " not found in the TreeSet.");
        }


        //Example demonstrating removing an element not present:
        int elementNotPresent = 15;
        boolean removed2 = numbers.remove(elementNotPresent);
        if (removed2) {
             System.out.println("TreeSet after discarding " + elementNotPresent + ": " + numbers);
        } else {
            System.out.println(elementNotPresent + " not found in the TreeSet.");
        }


    }
}
```
