```java
import java.util.HashSet;

public class HashSetIsEmptyCheck {

    public static void main(String[] args) {

        // Create a HashSet
        HashSet<String> hashSet = new HashSet<>();

        // Check if the HashSet is empty using isEmpty()
        boolean isEmpty = hashSet.isEmpty();

        // Print the result
        if (isEmpty) {
            System.out.println("The HashSet is empty.");
        } else {
            System.out.println("The HashSet is not empty."); 
        }


        // Add some elements to the HashSet
        hashSet.add("Apple");
        hashSet.add("Banana");
        hashSet.add("Orange");

        // Check again if the HashSet is empty
        isEmpty = hashSet.isEmpty();

         // Print the result
        if (isEmpty) {
            System.out.println("The HashSet is empty.");
        } else {
            System.out.println("The HashSet is not empty."); 
        }


        // Example using size() to check for emptiness (alternative, but isEmpty() is generally preferred)
        if (hashSet.size() == 0) {
            System.out.println("The HashSet is empty (using size()).");
        } else {
            System.out.println("The HashSet is not empty (using size()).");
        }


    }
}
```
