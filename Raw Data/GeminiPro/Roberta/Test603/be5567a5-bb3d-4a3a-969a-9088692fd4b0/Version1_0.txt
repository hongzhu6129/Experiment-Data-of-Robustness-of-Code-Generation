```java
import java.util.ArrayList;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Iterator;
import java.util.List;
import java.util.Map;
import java.util.Set;

public class IterateThroughHashList {

    public static void main(String[] args) {

        // Demonstrating with a List of HashMaps (a common "Hash List" scenario)
        List<HashMap<String, Integer>> listOfHashMaps = new ArrayList<>();

        // Create some sample HashMaps and populate them
        HashMap<String, Integer> map1 = new HashMap<>();
        map1.put("apple", 1);
        map1.put("banana", 2);

        HashMap<String, Integer> map2 = new HashMap<>();
        map2.put("orange", 3);
        map2.put("grape", 4);


        listOfHashMaps.add(map1);
        listOfHashMaps.add(map2);


        // Method 1: Enhanced for loop (for each loop) - Simplest for iterating through the List
        System.out.println("Method 1: Enhanced for loop (List)");
        for (HashMap<String, Integer> map : listOfHashMaps) { 
            // For each HashMap in the List:

            System.out.println("Map: "); // Indicate a new map is being processed.

            for (Map.Entry<String, Integer> entry : map.entrySet()) {
                System.out.println("Key: " + entry.getKey() + ", Value: " + entry.getValue());
            }
        }
        System.out.println("------------------------");


        // Method 2: Standard for loop with indexes - More control if you need the index
        System.out.println("Method 2: Standard for loop (List)");
        for (int i = 0; i < listOfHashMaps.size(); i++) {
            HashMap<String, Integer> map = listOfHashMaps.get(i);
            System.out.println("Map at index " + i + ":");

            for (Map.Entry<String, Integer> entry : map.entrySet()) {
                System.out.println("Key: " + entry.getKey() + ", Value: " + entry.getValue());
            }
        }
        System.out.println("------------------------");



        // Method 3: Iterator - offers remove() capability while iterating
        System.out.println("Method 3: Iterator (List)");
        Iterator<HashMap<String, Integer>> listIterator = listOfHashMaps.iterator();
        while (listIterator.hasNext()) {
            HashMap<String, Integer> map = listIterator.next();


            Iterator<Map.Entry<String, Integer>> mapIterator = map.entrySet().iterator();
            while (mapIterator.hasNext()) {
                Map.Entry<String, Integer> entry = mapIterator.next();
                System.out.println("Key: " + entry.getKey() + ", Value: " + entry.getValue());
            }

        }
        System.out.println("------------------------");



        // Demonstrating with a HashSet of custom objects (another "hash" structure):
        Set<MyObject> myObjects = new HashSet<>();
        myObjects.add(new MyObject("Object1", 10));
        myObjects.add(new MyObject("Object2", 20));


        System.out.println("Iterating through a HashSet:");
        for(MyObject obj : myObjects) {  // Enhanced for loop is most convenient for Sets
            System.out.println("Name: " + obj.name + ", Value: " + obj.value);
        }
        

    }



    // Simple example custom object
    static class MyObject {
        String name;
        int value;

        public MyObject(String name, int value) {
            this.name = name;
            this.value = value;
        }
    }

}
```
