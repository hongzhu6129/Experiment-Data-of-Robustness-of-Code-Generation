```java
import java.util.TreeSet;

public class TreeSetLessThanSeven {

    public static void main(String[] args) {
        // Create a TreeSet of integers
        TreeSet<Integer> numbers = new TreeSet<>();

        // Add some numbers to the TreeSet
        numbers.add(2);
        numbers.add(9);
        numbers.add(5);
        numbers.add(1);
        numbers.add(12);
        numbers.add(7);  // Include 7 to demonstrate exclusion
        numbers.add(3);

        // Find elements less than 7 (exclusive)
        TreeSet<Integer> lessThanSeven = (TreeSet<Integer>) numbers.headSet(7); // headSet is exclusive of the 'toElement'

        // Print the elements less than 7
        System.out.println("Elements less than 7: " + lessThanSeven);


        // Another way using streams and lambda expression (Java 8+):
        long countLessThanSeven = numbers.stream().filter(n -> n < 7).count();
        System.out.println("Number of elements less than 7 (using streams): " + countLessThanSeven);

       // Or, to get the set of those numbers using streams:
        TreeSet<Integer> lessThanSevenStream = numbers.stream()
                .filter(n -> n < 7)
                .collect(TreeSet::new, TreeSet::add, TreeSet::addAll); // Efficiently collect into a new TreeSet

        System.out.println("Elements less than 7 (using streams and collecting to TreeSet): " + lessThanSevenStream);

    }
}
```
