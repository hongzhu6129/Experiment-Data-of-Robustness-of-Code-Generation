import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class MathExpressionValidator {

    public static void main(String[] args) {
        String expression1 = "2 + 3 * 4";
        String expression2 = "x = 5";
        String expression3 = "Hello, world!";
        String expression4 = "10 / (2 + 3)";
        String expression5 = "5 + * 7"; // Invalid: consecutive operators
        String expression6 = "8 -"; // Invalid: trailing operator


        System.out.println("\"" + expression1 + "\" is a valid math expression: " + validate(expression1)); // True
        System.out.println("\"" + expression2 + "\" is a valid math expression: " + validate(expression2)); // True - assignment is considered valid here
        System.out.println("\"" + expression3 + "\" is a valid math expression: " + validate(expression3)); // False
        System.out.println("\"" + expression4 + "\" is a valid math expression: " + validate(expression4)); // True
        System.out.println("\"" + expression5 + "\" is a valid math expression: " + validate(expression5)); // False
        System.out.println("\"" + expression6 + "\" is a valid math expression: " + validate(expression6)); // False


    }

    public static boolean validate(String expression) {
        // Regular expression to match basic mathematical expressions.
        // Allows numbers, operators (+, -, *, /), parentheses, decimal points, 
        // whitespace, and also the equals sign for assignments.
        // It also checks for consecutive operators which are invalid.

        String regex = "^[\\d\\s+\\-*/().=x]+$";  // Basic validation (allows variables like 'x') -  can be expanded.
        String consecutiveOperatorsRegex = "[+\\-*/]{2,}"; // Checks for two or more consecutive operators

        if (expression == null || expression.trim().isEmpty()) {
            return false;
        }


        // Remove whitespace for easier checking of consecutive operators
        String expressionWithoutWhitespace = expression.replaceAll("\\s+", "");

        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(expression);

        Pattern consecutiveOperatorsPattern = Pattern.compile(consecutiveOperatorsRegex);
        Matcher consecutiveOperatorsMatcher = consecutiveOperatorsPattern.matcher(expressionWithoutWhitespace);


        return matcher.matches() && !consecutiveOperatorsMatcher.find();

    }
}
