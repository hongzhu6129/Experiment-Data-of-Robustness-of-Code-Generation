```java
import java.util.LinkedList;

public class RemoveAllLinkedList {

    public static void main(String[] args) {
        // Create a LinkedList of strings
        LinkedList<String> books = new LinkedList<>();

        // Add some elements to the LinkedList
        books.add("The Lord of the Rings");
        books.add("The Hitchhiker's Guide to the Galaxy");
        books.add("Pride and Prejudice");
        books.add("1984");


        System.out.println("Original LinkedList: " + books);

        // Remove all elements using clear()
        books.clear();


        System.out.println("LinkedList after removing all elements: " + books);


        // Demonstrating another way to remove all elements (by assigning a new empty list)
        books = new LinkedList<>(); // This also effectively clears the list


        System.out.println("LinkedList after assigning a new empty list : " + books);
    }
}
```
