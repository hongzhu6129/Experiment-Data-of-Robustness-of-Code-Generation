public class SubstringExample {

    public static void main(String[] args) {
        String str = "programming";
        int start = 3;
        int end = 7;


        if (start >= 0 && end <= str.length() && start <= end) {
            String subStr = str.substring(start, end);
            System.out.println("Substring: " + subStr);  // Output: gramm
        } else {
            System.out.println("Invalid start or end positions."); 
        }


        // Another example with different start and end
        start = 0;
        end = 4;

        if (start >= 0 && end <= str.length() && start <= end) {
            String subStr = str.substring(start, end);
            System.out.println("Substring: " + subStr); // Output: prog
        } else {
            System.out.println("Invalid start or end positions.");
        }


         // Example demonstrating handling of invalid input
        start = 7;
        end = 3;

        if (start >= 0 && end <= str.length() && start <= end) {
            String subStr = str.substring(start, end);
            System.out.println("Substring: " + subStr);
        } else {
            System.out.println("Invalid start or end positions.");  // This will be printed
        }


        // Example demonstrating handling of out-of-bounds input
        start = -1;
        end = 12;


        if (start >= 0 && end <= str.length() && start <= end) {
            String subStr = str.substring(start, end);
            System.out.println("Substring: " + subStr);
        } else {
            System.out.println("Invalid start or end positions.");  // This will be printed
        }
    }
}
