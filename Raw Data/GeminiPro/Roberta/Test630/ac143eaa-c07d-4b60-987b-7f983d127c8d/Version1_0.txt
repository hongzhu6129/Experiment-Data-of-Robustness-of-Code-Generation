        PriorityQueue<String> colorQueue = new PriorityQueue<>();

        colorQueue.add("Green");
        colorQueue.add("Blue");
        colorQueue.add("Yellow");
        colorQueue.add("Orange");


        System.out.println("Elements of the Priority Queue:");

        // Method 1:  Iterate and remove elements (destructive)
        System.out.println("Printing and removing elements (using poll()):");
        while (!colorQueue.isEmpty()) {
            System.out.println(colorQueue.poll()); // poll() removes and returns the head (highest priority)
        }


        // Re-add the colors for the next demonstration (since the previous loop emptied the queue)

        colorQueue.add("Blue");

        // Method 2: Iterate without removing (non-destructive) using enhanced for loop (Java 5+)
