```java
public class OctalToDecimal {

    public static void main(String[] args) {
        // Input octal number as a string (to handle larger numbers)
        String octalString = "10"; 

        try {
            // Convert the octal string to decimal using Integer.parseInt
            int decimal = Integer.parseInt(octalString, 8);

            // Print the equivalent decimal number
            System.out.println("Equivalent decimal number: " + decimal);

        } catch (NumberFormatException e) {
            System.out.println("Invalid octal number format.");
        }
    }
}
```
