class Solution {
    /**
     * Educational Java code to search for a target value in an m x n matrix.
     *
     * This solution treats the matrix as a sorted array. Because the matrix is sorted row-wise and
     * the first integer of each row is greater than the last integer of the previous row, we can
     * perform a binary search on the entire matrix.
     *
     * Time Complexity: O(log(m*n)) - Binary Search on the virtual array.
     * Space Complexity: O(1) - Constant extra space.
     *
     * @param matrix The m x n matrix to search.
     * @param target The value to search for.
     * @return True if the target is found, false otherwise.
     */
    boolean searchMatrix(int[][] matrix, int target) {
        if (matrix == null || matrix.length == 0 || matrix[0].length == 0) {
            return false; // Handle empty or null matrix.
        }

        int m = matrix.length;
        int n = matrix[0].length;
        int left = 0;
        int right = m * n - 1;

        while (left <= right) {
            int mid = left + (right - left) / 2; // Prevent potential integer overflow

            // Calculate row and column indices from the virtual index.
            int row = mid / n;
            int col = mid % n;

            if (matrix[row][col] == target) {
                return true; // Target found.
            } else if (matrix[row][col] < target) {
                left = mid + 1; // Search in the right half.
            } else {
                right = mid - 1; // Search in the left half.
            }
        }
        return false; // Target not found.
    }


    public static void main(String[] args) {
        Solution solution = new Solution();

        int[][] matrix1 = {{1, 3, 5, 7}, {10, 11, 16, 20}, {23, 30, 34, 60}};
        int target1 = 3;
        System.out.println("Matrix 1, Target 3: " + solution.searchMatrix(matrix1, target1)); // Expected: true

        int[][] matrix2 = {{1, 3, 5, 7}, {10, 11, 16, 20}, {23, 30, 34, 60}};
        int target2 = 13;
        System.out.println("Matrix 2, Target 13: " + solution.searchMatrix(matrix2, target2)); // Expected: false

        // Test cases with edge cases
        int[][] emptyMatrix = {};
        int target3 = 5;
        System.out.println("Empty Matrix, Target 5: " + solution.searchMatrix(emptyMatrix, target3)); // Expected: false

        int[][] nullMatrix = null;
        int target4 = 1;
        System.out.println("Null Matrix, Target 1: " + solution.searchMatrix(nullMatrix, target4)); // Expected: false

        int[][] matrix3 = {{1}};
        int target5 = 1;
        System.out.println("Matrix 3, Target 1: " + solution.searchMatrix(matrix3, target5)); // Expected: true


    }
}
