class Node {
    int data;
    Node left, right;

    public Node(int item) {
        data = item;
        left = right = null;
    }
}

public class BinaryTreeMaxDepth {

    public int maxDepth(Node node) {
        if (node == null) {
            return 0;
        } else {
            // Recursively calculate the depth of the left and right subtrees
            int leftDepth = maxDepth(node.left);
            int rightDepth = maxDepth(node.right);

            // Return the larger of the two depths, plus 1 for the current node
            return Math.max(leftDepth, rightDepth) + 1;
        }
    }


    public static void main(String[] args) {
        BinaryTreeMaxDepth tree = new BinaryTreeMaxDepth();

        // Construct a sample binary tree
        Node root = new Node(1);
        root.left = new Node(2);
        root.right = new Node(3);
        root.left.left = new Node(4);
        root.left.right = new Node(5);

        // Calculate and print the maximum depth
        int depth = tree.maxDepth(root);
        System.out.println("Maximum depth of the tree is: " + depth);


        //Another Test Case: Empty Tree
        Node emptyRoot = null;
        int emptyDepth = tree.maxDepth(emptyRoot);
        System.out.println("Maximum depth of the empty tree is: " + emptyDepth); // Output should be 0


        // Another test case: Skewed tree (all nodes on one side)
        Node skewedRoot = new Node(1);
        skewedRoot.right = new Node(2);
        skewedRoot.right.right = new Node(3);
        int skewedDepth = tree.maxDepth(skewedRoot);
        System.out.println("Maximum depth of the skewed tree is: " + skewedDepth); // Output should be 3
    }
}
