```java
import java.util.TreeSet;

public class TreeSetColors {

    public static void main(String[] args) {
        TreeSet<String> colors = new TreeSet<>();

        // Add some colors
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Orange");
        colors.add("Purple");

        // Print the TreeSet (elements will be sorted naturally)
        System.out.println(colors); 
    }
}
```
