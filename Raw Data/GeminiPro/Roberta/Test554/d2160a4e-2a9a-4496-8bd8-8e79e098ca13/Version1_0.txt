import java.util.ArrayList;
import java.util.List;

public class ColorList {

    public static void main(String[] args) {

        // Add some colors to the list
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Purple");

        // Print out the collection
        System.out.println(colors);

}
