public class LargestFirstLast {

    public static void main(String[] args) {

        System.out.println("Largest between first and last in [5, 1, 4]: " + largest2); // Output: 5

        System.out.println("Largest between first and last in [7, 7, 7]: " + largest3); // Output: 7

        int largest4 = findLargestFirstLast(array4);
        System.out.println("Largest between first and last in [0,0,0]: " + largest4); // Output: 0

    public static int findLargestFirstLast(int[] arr) {
        if (arr == null || arr.length == 0 ) {
            System.err.println("Error: Input array is null or empty.");
            return -1; // Or throw an exception
        }
         if(arr.length != 3) {
             System.err.println("Error: The length of the array must be 3");
             return -1;
         }


        int first = arr[0];
        int last = arr[arr.length - 1];

        return (first > last) ? first : last;
    }
}
