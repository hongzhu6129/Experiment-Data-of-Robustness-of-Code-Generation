```java
import java.util.HashMap;
import java.util.Map;

public class ClearMap {

    public static void main(String[] args) {
        // Create a sample map
        Map<String, Integer> myMap = new HashMap<>();
        myMap.put("Apple", 1);
        myMap.put("Banana", 2);
        myMap.put("Orange", 3);

        // Print the original map
        System.out.println("Original Map: " + myMap);

        // Clear the map
        myMap.clear();

        // Print the cleared map
        System.out.println("Cleared Map: " + myMap);


        // Example with different data types
        Map<Integer, Double> anotherMap = new HashMap<>();
        anotherMap.put(1, 1.1);
        anotherMap.put(2, 2.2);

        System.out.println("Another Original Map: " + anotherMap);
        anotherMap.clear();
        System.out.println("Another Cleared Map: " + anotherMap);
    }
}
```
