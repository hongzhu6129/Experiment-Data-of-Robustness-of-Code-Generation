```java
import java.util.Collection;
import java.util.HashMap;
import java.util.Map;

public class Main {
    public static void main(String[] args) {
        // Create a sample map
        Map<String, Integer> map = new HashMap<>();
        map.put("Apple", 1);
        map.put("Banana", 2);
        map.put("Orange", 3);

        // Get a Collection view of the values
        Collection<Integer> values = map.values();

        // Print the values
        System.out.println("Values in the map:");
        for (Integer value : values) {
            System.out.println(value);
        }


        //Demonstrate that changes in the collection are reflected in the map (and vice-versa)
        System.out.println("\nDemonstrating linked changes:");

        // Remove a value from the collection view
        values.remove(2); 

        System.out.println("Map after removing 2 from the values collection:");
        System.out.println(map); // Banana is gone because its value was 2


         // Add a value to the map directly
        map.put("Grape", 4);

        System.out.println("Values collection after adding 'Grape':4 to the map directly:");
        System.out.println(values); // 4 (the value of Grape) is now in the values collection

    }
}
```
