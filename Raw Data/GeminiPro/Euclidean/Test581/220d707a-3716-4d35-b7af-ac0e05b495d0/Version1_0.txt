```java
import java.util.LinkedList;

public class LinkedListInsertion {

    public static void main(String[] args) {
        // Create a new LinkedList
        LinkedList<String> linkedList = new LinkedList<>();

        // Insert elements at the beginning (head/took)
        linkedList.addFirst("Apple");  // Inserts "Apple" at the beginning
        linkedList.push("Banana"); // Another way to insert at the beginning (same as addFirst)
        linkedList.offerFirst("Cherry"); // Yet another way to insert at the beginning


        // Insert elements at the end (tail/last)
        linkedList.addLast("Date"); // Inserts "Date" at the end
        linkedList.add("Fig");     //  Appends "Fig" to the end (same as addLast)
        linkedList.offerLast("Grape"); // Another way to insert at the end


        // Print the LinkedList
        System.out.println("Linked List: " + linkedList);

        //Common Errors & Troubleshooting
        // NullPointerException: if you try to access elements from an empty LinkedList. Always check if it's empty using isEmpty() before accessing elements.
        // IndexOutOfBoundsException:  if you try to access elements using an index that is out of range (e.g., negative or greater than or equal to the size of the list).


    }
}
```
