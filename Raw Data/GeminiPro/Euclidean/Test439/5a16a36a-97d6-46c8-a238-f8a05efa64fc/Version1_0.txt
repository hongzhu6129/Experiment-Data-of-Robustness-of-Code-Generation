```java
import java.util.Calendar;
import java.util.GregorianCalendar;

public class CalendarInfo {

    public static void main(String[] args) {
        // Get a default calendar instance (GregorianCalendar)
        Calendar calendar = new GregorianCalendar();

        // Get year, month, day of week, hour, and minute
        int year = calendar.get(Calendar.YEAR);
        int month = calendar.get(Calendar.MONTH); // Month is 0-indexed (0 = January, 1 = February, etc.)
        int dayOfWeek = calendar.get(Calendar.DAY_OF_WEEK); // 1 = Sunday, 2 = Monday, ..., 7 = Saturday
        int hour = calendar.get(Calendar.HOUR_OF_DAY); // 24-hour format
        int minute = calendar.get(Calendar.MINUTE);

        // Determine if it's a weekend
        boolean isWeekend = (dayOfWeek == Calendar.SATURDAY || dayOfWeek == Calendar.SUNDAY);


        // Display the information
        System.out.println("Current Date and Time Information:");
        System.out.println("Year: " + year);
        System.out.println("Month: " + (month + 1)); // Add 1 to make it 1-indexed
        System.out.println("Day of Week: " + dayOfWeek);  // You can convert this to a String representation if desired.
        System.out.println("Is Weekend: " + isWeekend);
        System.out.println("Hour: " + hour);
        System.out.println("Minute: " + minute);



    }
}
```
