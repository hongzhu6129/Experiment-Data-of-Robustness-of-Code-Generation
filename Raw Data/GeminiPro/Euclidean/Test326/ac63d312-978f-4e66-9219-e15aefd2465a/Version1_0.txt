```java
public class StringCompare {

    public static void main(String[] args) {
        String str1 = "This is a test string.";
        CharSequence charSeq = "This is a test string.";

        // Method 1: Using String.contentEquals()
        boolean isEqual1 = str1.contentEquals(charSeq);
        System.out.println("Comparison using contentEquals(): " + isEqual1);


        // Method 2: Converting CharSequence to String and comparing
        String str2 = charSeq.toString();
        boolean isEqual2 = str1.equals(str2);
        System.out.println("Comparison after converting to String: " + isEqual2);



        //Demonstrating a case where they are NOT equal:
        String str3 = "This is a test String."; //Capital 'S'
        CharSequence charSeq2 = "This is a test string.";


        boolean isEqual3 = str3.contentEquals(charSeq2);
        System.out.println("Comparison with a difference (using contentEquals()): " + isEqual3);

         boolean isEqual4 = str3.equals(charSeq2.toString());
        System.out.println("Comparison with a difference (after converting): " + isEqual4);


    }
}
```
