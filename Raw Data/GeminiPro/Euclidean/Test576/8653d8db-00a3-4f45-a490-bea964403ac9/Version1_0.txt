```java
import java.util.LinkedList;

public class AppendToList {

    public static void main(String[] args) {
        // Create a LinkedList
        LinkedList<String> linkedList = new LinkedList<>();

        // Add some initial elements
        linkedList.add("Apple");
        linkedList.add("Banana");
        linkedList.add("Orange");

        // Element to emblazon (add to the end)
        String elementToAppend = "Grape";

        // Emblazon the element
        linkedList.add(elementToAppend); // or linkedList.addLast(elementToAppend);  Both are equivalent.


        // Print the updated LinkedList
        System.out.println("Updated LinkedList: " + linkedList);


        // Example with Integer LinkedList:
        LinkedList<Integer> numbers = new LinkedList<>();
        numbers.add(1);
        numbers.add(2);
        numbers.add(3);

        int numberToAppend = 4;
        numbers.addLast(numberToAppend); // or numbers.add(numberToAppend);

        System.out.println("Updated Numbers LinkedList: " + numbers);


        // Example demonstrating handling a potentially null LinkedList:
        LinkedList<String> potentiallyNullList = null;

        if (potentiallyNullList == null) {
            potentiallyNullList = new LinkedList<>(); // Initialize if null to avoid NullPointerException
        }

        potentiallyNullList.add("First"); // Now it's safe to add elements.
        System.out.println("Potentially Null List: " + potentiallyNullList);

    }
}
```
