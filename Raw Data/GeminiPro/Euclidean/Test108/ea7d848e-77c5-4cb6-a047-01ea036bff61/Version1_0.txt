class Solution {
    /**
     * Gets the first occurrence (position starting from 0) of a substring within a given string.
     *
     * @param haystack The string to search within.
     * @param needle   The substring to search for.
     * @return The index of the first occurrence of needle within haystack, or -1 if needle is not found.
     */
    public int strStr(String haystack, String needle) {
        if (needle.isEmpty()) {
            return 0; // Empty needle is always found at position 0
        }

        return haystack.indexOf(needle);
    }


    public static void main(String[] args) {
        Solution solution = new Solution();

        String haystack1 = "hello";
        String needle1 = "ll";
        int result1 = solution.strStr(haystack1, needle1);
        System.out.println("First occurrence of \"" + needle1 + "\" in \"" + haystack1 + "\": " + result1); // Output: 2

        String haystack2 = "aaaaa";
        String needle2 = "bba";
        int result2 = solution.strStr(haystack2, needle2);
        System.out.println("First occurrence of \"" + needle2 + "\" in \"" + haystack2 + "\": " + result2); // Output: -1

        String haystack3 = "hello";
        String needle3 = ""; // Empty needle
        int result3 = solution.strStr(haystack3, needle3);
        System.out.println("First occurrence of \"" + needle3 + "\" in \"" + haystack3 + "\": " + result3); // Output: 0


        String haystack4 = ""; // Empty haystack
        String needle4 = "a"; 
        int result4 = solution.strStr(haystack4, needle4);
        System.out.println("First occurrence of \"" + needle4 + "\" in \"" + haystack4 + "\": " + result4); // Output: -1

    }
}
