                int b3 = b * b * b;
                int sum = a3 + b3;
                if (sum > n) break; // Optimization: Stop if the sum exceeds n

                    int c3 = c * c * c;
                    if (c3 > sum) break;

                    if (d * d * d == d3) {
                        count++;
                    }
