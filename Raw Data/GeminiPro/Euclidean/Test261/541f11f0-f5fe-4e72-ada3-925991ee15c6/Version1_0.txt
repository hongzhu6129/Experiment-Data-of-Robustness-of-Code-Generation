```java
import java.util.Arrays;

public class RemoveDuplicates {

    static int removeDuplicates(int[] nums) {
        if (nums == null || nums.length == 0) {
            return 0;
        }

        int uniqueCount = 1; // Start with 1 because the first element is always unique.

        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[uniqueCount] = nums[i]; // Overwrite duplicates with unique elements.
                uniqueCount++;
            }
        }

        return uniqueCount;
    }

    public static void main(String[] args) {
        int[] nums1 = {1, 1, 2};
        int newLength1 = removeDuplicates(nums1);
        System.out.println("New length: " + newLength1 + ", Array: " + Arrays.toString(Arrays.copyOfRange(nums1, 0, newLength1)));

        int[] nums2 = {0, 0, 1, 1, 1, 2, 2, 3, 3, 4};
        int newLength2 = removeDuplicates(nums2);
        System.out.println("New length: " + newLength2 + ", Array: " + Arrays.toString(Arrays.copyOfRange(nums2, 0, newLength2)));
        
        // Test case with empty array
        int[] nums3 = {};
        int newLength3 = removeDuplicates(nums3);
        System.out.println("New length: " + newLength3 + ", Array: " + Arrays.toString(Arrays.copyOfRange(nums3, 0, newLength3))); // Empty array output

         // Test case with null array
        int[] nums4 = null;
        int newLength4 = removeDuplicates(nums4);
        System.out.println("New length: " + newLength4); // 0 output
    }
}
```

- **Clearer variable names:** `uniqueCount` replaces `k` for better readability.
- **More comprehensive tests:** The `main` method includes tests with different input arrays, including empty and `null` arrays, to demonstrate the robustness of the code.
- **Prints the updated array:**  The `main` method now prints the updated array (with duplicates removed) using `Arrays.copyOfRange` to avoid showing the original, unaltered elements beyond the new length. This clarifies the effect of the `removeDuplicates` function.
