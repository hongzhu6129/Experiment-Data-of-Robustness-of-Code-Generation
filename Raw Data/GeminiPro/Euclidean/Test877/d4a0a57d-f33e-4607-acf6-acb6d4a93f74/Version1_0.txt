```java
public class StringCopies {

    public static void main(String[] args) {
        String originalString = "Hello"; // Example string, can be changed

        if (originalString.length() < 3) {
            System.out.println("The original string must be 3 or more characters long.");
            return; // Exit if the string is too short
        }

        String lastThreeChars = originalString.substring(originalString.length() - 3);
        String repeatedString = lastThreeChars + lastThreeChars + lastThreeChars + lastThreeChars; 

        System.out.println(repeatedString);
    }
}
```
