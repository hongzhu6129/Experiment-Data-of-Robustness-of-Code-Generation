```java
import java.util.ArrayList;
import java.util.List;

public class ColorList {

    public static void main(String[] args) {
        // Create an ArrayList to store strings (colors)
        List<String> colors = new ArrayList<>();

        // Add some colors to the list
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Orange");

        // Print out the collection (ArrayList)
        System.out.println(colors);


        //  Alternative way to print, iterating through the list:

        // Using a for-each loop (enhanced for loop):
        System.out.println("\nUsing for-each loop:");
        for (String color : colors) {
            System.out.println(color);
        }


        // Using a traditional for loop:
         System.out.println("\nUsing traditional for loop:");
        for (int i = 0; i < colors.size(); i++) {
            System.out.println(colors.get(i));
        }

         //Using a while loop:
         System.out.println("\nUsing while loop:");
         int i=0;
         while(i< colors.size()){
             System.out.println(colors.get(i));
             i++;
         }

    }
}
```
