import java.util.ArrayList;
import java.util.List;

public class SumEvenOdd {

    public static void main(String[] args) {
        // Example list of numbers
        List<Integer> numbers = new ArrayList<>();
        numbers.add(1);
        numbers.add(2);
        numbers.add(3);
        numbers.add(4);
        numbers.add(5);
        numbers.add(6);
        numbers.add(7);
        numbers.add(8);
        numbers.add(9);
        numbers.add(10);


        int sumEven = 0;
        int sumOdd = 0;

        // Iterate through the list and calculate sums
        for (int number : numbers) {
            if (number % 2 == 0) {
                sumEven += number;
            } else {
                sumOdd += number;
            }
        }

        // Print the results
        System.out.println("Sum of even numbers: " + sumEven);
        System.out.println("Sum of odd numbers: " + sumOdd);


         // Example with an empty list:
        List<Integer> emptyList = new ArrayList<>();
        sumEven = 0;
        sumOdd = 0;

        for (int number : emptyList) {
            if (number % 2 == 0) {
                sumEven += number;
            } else {
                sumOdd += number;
            }
        }
        System.out.println("Sum of even numbers (empty list): " + sumEven);
        System.out.println("Sum of odd numbers (empty list): " + sumOdd);



        // Example with a list containing only even numbers:
        List<Integer> evenList = new ArrayList<>();
        evenList.add(2);
        evenList.add(4);
        evenList.add(6);

        sumEven = 0;
        sumOdd = 0;

        for (int number : evenList) {
            if (number % 2 == 0) {
                sumEven += number;
            } else {
                sumOdd += number;
            }
        }
        System.out.println("Sum of even numbers (even list): " + sumEven);
        System.out.println("Sum of odd numbers (even list): " + sumOdd);



        // Example with a list containing only odd numbers:
        List<Integer> oddList = new ArrayList<>();
        oddList.add(1);
        oddList.add(3);
        oddList.add(5);

        sumEven = 0;
        sumOdd = 0;

        for (int number : oddList) {
            if (number % 2 == 0) {
                sumEven += number;
            } else {
                sumOdd += number;
            }
        }
        System.out.println("Sum of even numbers (odd list): " + sumEven);
        System.out.println("Sum of odd numbers (odd list): " + sumOdd);
    }
}
