```java
import java.util.PriorityQueue;

public class RemoveAllPriorityQueue {

    public static void main(String[] args) {
        // Create a PriorityQueue
        PriorityQueue<Integer> pq = new PriorityQueue<>();

        // Add elements to the PriorityQueue
        pq.add(10);
        pq.add(20);
        pq.add(15);
        pq.add(5);
        pq.add(30);


        System.out.println("Original PriorityQueue: " + pq);

        // Remove all elements using clear()
        pq.clear(); 

        System.out.println("PriorityQueue after removing all elements: " + pq);


       //Alternative way to remove all elements by repeatedly removing the head until it is empty.  Less efficient, but demonstrates another approach.
        PriorityQueue<Integer> pq2 = new PriorityQueue<>();

        // Add elements to the PriorityQueue
        pq2.add(10);
        pq2.add(20);
        pq2.add(15);
        pq2.add(5);
        pq2.add(30);


        System.out.println("Original PriorityQueue 2: " + pq2);

        while (!pq2.isEmpty()) {
            pq2.poll(); // Removes the head (smallest element)
        }

        System.out.println("PriorityQueue 2 after removing all elements by polling: " + pq2);

    }
}
```
