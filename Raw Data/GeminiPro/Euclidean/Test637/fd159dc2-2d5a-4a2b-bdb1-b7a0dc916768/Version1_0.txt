```java
import java.util.PriorityQueue;

public class PriorityQueueFirstElement {

    public static void main(String[] args) {
        // Create a PriorityQueue of integers
        PriorityQueue<Integer> pq = new PriorityQueue<>();

        // Add elements to the PriorityQueue
        pq.add(5);
        pq.add(2);
        pq.add(8);
        pq.add(1);
        pq.add(9);

        // Retrieve and print the first element (smallest element in this case)
        if (!pq.isEmpty()) {
            Integer firstElement = pq.peek(); 
            System.out.println("First element of the PriorityQueue: " + firstElement);
        } else {
            System.out.println("PriorityQueue is empty.");
        }
    }
}
```
