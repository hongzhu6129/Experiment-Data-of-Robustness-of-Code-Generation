```java
public class InsertWord {

    public static void main(String[] args) {
        String originalString = "Python";
        String wordToInsert = "Tutorial";

        String result = insertInTheMiddle(originalString, wordToInsert);
        System.out.println(result); // Output: PyTutorialthon

    }

    private static String insertInTheMiddle(String originalString, String wordToInsert) {
        int middleIndex = originalString.length() / 2;
        return originalString.substring(0, middleIndex) + wordToInsert + originalString.substring(middleIndex);

    }


}
```
