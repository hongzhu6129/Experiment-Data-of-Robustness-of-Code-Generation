```java
import java.util.TreeMap;
import java.util.Set;

public class TreeMapKeys {

    public static void main(String[] args) {

        // Create a TreeMap
        TreeMap<String, Integer> treeMap = new TreeMap<>();

        // Put some key-value pairs into the TreeMap
        treeMap.put("Apple", 1);
        treeMap.put("Banana", 2);
        treeMap.put("Orange", 3);
        treeMap.put("Grape", 4);

        // Get all keys using keySet()
        Set<String> keys = treeMap.keySet();

        // Print all keys
        System.out.println("Keys in the TreeMap:");
        for (String key : keys) {
            System.out.println(key);
        }


        // Alternatively, you can iterate directly using the keySet:
        System.out.println("\nKeys using direct iteration:");
        for(String key : treeMap.keySet()) {
            System.out.println(key);
        }
    }
}
```
