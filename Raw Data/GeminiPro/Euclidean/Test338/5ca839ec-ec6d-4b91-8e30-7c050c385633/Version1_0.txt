```java
public class LastIndexOfString {

    public static void main(String[] args) {

        // Test cases
        String mainString1 = "This is a test string.";
        String subString1 = "is";
        int lastIndex1 = mainString1.lastIndexOf(subString1);
        System.out.println("Last index of '" + subString1 + "' in '" + mainString1 + "': " + lastIndex1); // Output: 5

        String mainString2 = "Hello world, world!";
        String subString2 = "world";
        int lastIndex2 = mainString2.lastIndexOf(subString2);
        System.out.println("Last index of '" + subString2 + "' in '" + mainString2 + "': " + lastIndex2); // Output: 13


        String mainString3 = "Coding is fun, coding is challenging.";
        String subString3 = "coding";  // Case-sensitive!
        int lastIndex3 = mainString3.lastIndexOf(subString3);
        System.out.println("Last index of '" + subString3 + "' in '" + mainString3 + "': " + lastIndex3); // Output: 18


        String mainString4 = "abababa";
        String subString4 = "aba";
        int lastIndex4 = mainString4.lastIndexOf(subString4);
        System.out.println("Last index of '" + subString4 + "' in '" + mainString4 + "': " + lastIndex4); // Output: 4

        String mainString5 =  "The quick brown fox jumps over the lazy fox";
        String subString5 = "fox";
        int lastIndex5 = mainString5.lastIndexOf(subString5);
        System.out.println("Last index of '" + subString5 + "' in '" + mainString5 + "': " + lastIndex5); // Output: 40

        String mainString6 = "This string does not contain the substring";
        String subString6 = "notfound";
        int lastIndex6 = mainString6.lastIndexOf(subString6);
        System.out.println("Last index of '" + subString6 + "' in '" + mainString6 + "': " + lastIndex6); // Output: -1 (not found)



    }
}
```
