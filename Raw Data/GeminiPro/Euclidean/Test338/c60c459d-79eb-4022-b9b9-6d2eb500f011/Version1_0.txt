```java
public class StringIndex {

    public static void main(String[] args) {
        String mainString = "This is a test string.";
        String subString = "is";

        int index = mainString.indexOf(subString);

        if (index != -1) {
            System.out.println("The earlier index of \"" + subString + "\" is: " + index);
        } else {
            System.out.println("\"" + subString + "\" is not found in the main string.");
        }



        // Demonstrating multiple occurrences and finding the first one:
         mainString = "This is a test string is.";
         subString = "is";

         index = mainString.indexOf(subString);

        if (index != -1) {
            System.out.println("The earlier index of \"" + subString + "\" is: " + index);
        } else {
            System.out.println("\"" + subString + "\" is not found in the main string.");
        }



        // Demonstrating case sensitivity:
        mainString = "This Is a test string.";
        subString = "is"; // Lowercase "is"

        index = mainString.indexOf(subString);

        if (index != -1) {
            System.out.println("The earlier index of \"" + subString + "\" is: " + index);
        } else {
            System.out.println("\"" + subString + "\" is not found in the main string.");
        }


        // Demonstrating empty substring
        mainString = "This is a test string.";
        subString = ""; // Empty String

        index = mainString.indexOf(subString);

        if (index != -1) {
            System.out.println("The earlier index of \"" + subString + "\" is: " + index);
        } else {
            System.out.println("\"" + subString + "\" is not found in the main string.");
        }
    }
}
```
