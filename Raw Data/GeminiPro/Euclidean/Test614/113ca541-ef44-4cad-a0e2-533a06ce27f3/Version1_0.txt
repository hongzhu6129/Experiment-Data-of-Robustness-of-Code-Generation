```java
import java.util.TreeSet;

public class TreeColors {

    public static void main(String[] args) {
        // Create a TreeSet to store colors (strings). 
        // TreeSet automatically sorts elements in natural order (alphabetical for strings).
        TreeSet<String> tree = new TreeSet<>();

        // Add some colors
        tree.add("Red");
        tree.add("Green");
        tree.add("Blue");
        tree.add("Yellow");
        tree.add("Orange");

        // Print the TreeSet.  The elements will be sorted.
        System.out.println(tree); 
    }
}
```
