```java
import java.util.TreeSet;

public class TreeSetExample {

    public static void main(String[] args) {
        // Create a TreeSet to store colors (Strings).
        // TreeSet automatically stores elements in sorted order (natural ordering for Strings).
        TreeSet<String> colors = new TreeSet<>();

        // Add some colors to the TreeSet
        colors.add("Red");
        colors.add("Green");
        colors.add("Blue");
        colors.add("Yellow");
        colors.add("Orange");


        // Print the TreeSet (it will be sorted alphabetically)
        System.out.println("Colors in the TreeSet:");
        System.out.println(colors);


        // Alternatively, you can iterate through the TreeSet using a for-each loop:
        //System.out.println("Colors in the TreeSet (iterated):");
        //for (String color : colors) {
        //    System.out.println(color);
        //}

    }
}
```
