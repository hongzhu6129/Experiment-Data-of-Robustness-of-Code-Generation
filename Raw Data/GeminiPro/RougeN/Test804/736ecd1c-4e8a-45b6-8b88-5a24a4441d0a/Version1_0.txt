public class AmericanFlag {

    public static void main(String[] args) {
        // Print the striped section
        for (int i = 0; i < 6; i++) {
            // Print stars
            for (int j = 0; j < 6; j++) {
                System.out.print("* ");
            }
            System.out.print("==");
            for (int j = 0; j < 26; j++) {
                System.out.print("=");
            }
            System.out.println();

            // Print stripes (alternate blank line)
            if (i % 2 != 0) {
                for (int j = 0; j < 6; j++) {
                    System.out.print("* ");
                }
                System.out.print("==");
                for (int j = 0; j < 26; j++) {
                    System.out.print("=");
                }
                System.out.println();
            }


        }


        // Print the red stripes
        for (int i = 0; i < 6; i++) {
            System.out.print("==");
            for (int j = 0; j < 32; j++) {
                System.out.print("=");
            }
            System.out.println();
        }
    }
}
