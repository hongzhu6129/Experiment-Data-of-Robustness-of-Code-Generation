```java
import java.util.LinkedList;

public class LinkedListInsertAtEnd {

    public static void main(String[] args) {
        // Create a LinkedList
        LinkedList<String> linkedList = new LinkedList<>();

        // Add some initial elements
        linkedList.add("Apple");
        linkedList.add("Banana");
        linkedList.add("Orange");

        // Element to insert at the end
        String newElement = "Grape";

        // Insert at the end using add() - it automatically appends to the end
        linkedList.add(newElement); 

        // Print the updated LinkedList
        System.out.println("Updated LinkedList: " + linkedList);


        //  --- Example with Integer LinkedList ---
        LinkedList<Integer> numbers = new LinkedList<>();
        numbers.add(1);
        numbers.add(2);
        numbers.add(3);

        int newNumber = 4;
        numbers.add(newNumber);

        System.out.println("Updated Integer LinkedList: " + numbers);

    }
}
```
