public class AmericanFlag {

    public static void main(String[] args) {
        // Print the top part of the flag (stars and stripes)
        for (int i = 0; i < 5; i++) {
            // Print stars
            for (int j = 0; j < 6; j++) {
                System.out.print("* ");
            }
            System.out.print("==");
            // Print stripes
            for (int j = 0; j < 20; j++) {
                System.out.print("=");
            }
            System.out.println(); // New line
        }

        for (int i = 0; i < 4; i++) {
            // Print stars
            for (int j = 0; j < 5; j++) {
                System.out.print("* ");
            }
            System.out.print(" ==");

            for (int j = 0; j < 20; j++) {
                System.out.print("=");
            }

            System.out.println();
        }


        // Print the bottom part of the flag (stripes only)
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j < 30; j++) {
                System.out.print("=");
            }
            System.out.println(); // New line
        }
    }
}
