```java
import java.util.TreeSet;

public class TreeSetAddition {

    public static void main(String[] args) {
        // Create the first TreeSet
        TreeSet<Integer> set1 = new TreeSet<>();
        set1.add(1);
        set1.add(2);
        set1.add(3);
        set1.add(4);
        set1.add(5);

        // Create the second TreeSet
        TreeSet<Integer> set2 = new TreeSet<>();
        set2.add(6);
        set2.add(7);
        set2.add(8);
        set2.add(9);
        set2.add(10);


        // Add all elements of set2 to set1
        set1.addAll(set2);


        // Print the updated set1 containing all elements
        System.out.println("Combined TreeSet: " + set1);


        // Example with different data types (String)
        TreeSet<String> stringSet1 = new TreeSet<>();
        stringSet1.add("apple");
        stringSet1.add("banana");

        TreeSet<String> stringSet2 = new TreeSet<>();
        stringSet2.add("cherry");
        stringSet2.add("date");

        stringSet1.addAll(stringSet2);
        System.out.println("Combined String TreeSet: " + stringSet1);
    }
}
```
