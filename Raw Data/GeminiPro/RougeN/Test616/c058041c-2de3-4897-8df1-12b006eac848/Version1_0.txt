```java
import java.util.TreeSet;

public class AddTreeSetElements {

    public static void main(String[] args) {
        // Create the first TreeSet
        TreeSet<String> treeSet1 = new TreeSet<>();
        treeSet1.add("Apple");
        treeSet1.add("Banana");
        treeSet1.add("Cherry");

        // Create the second TreeSet
        TreeSet<String> treeSet2 = new TreeSet<>();
        treeSet2.add("Date");
        treeSet2.add("Elderberry");
        treeSet2.add("Fig");


        // Add all elements from treeSet1 to treeSet2
        treeSet2.addAll(treeSet1);

        // Print the combined TreeSet
        System.out.println("Combined TreeSet: " + treeSet2);


        //Example with Integers
        TreeSet<Integer> numbers1 = new TreeSet<>();
        numbers1.add(1);
        numbers1.add(2);
        numbers1.add(3);

        TreeSet<Integer> numbers2 = new TreeSet<>();
        numbers2.add(4);
        numbers2.add(5);
        numbers2.add(6);

        numbers2.addAll(numbers1);
        System.out.println("Combined Integer TreeSet: " + numbers2);
    }
}
```
