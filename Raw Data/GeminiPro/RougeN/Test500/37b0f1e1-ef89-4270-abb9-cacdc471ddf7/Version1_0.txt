class Solution {
    /**
     * Counts the number of digits equal to 2 in a non-negative integer.
     *
     * @param n The non-negative integer.
     * @return The count of digits equal to 2.
     */
    public static int test(int n) {
        int count = 0;
        String numStr = String.valueOf(n);

        for (int i = 0; i < numStr.length(); i++) {
            if (numStr.charAt(i) == '2') {
                count++;
            }
        }

        return count;
    }


    public static void main(String[] args) {
        // Example usage:
        int num1 = 222;
        int count1 = test(num1);
        System.out.println("Number of 2s in " + num1 + ": " + count1); // Output: 3

        int num2 = 12342;
        int count2 = test(num2);
        System.out.println("Number of 2s in " + num2 + ": " + count2); // Output: 2


        int num3 = 0;
        int count3 = test(num3);
        System.out.println("Number of 2s in " + num3 + ": " + count3); // Output: 0
    }
}
