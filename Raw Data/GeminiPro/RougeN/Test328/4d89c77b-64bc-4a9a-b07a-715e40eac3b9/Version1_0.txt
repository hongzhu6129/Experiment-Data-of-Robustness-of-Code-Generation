public class FictionalArrayString {

    public static void main(String[] args) {
        // Create a string representation of a fictional array
        String fictionalArray = "[ 'apple', 'banana', 123, true, 3.14 ]";

        // Print the string
        System.out.println(fictionalArray);


        //  More sophisticated representation (if needed)
        StringBuilder sb = new StringBuilder("[");
        String[] elements = {"apple", "banana", "123", "true", "3.14"};

        for (int i = 0; i < elements.length; i++) {
            sb.append(String.format("'%s'", elements[i])); // Add quotes around strings
            if (i < elements.length - 1) {
                sb.append(", ");
            }
        }
        sb.append("]");

        System.out.println(sb.toString());



    }
}
