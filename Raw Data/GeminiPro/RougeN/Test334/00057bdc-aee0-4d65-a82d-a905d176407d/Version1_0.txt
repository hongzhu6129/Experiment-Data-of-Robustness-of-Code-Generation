public class StringToArray {

    public static void main(String[] args) {
        String str = "This is a sample string.";

        // Method 1: Using toCharArray()
        char[] charArray1 = str.toCharArray();
        System.out.print("Character array using toCharArray(): [");
        for (int i = 0; i < charArray1.length; i++) {
            System.out.print("'" + charArray1[i] + "'");
            if (i < charArray1.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");


        // Method 2: Using charAt() and a loop
        char[] charArray2 = new char[str.length()];
        for (int i = 0; i < str.length(); i++) {
            charArray2[i] = str.charAt(i);
        }

        System.out.print("Character array using charAt(): [");
        for (int i = 0; i < charArray2.length; i++) {
            System.out.print("'" + charArray2[i] + "'");
            if (i < charArray2.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");


        //Example with another String (including special characters and spaces)
        String str2 = "Hello, World! 123";
        char[] charArray3 = str2.toCharArray();

        System.out.print("Character array of 'Hello, World! 123': [");
        for (int i = 0; i < charArray3.length; i++) {
            System.out.print("'" + charArray3[i] + "'");
            if (i < charArray3.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");


    }
}
