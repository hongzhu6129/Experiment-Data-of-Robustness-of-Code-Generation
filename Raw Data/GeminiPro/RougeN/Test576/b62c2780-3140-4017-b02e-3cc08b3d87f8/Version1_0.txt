```java
import java.util.LinkedList;

public class AppendToLinkedList {

    public static void main(String[] args) {
        // Create a LinkedList of Strings
        LinkedList<String> linkedList = new LinkedList<>();

        // Add some initial elements
        linkedList.add("Apple");
        linkedList.add("Banana");
        linkedList.add("Cherry");

        // Element to append
        String newElement = "Date";

        // Append the element to the end of the LinkedList
        linkedList.add(newElement); //  Or linkedList.addLast(newElement);  - both do the same thing

        // Print the updated LinkedList
        System.out.println("Updated LinkedList: " + linkedList);


        // Example with integers:
        LinkedList<Integer> numbers = new LinkedList<>();
        numbers.add(10);
        numbers.add(20);
        numbers.add(30);

        int newNumber = 40;
        numbers.addLast(newNumber); // Or numbers.add(newNumber);

        System.out.println("Updated LinkedList of numbers: " + numbers);



    }
}
```
