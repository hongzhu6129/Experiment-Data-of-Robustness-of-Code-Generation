```java
import java.util.LinkedList;

public class RemoveFirstLinkedList {

    public static void main(String[] args) {
        // Create a LinkedList
        LinkedList<String> linkedList = new LinkedList<>();
        linkedList.add("Apple");
        linkedList.add("Banana");
        linkedList.add("Orange");
        linkedList.add("Grape");

        System.out.println("Original LinkedList: " + linkedList);

        // Discard and return the first element 
        String removedElement = discardAndReturnFirst(linkedList);


        System.out.println("Removed element: " + removedElement);
        System.out.println("LinkedList after removal: " + linkedList);



        // Example with an empty list:
        LinkedList<String> emptyList = new LinkedList<>();
        String removedFromEmpty = discardAndReturnFirst(emptyList);
        System.out.println("Removed from empty list: " + removedFromEmpty); // Should be null
        System.out.println("Empty list after removal: " + emptyList); // Should still be empty
    }


    // Method to remove and return the first element of a LinkedList
    public static <T> T discardAndReturnFirst(LinkedList<T> list) {
        if (list == null || list.isEmpty()) {
            return null; // Return null if the list is null or empty
        } else {
            return list.removeFirst(); // Remove and return the first element
        }
    }
}
```
