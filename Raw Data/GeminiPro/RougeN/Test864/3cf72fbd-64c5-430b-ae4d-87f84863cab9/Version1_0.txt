    public static double calculatePi(int numTerms) {
        double sum = 1.0; // Initialize sum to 1.0 (not 1)
        int denominator = 3;
        int sign = -1;

        for (int i = 2; i <= numTerms; i++) {
            
            if (i % 2 == 0) { // Handle the + 1/9 case
                sum += (1.0 / denominator) * sign;
            } else {
                sum += (1.0 / denominator) * sign;
            }

            if (i%2 == 0 )

                denominator += 2;
            }

            sign *= -1; // Alternate the sign
        }
