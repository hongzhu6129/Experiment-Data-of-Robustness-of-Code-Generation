import java.util.TreeSet;

public class LeastKeyGreaterThanOrEqualTo {

    public static void main(String[] args) {
        // Example usage:
        TreeSet<Integer> treeSet = new TreeSet<>();
        treeSet.add(10);
        treeSet.add(20);
        treeSet.add(30);
        treeSet.add(40);

        System.out.println("Least key greater than or equal to " + key2 + ": " + result2); // Output: 10

        Integer result3 = getLeastKeyGreaterThanOrEqualTo(treeSet, key3);

        stringTreeSet.add("banana");
        stringTreeSet.add("cherry");

        System.out.println("Least key greater than or equal to " + key4 + ": " + result4); // Output: cherry

    public static <T> T getLeastKeyGreaterThanOrEqualTo(TreeSet<T> treeSet, T key) {
        if (treeSet == null || treeSet.isEmpty()) {
            return null;
        }
         return treeSet.ceiling(key); 
    }
}
