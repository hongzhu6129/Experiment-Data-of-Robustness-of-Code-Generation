```java
import java.util.HashMap;
import java.util.Map;

public class CheckValueInMap {

    public static void main(String[] args) {
        // Create a sample map
        Map<String, Integer> map = new HashMap<>();
        map.put("apple", 1);
        map.put("banana", 2);
        map.put("orange", 3);

        // Value to search for
        int valueToSearch = 2;

        // Check if the map contains the value
        boolean containsValue = map.containsValue(valueToSearch);

        // Print the result
        if (containsValue) {
            System.out.println("The map contains the value " + valueToSearch);
        } else {
            System.out.println("The map does not contain the value " + valueToSearch);
        }


        // Another Example with a different value.
         valueToSearch = 5; //This value is not in the map.

        containsValue = map.containsValue(valueToSearch);

        if (containsValue) {
            System.out.println("The map contains the value " + valueToSearch);
        } else {
            System.out.println("The map does not contain the value " + valueToSearch);
        }



        // Example demonstrating null handling
        map.put("grape", null);  //Adding a key with a null value.

        valueToSearch = 42; //Arbitrary Value

        containsValue = map.containsValue(valueToSearch); //false
        System.out.println("Contains " + valueToSearch + ": " + containsValue);

        containsValue = map.containsValue(null); //true
        System.out.println("Contains null: " + containsValue);
    }
}
```
