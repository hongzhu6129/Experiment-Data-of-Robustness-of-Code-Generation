```java
public class TrailingZeros {

    public static void main(String[] args) {
        int n = 7; // Example input, you can change this value

        int count = 0;
        for (int i = 5; n / i >= 1; i *= 5) {
            count += n / i;
        }

        System.out.println(count); 
    }
}
```
