```java
import java.util.PriorityQueue;

public class RetrieveFirstElementPriorityQueue {

    public static void main(String[] args) {
        // Create a PriorityQueue of Integers (you can use any comparable type)
        PriorityQueue<Integer> priorityQueue = new PriorityQueue<>();

        // Add some elements to the priority queue
        priorityQueue.add(5);
        priorityQueue.add(1);
        priorityQueue.add(8);
        priorityQueue.add(3);

        // Retrieve the first element (head/highest priority) without removing it.
        if (!priorityQueue.isEmpty()) {
            Integer firstElement = priorityQueue.peek();
            System.out.println("First element (highest priority): " + firstElement); 
        } else {
            System.out.println("Priority queue is empty.");
        }


        // Example of retrieving and removing the first element (poll)
        if (!priorityQueue.isEmpty()) {
             Integer removedElement = priorityQueue.poll();
             System.out.println("Removed first element: " + removedElement);
         } else {
             System.out.println("Priority queue is empty.");
         }


        System.out.println("Remaining elements in the queue: " + priorityQueue);

        /*  Remember:
         *    - peek(): Retrieves the head of the queue (highest priority) without removing it. Returns null if the queue is empty.
         *    - poll(): Retrieves and removes the head of the queue. Returns null if the queue is empty.
         */


    }
}
```
