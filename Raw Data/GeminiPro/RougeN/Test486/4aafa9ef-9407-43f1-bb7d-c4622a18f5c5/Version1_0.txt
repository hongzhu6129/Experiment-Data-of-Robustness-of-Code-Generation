```java
public class MiddleCharacter {

    public static void main(String[] args) {
        System.out.println(middle("hello")); // l
        System.out.println(middle("java")); // av
        System.out.println(middle("a")); // a
        System.out.println(middle("")); // (empty string)
        System.out.println(middle("coding")); // di
    }

    public static String middle(String str) {
        int len = str.length();
        if (len == 0) {
            return "";
        }

        if (len % 2 == 0) { // Even length
            int midIndex = len / 2;
            return str.substring(midIndex - 1, midIndex + 1);
        } else { // Odd length
            int midIndex = len / 2;
            return str.substring(midIndex, midIndex + 1);
        }
    }
}
```
