```java
import java.util.TreeSet;

public class TreeSetSize {

    public static void main(String[] args) {
        // Create a TreeSet
        TreeSet<String> treeSet = new TreeSet<>();

        // Add elements to the TreeSet
        treeSet.add("Apple");
        treeSet.add("Banana");
        treeSet.add("Orange");
        treeSet.add("Grape");
        treeSet.add("Mango");

        // Get the number of elements in the TreeSet
        int size = treeSet.size();

        // Print the size
        System.out.println("Number of elements in the TreeSet: " + size);


        // Example with Integer TreeSet
        TreeSet<Integer> numbers = new TreeSet<>();
        numbers.add(5);
        numbers.add(2);
        numbers.add(8);
        numbers.add(1);

        System.out.println("Number of elements in the Integer TreeSet: " + numbers.size());

        // Example with an empty treeset
        TreeSet<Double> emptySet = new TreeSet<>();
        System.out.println("Number of elements in empty TreeSet: " + emptySet.size());



    }
}
```
