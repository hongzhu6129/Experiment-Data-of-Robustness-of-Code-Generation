```java
import java.util.HashSet;

public class EmptyHashSet {

    public static void main(String[] args) {
        // Create a HashSet
        HashSet<String> hashSet = new HashSet<>();

        // Add some elements to the HashSet
        hashSet.add("Apple");
        hashSet.add("Banana");
        hashSet.add("Orange");

        // Print the initial HashSet
        System.out.println("Initial HashSet: " + hashSet);

        // Empty the HashSet using the clear() method
        hashSet.clear();

        // Print the HashSet after clearing
        System.out.println("HashSet after clearing: " + hashSet); 
    }
}
```
