class Solution {
    /**
     * Sorts the given integer array using the Bubble Sort algorithm.
     *
     * @param arr The integer array to be sorted.
     */
    void bubbleSort(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - i - 1; j++) {
                if (arr[j] > arr[j + 1]) {
                    // Swap arr[j] and arr[j+1]
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                }
            }
        }
    }



    public static void main(String[] args) {
        Solution solution = new Solution();

        int[] arr1 = {64, 34, 25, 12, 22, 11, 90};
        solution.bubbleSort(arr1);
        System.out.print("Sorted array: ");
        for (int num : arr1) {
            System.out.print(num + " ");
        }
        System.out.println();


        int[] arr2 = {5, 1, 4, 2, 8};
        solution.bubbleSort(arr2);
        System.out.print("Sorted array: ");
        for (int num : arr2) {
            System.out.print(num + " ");
        }
        System.out.println();


        // Test case with an already sorted array
        int[] arr3 = {1, 2, 3, 4, 5};
        solution.bubbleSort(arr3);
        System.out.print("Sorted array: ");
        for (int num : arr3) {
            System.out.print(num + " ");
        }
        System.out.println();


         // Test case with a reverse sorted array
        int[] arr4 = {5,4,3,2,1};
        solution.bubbleSort(arr4);
        System.out.print("Sorted array: ");
        for (int num : arr4) {
            System.out.print(num + " ");
        }
        System.out.println();
    }
}
