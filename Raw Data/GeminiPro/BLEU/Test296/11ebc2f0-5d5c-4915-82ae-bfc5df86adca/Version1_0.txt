import java.util.Arrays;
import java.util.PriorityQueue;

public class PLargestElements {

    public static void main(String[] args) {
        int[] arr = {1, 23, 12, 9, 30, 2, 50};
        int p = 3;
        findPLargest(arr, p);

        arr = new int[]{1, 23, 12, 9, 30, 2, 50, 100, 1, 3, 5, 6, 7, 7, 8, 10, 11, 23, 34, 55, 45, 67, 78, 1, 34, 56, 7, 8, 9, 0, 43, 23, 5, 5, 6, 7, 8, 8};
        p = 5;
        findPLargest(arr, p);

        //Edge cases
        arr = new int[]{1};
        p = 1;
        findPLargest(arr, p);
        
        arr = new int[]{1};
        p = 0; // p cannot be negative or greater than array length
        findPLargest(arr, p);
        
        arr = new int[]{1, 2, 3};
        p = 4;
        findPLargest(arr, p);

        arr = new int[]{};
        p = 1;
        findPLargest(arr, p);

    }

    public static void findPLargest(int[] arr, int p) {
        if (arr == null || p < 0 || p > arr.length) {
            System.out.println("Invalid input");
            return;
        }
        if(arr.length == 0 && p > 0) {
            System.out.println("Invalid input");
            return;
        }



        PriorityQueue<Integer> minHeap = new PriorityQueue<>();

        for (int num : arr) {
            minHeap.offer(num);
            if (minHeap.size() > p) {
                minHeap.poll(); //Remove smallest element if heap size exceeds 'p'
            }
        }

        // Print the 'p' largest elements (which are now in the minHeap)
        Integer[] result = minHeap.toArray(new Integer[0]);
        Arrays.sort(result); // For consistent output order, sort the result.

        System.out.print("[");
        for(int i = 0; i < result.length; i++) {
            System.out.print(result[i]);
            if(i < result.length - 1) {
                System.out.print(", ");
            }

        }
        System.out.println("]");
    }
}
