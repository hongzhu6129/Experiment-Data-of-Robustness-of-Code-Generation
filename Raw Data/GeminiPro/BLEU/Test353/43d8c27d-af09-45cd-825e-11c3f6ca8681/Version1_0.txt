```java
import java.util.List;
import java.util.Arrays;

public class CharacterList {

    public static void main(String[] args) {
        String word = "Hello";
        List<Character> characters = Arrays.asList(word.toCharArray());
        System.out.println(characters);
    }
}
```

Test cases:

* Empty string: "" -> []
* Single character string: "a" -> ['a']
* Multi-character string: "Hello" -> ['H', 'e', 'l', 'l', 'o']
* String with spaces: "Hello World" -> ['H', 'e', 'l', 'l', 'o', ' ', 'W', 'o', 'r', 'l', 'd']
* String with special characters: "Hello!@#$%" -> ['H', 'e', 'l', 'l', 'o', '!', '@', '#', '$', '%']
