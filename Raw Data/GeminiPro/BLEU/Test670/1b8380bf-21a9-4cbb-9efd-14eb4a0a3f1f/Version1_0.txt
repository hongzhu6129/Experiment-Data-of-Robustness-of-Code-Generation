import java.util.TreeMap;

public class GreatestKeyLessThan {

    public static void main(String[] args) {
        // Example usage:
        TreeMap<Integer, String> treeMap = new TreeMap<>();
        treeMap.put(10, "Ten");
        treeMap.put(20, "Twenty");
        treeMap.put(30, "Thirty");
        treeMap.put(40, "Forty");
        treeMap.put(50, "Fifty");

        Integer key1 = 35;
        Integer key2 = 15;
        Integer key3 = 5;

        Integer result1 = getGreatestKeyLessThan(treeMap, key1);
        Integer result2 = getGreatestKeyLessThan(treeMap, key2);
        Integer result3 = getGreatestKeyLessThan(treeMap, key3);


        System.out.println("Greatest key less than " + key1 + ": " + result1); // Output: 30
        System.out.println("Greatest key less than " + key2 + ": " + result2); // Output: 10
        System.out.println("Greatest key less than " + key3 + ": " + result3); // Output: null


        // Additional Test cases for more comprehensive validation
         TreeMap<Integer, String> emptyTreeMap = new TreeMap<>();
        Integer emptyResult = getGreatestKeyLessThan(emptyTreeMap, 25);
        System.out.println("Greatest key less than 25 in empty map: " + emptyResult); // Output: null


        TreeMap<Integer, String> singleTreeMap = new TreeMap<>();
         singleTreeMap.put(100, "Hundred");
        Integer singleResult1 = getGreatestKeyLessThan(singleTreeMap, 150);
        Integer singleResult2 = getGreatestKeyLessThan(singleTreeMap, 50);


        System.out.println("Greatest key less than 150 in single map: " + singleResult1);//100
        System.out.println("Greatest key less than 50 in single map: " + singleResult2); // null




    }

    public static <K, V> K getGreatestKeyLessThan(TreeMap<K, V> treeMap, K key) {
        if (treeMap == null || treeMap.isEmpty()) {
            return null;
        }

        return treeMap.lowerKey(key); 
    }
}
