```java
public class StringContains {

    public static boolean is_str_contains(String str1, String str2) {
        return str2.contains(str1);
    }

    public static void main(String[] args) {
        String str1 = "Once in a blue moon";
        String str2 = "See eye to eye";

        boolean contains = is_str_contains(str1, str2);
        System.out.println("If the second string contains the first one? " + contains);


        str1 = "blue";
        str2 = "Once in a blue moon";
        contains = is_str_contains(str1, str2);
        System.out.println("If the second string contains the first one? " + contains);
    }
}
```
