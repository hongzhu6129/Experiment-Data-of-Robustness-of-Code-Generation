```java
import java.util.ArrayList;
import java.util.List;

public class PrintArrayListElementsWithPosition {

    public static void main(String[] args) {
        // Create an ArrayList of strings
        List<String> myArrayList = new ArrayList<>();
        myArrayList.add("Apple");
        myArrayList.add("Banana");
        myArrayList.add("Orange");
        myArrayList.add("Grape");
        myArrayList.add("Mango");


        // Print elements with their positions using a for loop and get()
        System.out.println("Using traditional for loop:");
        for (int i = 0; i < myArrayList.size(); i++) {
            System.out.println("Element at position " + i + ": " + myArrayList.get(i));
        }


         // Print elements with their position using enhanced for loop and a separate counter
        System.out.println("\nUsing enhanced for loop with counter:");
        int counter = 0;
        for (String fruit : myArrayList) {
            System.out.println("Element at position " + counter + ": " + fruit);
            counter++;
        }
    }
}
```
