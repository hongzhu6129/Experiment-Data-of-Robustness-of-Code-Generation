```java
import java.util.ArrayList;
import java.util.List;

public class ReplaceSecondElement {

    public static void main(String[] args) {
        // Create an ArrayList
        List<String> myList = new ArrayList<>();
        myList.add("apple");
        myList.add("banana"); // Element to be replaced
        myList.add("orange");
        myList.add("grape");


        // Specified element to replace with
        String newElement = "mango";

        // Check if the list has at least two elements 
        if (myList.size() >= 2) {
            // Replace the second element (index 1)
            myList.set(1, newElement); 

            // Print the updated ArrayList
            System.out.println("Updated ArrayList: " + myList);
        } else {
            System.out.println("The ArrayList does not have a second element to replace.");
        }



         // Example with an Integer ArrayList
        List<Integer> numList = new ArrayList<>();
        numList.add(10);
        numList.add(20);
        numList.add(30);

        Integer newNum = 25;

        if (numList.size() >= 2) {
             numList.set(1, newNum);
             System.out.println("Updated Integer ArrayList: " + numList);
        } else {
            System.out.println("The Integer ArrayList does not have a second element to replace.");
        }
    }
}
```
