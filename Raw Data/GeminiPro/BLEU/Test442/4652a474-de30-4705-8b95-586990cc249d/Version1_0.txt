```java
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;

public class CurrentTimeInDC {

    public static void main(String[] args) {
        // Get the current time in Washington, D.C. (Eastern Time)
        ZoneId dcZoneId = ZoneId.of("America/New_York"); //  "America/New_York" for Eastern Time
        ZonedDateTime nowDC = ZonedDateTime.now(dcZoneId);


        // Format the time for output  (optional, but makes it more readable)
        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss z");
        String formattedDateTime = nowDC.format(formatter);


        // Print the current time
        System.out.println("Current time in Washington, D.C.: " + formattedDateTime); 

    }
}
```
