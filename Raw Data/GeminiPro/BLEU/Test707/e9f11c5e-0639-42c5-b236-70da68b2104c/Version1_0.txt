public class ExcelColumnName {

    public static String Column(int n) {
        StringBuilder columnName = new StringBuilder();

        while (n > 0) {
            int rem = n % 26;

            if (rem == 0) {
                columnName.insert(0, 'Z');
                n = (n / 26) - 1;
            } else {
                columnName.insert(0, (char)('A' + rem - 1));
                n = n / 26;
            }
        }

        return columnName.toString();
    }

    public static void main(String[] args) {
        System.out.println("Column name for 1: " + Column(1)); // A
        System.out.println("Column name for 26: " + Column(26)); // Z
        System.out.println("Column name for 27: " + Column(27)); // AA
        System.out.println("Column name for 52: " + Column(52)); // AZ
        System.out.println("Column name for 701: " + Column(701)); // ZY
        System.out.println("Column name for 702: " + Column(702)); // ZZ
        System.out.println("Column name for 703: " + Column(703)); // AAA
        System.out.println("Column name for 16384: " + Column(16384)); // XFD

    }
}
