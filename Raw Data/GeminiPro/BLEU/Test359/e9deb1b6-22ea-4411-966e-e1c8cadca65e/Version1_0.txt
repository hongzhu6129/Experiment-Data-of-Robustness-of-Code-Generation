import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class WordCharacterChecker {

    static void checkExistance(String word, String item) {
        char[] wordChars = word.toCharArray();
        Arrays.sort(wordChars); // Sort for efficient checking

        char[] itemChars = item.toCharArray();
        Arrays.sort(itemChars); // Sort item characters as well

        int wordIndex = 0;
        for (int itemIndex = 0; itemIndex < itemChars.length && wordIndex < wordChars.length; itemIndex++) {
            if (itemChars[itemIndex] == wordChars[wordIndex]) {
                wordIndex++;
            } else if (itemChars[itemIndex] > wordChars[wordIndex]) {
                // If item char is greater, the word char might appear later in item
                continue;
            }
        }

        if (wordIndex == wordChars.length) {
            System.out.println(item); // Print the item if all word characters are present
        }
    }

    public static void main(String[] args) {
        String word = "sun";
        List<String> items = Arrays.asList("sunday", "unsung", "suns", "nut", "son", "sonar");

        System.out.println("Items containing all characters of '" + word + "':");

        for (String item : items) {
            checkExistance(word, item);
        }
    }
}
