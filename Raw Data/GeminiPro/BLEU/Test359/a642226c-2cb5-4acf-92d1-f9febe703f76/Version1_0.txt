import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class WordChecker {

    static void checkExistance(String word, String item) {
        char[] wordChars = word.toCharArray();
        Arrays.sort(wordChars);
        char[] itemChars = item.toLowerCase().toCharArray(); // Case-insensitive check
        Arrays.sort(itemChars);

        int i = 0, j = 0;

        while (i < wordChars.length && j < itemChars.length) {
            if (wordChars[i] == itemChars[j]) {
                i++; // Move to the next character in the word
            }
            j++; // Always move to the next character in the item
        }

        if (i == wordChars.length) { // All characters of the word found in the item
            System.out.println(item);
        }
    }

    public static void main(String[] args) {
        String word = "sun"; // Example word
        List<String> items = new ArrayList<>(Arrays.asList("sunday", "Unsung", "another", "sound", "noSuchWord")); // Sample items

        System.out.println("Items containing all characters of '" + word + "':");

        for (String item : items) {
            checkExistance(word, item);
        }

    }
}
