```java
import java.util.LinkedList;

public class LinkedListInsertEnd {

    public static void main(String[] args) {
        // Create a LinkedList of Strings
        LinkedList<String> books = new LinkedList<>();

        // Add some initial elements
        books.add("The Hitchhiker's Guide to the Galaxy");
        books.add("Pride and Prejudice");
        books.add("1984");

        // Element to be inserted at the end
        String newBook = "To Kill a Mockingbird";

        // Insert the element at the end using add() 
        books.add(newBook);


        // Print the updated LinkedList
        System.out.println("Updated LinkedList: " + books);


        // Example with Integer LinkedList
        LinkedList<Integer> numbers = new LinkedList<>();
        numbers.add(10);
        numbers.add(20);
        numbers.add(30);

        int newNumber = 40;
        numbers.add(newNumber);  // or numbers.addLast(newNumber); which is equivalent

        System.out.println("Updated Integer LinkedList: " + numbers);
    }
}
```
