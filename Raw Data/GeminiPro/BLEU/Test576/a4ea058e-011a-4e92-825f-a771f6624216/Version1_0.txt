```java
import java.util.LinkedList;

public class AppendToLinkedList {

    public static void main(String[] args) {
        // Create a LinkedList of strings (names)
        LinkedList<String> names = new LinkedList<>();

        // Add some initial names
        names.add("Alice");
        names.add("Bob");
        names.add("Charlie");

        // Element to append
        String new_name = "David";


        // Append the new name to the end of the list. 
        names.add(new_name); // Or names.addLast(new_name) - both are equivalent


        // Print the updated LinkedList to verify
        System.out.println("Updated LinkedList: " + names); 
    }
}
```
