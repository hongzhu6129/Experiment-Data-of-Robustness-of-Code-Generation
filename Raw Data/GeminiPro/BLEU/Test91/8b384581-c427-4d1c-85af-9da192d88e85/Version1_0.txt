```java
public class CheckArray {

    public static void main(String[] args) {
        int[] numbers1 = {1, 2, 3, 4, 5};
        int[] numbers2 = {10, 20, 30, 40, 50};
        int[] numbers3 = {1, 2, 10, 4, 30};

        System.out.println("Array numbers1 contains 10 or 30: " + contains10or30(numbers1));
        System.out.println("Array numbers2 contains 10 or 30: " + contains10or30(numbers2));
        System.out.println("Array numbers3 contains 10 or 30: " + contains10or30(numbers3));
    }

    private static boolean contains10or30(int[] arr) {
        for (int num : arr) {
            if (num == 10 || num == 30) {
                return true;
            }
        }
        return false;
    }
}
```
