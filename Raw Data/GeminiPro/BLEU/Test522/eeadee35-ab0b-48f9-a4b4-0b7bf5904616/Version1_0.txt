    private static int sumOfProperDivisors(int num) {
        int sum = 0;
        for (int i = 1; i < num; i++) {
            if (num % i == 0) {
                sum += i;
            }
        }
        return sum;
    }

    private static boolean isAmicable(int num1, int num2) {
        return sumOfProperDivisors(num1) == num2 && sumOfProperDivisors(num2) == num1;
    }
