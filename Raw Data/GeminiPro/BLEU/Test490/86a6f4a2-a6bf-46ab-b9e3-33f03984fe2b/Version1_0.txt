public class PentagonalNumbers {

    public static void main(String[] args) {
        // Display the first 50 pentagonal numbers
        final int NUM_PENTAGONAL_NUMBERS = 50;

        for (int i = 1; i <= NUM_PENTAGONAL_NUMBERS; i++) {
            System.out.printf("%7d ", getPentagonalNumber(i)); // Use printf for aligned output
            if (i % 10 == 0) { // Add a newline after every 10 numbers
                System.out.println(); 
            }
        }
    }

    public static int getPentagonalNumber(int n) {
        // Formula for nth pentagonal number: n * (3n - 1) / 2
        return n * (3 * n - 1) / 2;
    }
}
