```java
public class Main {
    public static void main(String[] args) {
        System.out.println("a\ta^2\ta^3\ta^4"); // Print the header

        for (int a = 1; a <= 4; a++) {
            System.out.print(a + "\t"); // Print the value of 'a'
            System.out.print((int) Math.pow(a, 2) + "\t"); // Print a^2
            System.out.print((int) Math.pow(a, 3) + "\t"); // Print a^3
            System.out.print((int) Math.pow(a, 4) + "\n"); // Print a^4 and a newline
        }
    }
}
```
