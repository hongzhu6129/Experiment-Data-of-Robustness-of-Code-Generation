import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class WordCounter {

    public int endWithmOrn(String text) {
        if (text == null || text.trim().isEmpty()) {
            return 0;
        }

        // Use regex to find words ending with m or n (case-insensitive)
        Pattern pattern = Pattern.compile("\\b\\w+[mnMN]\\b"); // \b for word boundaries
        Matcher matcher = pattern.matcher(text);

        int count = 0;
        while (matcher.find()) {
            count++;
        }

        return count;
    }

    public static void main(String[] args) {
        WordCounter counter = new WordCounter();
        String text1 = "The Man ran to the moon.";
        int count1 = counter.endWithmOrn(text1);
        System.out.println("Text 1 Count: " + count1); // Output: 2 (Man, moon)


        String text2 = "Women in STEM.";
        int count2 = counter.endWithmOrn(text2);
        System.out.println("Text 2 Count: " + count2); // Output: 2 (Women, STEM)

        String text3 = ""; // Empty string
        int count3 = counter.endWithmOrn(text3);
        System.out.println("Text 3 Count: " + count3); // Output: 0

        String text4 = null; // Null string
        int count4 = counter.endWithmOrn(text4);
        System.out.println("Text 4 Count: " + count4); // Output: 0

         String text5 = "m"; // single letter
        int count5 = counter.endWithmOrn(text5);
        System.out.println("Text 5 Count: " + count5); // Output: 1
    }
}
