```java
public class LucasNumbers {

    public static void main(String[] args) {
        int n = 10; // Number of Lucas numbers to display
        int a = 2; // First Lucas number
        int b = 1; // Second Lucas number

        System.out.print("First 10 Lucas numbers: ");

        if (n >= 1) {
            System.out.print(a + " "); // Print the first Lucas number
            n--;
        }

        if (n >= 1) {
            System.out.print(b + " "); // Print the second Lucas number
            n--;
        }
        
        for (int i = 0; i < n; i++) {
            int c = a + b;
            System.out.print(c + " ");
            a = b;
            b = c;
        }
        System.out.println(); // Add a newline at the end of the output

    }
}
```
