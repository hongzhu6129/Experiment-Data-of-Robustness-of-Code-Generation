import java.util.TreeMap;

public class TreeMapSearch {

    public static void main(String[] args) {
        // Create a TreeMap
        TreeMap<Integer, String> treeMap = new TreeMap<>();

        // Populate the TreeMap
        treeMap.put(1, "One");
        treeMap.put(2, "Two");
        treeMap.put(3, "Three");
        treeMap.put(4, "Four");
        treeMap.put(5, "Five");

        // Value to search for
        int searchKey = 3;

        // Search for the value
        if (treeMap.containsKey(searchKey)) {
            String value = treeMap.get(searchKey);
            System.out.println("Value found for key " + searchKey + ": " + value);
        } else {
            System.out.println("Value not found for key " + searchKey);
        }


        // Search for a non-existent key
        searchKey = 6;  // Key not present

         if (treeMap.containsKey(searchKey)) {
            String value = treeMap.get(searchKey);
            System.out.println("Value found for key " + searchKey + ": " + value);
        } else {
            System.out.println("Value not found for key " + searchKey);
        }
    }
}
