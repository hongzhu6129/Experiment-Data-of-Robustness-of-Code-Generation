public class Check20s {

    public static void main(String[] args) {
        int[] arr1 = {20, 10, 20, 30, 20}; // Valid
        int[] arr2 = {20, 20, 10, 20}; // Invalid (two 20's together)
        int[] arr3 = {20, 10, 20, 20, 30}; // Invalid (two 20's together)
        int[] arr4 = {20, 10, 30, 20}; // Invalid (only two 20s)
        int[] arr5 = {20, 20, 20, 20, 10}; // Invalid (too many 20s and together)


        System.out.println("arr1: " + check20s(arr1)); // Output: true
        System.out.println("arr2: " + check20s(arr2)); // Output: false
        System.out.println("arr3: " + check20s(arr3)); // Output: false
        System.out.println("arr4: " + check20s(arr4)); // Output: false
        System.out.println("arr5: " + check20s(arr5)); // Output: false


    }

    public static boolean check20s(int[] arr) {
        int count = 0;
        boolean adjacent = false;

        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == 20) {
                count++;
                if (i > 0 && arr[i - 1] == 20) {
                    adjacent = true;
                    break; // No need to continue if we find adjacent 20s
                }
            }
        }

        return count == 3 && !adjacent;
    }
}
